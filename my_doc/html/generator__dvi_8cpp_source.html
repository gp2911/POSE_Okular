173 QLinkedList<Okular::ObjectRect*> DviGenerator::generateDviLinks( const dviPageInfo *pageInfo )
174 {
175  QLinkedList<Okular::ObjectRect*> dviLinks;
176 
177  int pageWidth = pageInfo->width, pageHeight = pageInfo->height;
178 
179  foreach( const Hyperlink &dviLink, pageInfo->hyperLinkList )
180  {
181  QRect boxArea = dviLink.box;
182  double nl = (double)boxArea.left() / pageWidth,
183  nt = (double)boxArea.top() / pageHeight,
184  nr = (double)boxArea.right() / pageWidth,
185  nb = (double)boxArea.bottom() / pageHeight;
186 
187  QString linkText = dviLink.linkText;
188  if ( linkText.startsWith("#") )
189  linkText = linkText.mid( 1 );
190  Anchor anch = m_dviRenderer->findAnchor( linkText );
191 
192  Okular::Action *okuLink = 0;
193 
194  /* distinguish between local (-> anchor) and remote links */
195  if (anch.isValid())
196  {
197  /* internal link */
198  Okular::DocumentViewport vp;
199  fillViewportFromAnchor( vp, anch, pageWidth, pageHeight );
200 
201  okuLink = new Okular::GotoAction( "", vp );
202  }
203  else
204  {
205  okuLink = new Okular::BrowseAction( dviLink.linkText );
206  }
207  if ( okuLink )
208  {
209  Okular::ObjectRect *orlink = new Okular::ObjectRect( nl, nt, nr, nb,
210  false, Okular::ObjectRect::Action, okuLink );
211  dviLinks.push_front( orlink );
212  }
213 
214  }
215  return dviLinks;

365 {

370 
371  userMutex()->lock();

374 
375  userMutex()->unlock();

377  if ( prebookmarks.isEmpty() )

380  QStack<QDomElement> stack;

382  QVector<PreBookmark>::ConstIterator it = prebookmarks.constBegin();

388  if ( a.isValid() )

393 
394  fillViewportFromAnchor( vp, a, (int)p->width(), (int)p->height() );
395  domel.setAttribute( "Viewport", vp.toString() );
396  }
397  if ( stack.isEmpty() )
398  m_docSynopsis->appendChild( domel );
399  else
400  {
401  stack.top().appendChild( domel );
402  stack.pop();
403  }
404  for ( int i = 0; i < (*it).noOfChildren; ++i )
405  stack.push( domel );
406  }

497 void DviGenerator::loadPages( QVector< Okular::Page * > &pagesVector )
498 {
499  QSize pageRequiredSize;
500 
501  int numofpages = m_dviRenderer->dviFile->total_pages;
502  pagesVector.resize( numofpages );
503 
504  m_linkGenerated.fill( false, numofpages );
505 
506  //kDebug(DviDebug) << "resolution:" << m_resolution << ", dviFile->preferred?";
507 
508  /* get the suggested size */
509  if ( m_dviRenderer->dviFile->suggestedPageSize )
510  {
511  pageRequiredSize = m_dviRenderer->dviFile->suggestedPageSize->sizeInPixel(
512  m_resolution );
513  }
514  else
515  {
516  pageSize ps;
517  pageRequiredSize = ps.sizeInPixel( m_resolution );
518  }
519 
520  for ( int i = 0; i < numofpages; ++i )
521  {
522  //kDebug(DviDebug) << "getting status of page" << i << ":";
523 
524  if ( pagesVector[i] )
525  {
526  delete pagesVector[i];
527  }
528 
529  Okular::Page * page = new Okular::Page( i,
530  pageRequiredSize.width(),
531  pageRequiredSize.height(),
532  Okular::Rotation0 );
533 
534  pagesVector[i] = page;
535  }
536  kDebug(DviDebug) << "pagesVector successfully inizialized!";
537 
538  // filling the pages with the source references rects
539  const QVector<DVI_SourceFileAnchor>& sourceAnchors = m_dviRenderer->sourceAnchors();
540  QVector< QLinkedList< Okular::SourceRefObjectRect * > > refRects( numofpages );
541  foreach ( const DVI_SourceFileAnchor& sfa, sourceAnchors )
542  {
543  if ( sfa.page < 1 || (int)sfa.page > numofpages )
544  continue;
545 
546  Okular::NormalizedPoint p( -1.0, (double)sfa.distance_from_top.getLength_in_pixel( Okular::Utils::dpiY() ) / (double)pageRequiredSize.height() );
547  Okular::SourceReference * sourceRef = new Okular::SourceReference( sfa.fileName, sfa.line );
548  refRects[ sfa.page - 1 ].append( new Okular::SourceRefObjectRect( p, sourceRef ) );
549  }
550  for ( int i = 0; i < refRects.size(); ++i )
551  if ( !refRects.at(i).isEmpty() )
552  pagesVector[i]->setSourceReferences( refRects.at(i) );
553 }
