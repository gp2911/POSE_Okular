46 bool StyleParser::parseContentFile()
47 {
48  const QDomElement documentElement = mDomDocument.documentElement();
49  QDomElement element = documentElement.firstChildElement();
50  while ( !element.isNull() ) {
51  if ( element.tagName() == QLatin1String( "document-common-attrs" ) ) {
52  if ( !parseDocumentCommonAttrs( element ) )
53  return false;
54  } else if ( element.tagName() == QLatin1String( "font-face-decls" ) ) {
55  if ( !parseFontFaceDecls( element ) )
56  return false;
57  } else if ( element.tagName() == QLatin1String( "styles" ) ) {
58  if ( !parseStyles( element ) )
59  return false;
60  } else if ( element.tagName() == QLatin1String( "automatic-styles" ) ) {
61  if ( !parseAutomaticStyles( element ) )
62  return false;
63  }
64 
65  element = element.nextSiblingElement();
66  }
67 
68  return true;

292  map.insert( "tb-rl", ParagraphFormatProperty::TBRL );

295  map.insert( "rl", ParagraphFormatProperty::RL );
296  map.insert( "tb", ParagraphFormatProperty::TB );
297  map.insert( "page", ParagraphFormatProperty::PAGE );
298  }
299  property.setWritingMode( map[ parent.attribute( "writing-mode" ) ] );
300 
301  static QMap<QString, Qt::Alignment> alignMap;
302  if ( alignMap.isEmpty() ) {
303  alignMap.insert( "center", Qt::AlignCenter );
304  alignMap.insert( "left", Qt::AlignLeft );
305  alignMap.insert( "right", Qt::AlignRight );
306  alignMap.insert( "justify", Qt::AlignJustify );
307  if ( property.writingModeIsRightToLeft() ) {
308  alignMap.insert( "start", Qt::AlignRight );
309  alignMap.insert( "end", Qt::AlignLeft );
310  } else {
311  // not right to left
312  alignMap.insert( "start", Qt::AlignLeft );
313  alignMap.insert( "end", Qt::AlignRight );
314  }
315  }
316  if ( parent.hasAttribute( "text-align" ) ) {
317  property.setTextAlignment( alignMap[ parent.attribute( "text-align", "left" ) ] );
318  }
319 
320  const QString marginLeft = parent.attribute( "margin-left" );
321  if ( !marginLeft.isEmpty() ) {
322  qreal leftMargin = qRound( convertUnit( marginLeft ) );
323  property.setLeftMargin( leftMargin );
324  }
325 
326  const QString colorText = parent.attribute( "background-color" );
327  if ( !colorText.isEmpty() && colorText != QLatin1String( "transparent" ) ) {
328  property.setBackgroundColor( QColor( colorText ) );

455 double StyleParser::convertUnit( const QString &data )
456 {
457  #define MM_TO_POINT(mm) ((mm)*2.83465058)
458  #define CM_TO_POINT(cm) ((cm)*28.3465058)
459  #define DM_TO_POINT(dm) ((dm)*283.465058)
460  #define INCH_TO_POINT(inch) ((inch)*72.0)
461  #define PI_TO_POINT(pi) ((pi)*12)
462  #define DD_TO_POINT(dd) ((dd)*154.08124)
463  #define CC_TO_POINT(cc) ((cc)*12.840103)
464 
465  double points = 0;
466  if ( data.endsWith( "pt" ) ) {
467  points = data.left( data.length() - 2 ).toDouble();
468  } else if ( data.endsWith( "cm" ) ) {
469  double value = data.left( data.length() - 2 ).toDouble();
470  points = CM_TO_POINT( value );
471  } else if ( data.endsWith( "mm" ) ) {
472  double value = data.left( data.length() - 2 ).toDouble();
473  points = MM_TO_POINT( value );
474  } else if ( data.endsWith( "dm" ) ) {
475  double value = data.left( data.length() - 2 ).toDouble();
476  points = DM_TO_POINT( value );
477  } else if ( data.endsWith( "in" ) ) {
478  double value = data.left( data.length() - 2 ).toDouble();
479  points = INCH_TO_POINT( value );
480  } else if ( data.endsWith( "inch" ) ) {
481  double value = data.left( data.length() - 4 ).toDouble();
482  points = INCH_TO_POINT( value );
483  } else if ( data.endsWith( "pi" ) ) {
484  double value = data.left( data.length() - 4 ).toDouble();
485  points = PI_TO_POINT( value );
486  } else if ( data.endsWith( "dd" ) ) {
487  double value = data.left( data.length() - 4 ).toDouble();
488  points = DD_TO_POINT( value );
489  } else if ( data.endsWith( "cc" ) ) {
490  double value = data.left( data.length() - 4 ).toDouble();
491  points = CC_TO_POINT( value );
492  } else {
493  if ( !data.isEmpty() ) {
494  qDebug( "unknown unit for '%s'", qPrintable( data ) );
495  }
496  points = 12;
497  }
498 
499  return points;
