79 QButtonGroup* FormWidgetsController::registerRadioButton( QAbstractButton *button, Okular::FormFieldButton *formButton )
80 {
81  if ( !button )
82  return 0;
83 
84 
85 
86  QList< RadioData >::iterator it = m_radios.begin(), itEnd = m_radios.end();
87  const int id = formButton->id();
88  m_formButtons.insert( id, formButton );
89  m_buttons.insert( id, button );
90  for ( ; it != itEnd; ++it )
91  {
92  const QList< int >::const_iterator idsIt = qFind( (*it).ids, id );
93  if ( idsIt != (*it).ids.constEnd() )
94  {
95  kDebug(4700) << "Adding id" << id << "To group including" << (*it).ids;
96  (*it).group->addButton( button );
97  (*it).group->setId( button, id );
98  return (*it).group;
99  }
100  }
101 
102  const QList< int > siblings = formButton->siblings();
103 
104  RadioData newdata;
105  newdata.ids = siblings;
106  newdata.ids.append( id );
107  newdata.group = new QButtonGroup();
108  newdata.group->addButton( button );
109  newdata.group->setId( button, id );
110 
111  // Groups of 1 (like checkboxes) can't be exclusive
112  if (siblings.isEmpty())
113  newdata.group->setExclusive( false );
114 
115  connect( newdata.group, SIGNAL( buttonClicked(QAbstractButton* ) ),
116  this, SLOT( slotButtonClicked( QAbstractButton* ) ) );
117  m_radios.append( newdata );
118  return newdata.group;
119 }

143 void FormWidgetsController::slotButtonClicked( QAbstractButton *button )
144 {
145  int pageNumber = -1;
146  if ( CheckBoxEdit *check = qobject_cast< CheckBoxEdit * >( button ) )
147  {
148  pageNumber = check->pageItem()->pageNumber();
149  }
150  else if ( RadioButtonEdit *radio = qobject_cast< RadioButtonEdit * >( button ) )
151  {
152  pageNumber = radio->pageItem()->pageNumber();
153  }
154 
155  const QList< QAbstractButton* > buttons = button->group()->buttons();
156  QList< bool > checked;
157  QList< bool > prevChecked;
158  QList< Okular::FormFieldButton*> formButtons;
159 
160  foreach ( QAbstractButton* button, buttons )
161  {
162  checked.append( button->isChecked() );
163  int id = button->group()->id( button );
164  formButtons.append( m_formButtons[id] );
165  prevChecked.append( m_formButtons[id]->state() );
166  }
167  if (checked != prevChecked)
168  emit formButtonsChangedByWidget( pageNumber, formButtons, checked );
169 }

184 FormWidgetIface * FormWidgetFactory::createWidget( Okular::FormField * ff, QWidget * parent )
185 {
186  FormWidgetIface * widget = 0;
187  switch ( ff->type() )
188  {
189  case Okular::FormField::FormButton:
190  {
191  Okular::FormFieldButton * ffb = static_cast< Okular::FormFieldButton * >( ff );
192  switch ( ffb->buttonType() )
193  {
194  case Okular::FormFieldButton::Push:
195  widget = new PushButtonEdit( ffb, parent );
196  break;
197  case Okular::FormFieldButton::CheckBox:
198  widget = new CheckBoxEdit( ffb, parent );
199  break;
200  case Okular::FormFieldButton::Radio:
201  widget = new RadioButtonEdit( ffb, parent );
202  break;
203  default: ;
204  }
205  break;
206  }
207  case Okular::FormField::FormText:
208  {
209  Okular::FormFieldText * fft = static_cast< Okular::FormFieldText * >( ff );
210  switch ( fft->textType() )
211  {
212  case Okular::FormFieldText::Multiline:
213  widget = new TextAreaEdit( fft, parent );
214  break;
215  case Okular::FormFieldText::Normal:
216  widget = new FormLineEdit( fft, parent );
217  break;
218  case Okular::FormFieldText::FileSelect:
219  widget = new FileEdit( fft, parent );
220  break;
221  }
222  break;
223  }
224  case Okular::FormField::FormChoice:
225  {
226  Okular::FormFieldChoice * ffc = static_cast< Okular::FormFieldChoice * >( ff );
227  switch ( ffc->choiceType() )
228  {
229  case Okular::FormFieldChoice::ListBox:
230  widget = new ListEdit( ffc, parent );
231  break;
232  case Okular::FormFieldChoice::ComboBox:
233  widget = new ComboEdit( ffc, parent );
234  break;
235  }
236  break;
237  }
238  default: ;
239  }
240  return widget;
241 }
