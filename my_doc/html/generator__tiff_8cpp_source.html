108 static void adaptSizeToResolution( TIFF *tiff, ttag_t whichres, double dpi, uint32 *size )
109 {
110  float resvalue = 1.0;
111  uint16 resunit = 0;
112  if ( !TIFFGetField( tiff, whichres, &resvalue )
113  || !TIFFGetFieldDefaulted( tiff, TIFFTAG_RESOLUTIONUNIT, &resunit ) )
114  return;
115 
116  float newsize = *size / resvalue;
117  switch ( resunit )
118  {
119  case RESUNIT_INCH:
120  *size = (uint32)( newsize * dpi );
121  break;
122  case RESUNIT_CENTIMETER:
123  *size = (uint32)( newsize * 10.0 / 25.4 * dpi );
124  break;
125  case RESUNIT_NONE:
126  break;
127  }
128 }

130 static Okular::Rotation readTiffRotation( TIFF *tiff )
131 {
132  uint32 tiffOrientation = 0;
133 
134  if ( !TIFFGetField( tiff, TIFFTAG_ORIENTATION, &tiffOrientation ) )
135  return Okular::Rotation0;
136 
137  Okular::Rotation ret = Okular::Rotation0;
138  switch ( tiffOrientation )
139  {
140  case ORIENTATION_TOPLEFT:
141  case ORIENTATION_TOPRIGHT:
142  ret = Okular::Rotation0;
143  break;
144  case ORIENTATION_BOTRIGHT:
145  case ORIENTATION_BOTLEFT:
146  ret = Okular::Rotation180;
147  break;
148  case ORIENTATION_LEFTTOP:
149  case ORIENTATION_LEFTBOT:
150  ret = Okular::Rotation270;
151  break;
152  case ORIENTATION_RIGHTTOP:
153  case ORIENTATION_RIGHTBOT:
154  ret = Okular::Rotation90;
155  break;
156  }
157 
158  return ret;
159 }

256 QImage TIFFGenerator::image( Okular::PixmapRequest * request )
257 {
258  bool generated = false;
259  QImage img;
260 
261  if ( TIFFSetDirectory( d->tiff, mapPage( request->page()->number() ) ) )
262  {
263  int rotation = request->page()->rotation();
264  uint32 width = 1;
265  uint32 height = 1;
266  uint32 orientation = 0;
267  TIFFGetField( d->tiff, TIFFTAG_IMAGEWIDTH, &width );
268  TIFFGetField( d->tiff, TIFFTAG_IMAGELENGTH, &height );
269 
270  if ( !TIFFGetField( d->tiff, TIFFTAG_ORIENTATION, &orientation ) )
271  orientation = ORIENTATION_TOPLEFT;
272 
273  QImage image( width, height, QImage::Format_RGB32 );
274  uint32 * data = (uint32 *)image.bits();
275 
276  // read data
277  if ( TIFFReadRGBAImageOriented( d->tiff, width, height, data, orientation ) != 0 )
278  {
279  // an image read by ReadRGBAImage is ABGR, we need ARGB, so swap red and blue
280  uint32 size = width * height;
281  for ( uint32 i = 0; i < size; ++i )
282  {
283  uint32 red = ( data[i] & 0x00FF0000 ) >> 16;
284  uint32 blue = ( data[i] & 0x000000FF ) << 16;
285  data[i] = ( data[i] & 0xFF00FF00 ) + red + blue;
286  }
287 
288  int reqwidth = request->width();
289  int reqheight = request->height();
290  if ( rotation % 2 == 1 )
291  qSwap( reqwidth, reqheight );
292  img = image.scaled( reqwidth, reqheight, Qt::IgnoreAspectRatio, Qt::SmoothTransformation );
293 
294  generated = true;
295  }
296  }
297 
298  if ( !generated )
299  {
300  img = QImage( request->width(), request->height(), QImage::Format_RGB32 );
301  img.fill( qRgb( 255, 255, 255 ) );
302  }
303 
304  return img;
305 }

343 void TIFFGenerator::loadPages( QVector<Okular::Page*> & pagesVector )
344 {
345  if ( !d->tiff )
346  return;
347 
348  tdir_t dirs = TIFFNumberOfDirectories( d->tiff );
349  pagesVector.resize( dirs );
350  tdir_t realdirs = 0;
351 
352  uint32 width = 0;
353  uint32 height = 0;
354 
355  const double dpiX = Okular::Utils::dpiX();
356  const double dpiY = Okular::Utils::dpiY();
357 
358  for ( tdir_t i = 0; i < dirs; ++i )
359  {
360  if ( !TIFFSetDirectory( d->tiff, i ) )
361  continue;
362 
363  if ( TIFFGetField( d->tiff, TIFFTAG_IMAGEWIDTH, &width ) != 1 ||
364  TIFFGetField( d->tiff, TIFFTAG_IMAGELENGTH, &height ) != 1 )
365  continue;
366 
367  adaptSizeToResolution( d->tiff, TIFFTAG_XRESOLUTION, dpiX, &width );
368  adaptSizeToResolution( d->tiff, TIFFTAG_YRESOLUTION, dpiY, &height );
369 
370  Okular::Page * page = new Okular::Page( realdirs, width, height, readTiffRotation( d->tiff ) );
371  pagesVector[ realdirs ] = page;
372 
373  m_pageMapping[ realdirs ] = i;
374 
375  ++realdirs;
376  }
377 
378  pagesVector.resize( realdirs );
379 }

381 bool TIFFGenerator::print( QPrinter& printer )
382 {
383  uint32 width = 0;
384  uint32 height = 0;
385 
386  QPainter p( &printer );
387 
388  QList<int> pageList = Okular::FilePrinter::pageList( printer, document()->pages(),
389  document()->currentPage() + 1,
390  document()->bookmarkedPageList() );
391 
392  for ( tdir_t i = 0; i < pageList.count(); ++i )
393  {
394  if ( !TIFFSetDirectory( d->tiff, mapPage( pageList[i] - 1 ) ) )
395  continue;
396 
397  if ( TIFFGetField( d->tiff, TIFFTAG_IMAGEWIDTH, &width ) != 1 ||
398  TIFFGetField( d->tiff, TIFFTAG_IMAGELENGTH, &height ) != 1 )
399  continue;
400 
401  QImage image( width, height, QImage::Format_RGB32 );
402  uint32 * data = (uint32 *)image.bits();
403 
404  // read data
405  if ( TIFFReadRGBAImageOriented( d->tiff, width, height, data, ORIENTATION_TOPLEFT ) != 0 )
406  {
407  // an image read by ReadRGBAImage is ABGR, we need ARGB, so swap red and blue
408  uint32 size = width * height;
409  for ( uint32 i = 0; i < size; ++i )
410  {
411  uint32 red = ( data[i] & 0x00FF0000 ) >> 16;
412  uint32 blue = ( data[i] & 0x000000FF ) << 16;
413  data[i] = ( data[i] & 0xFF00FF00 ) + red + blue;
414  }
415  }
416 
417  if ( i != 0 )
418  printer.newPage();
419 
420  QSize targetSize = printer.pageRect().size();
421 
422  if ( (image.width() < targetSize.width()) && (image.height() < targetSize.height()) )
423  {
424  // draw small images at 100% (don't scale up)
425  p.drawImage( 0, 0, image );
426  } else {
427  // fit to page
428  p.drawImage( 0, 0, image.scaled( targetSize, Qt::IgnoreAspectRatio, Qt::SmoothTransformation ) );
429  }
430  }
431 
432  return true;
433 }
