118 void MagnifierView::requestPixmap()
119 {
120  const int full_width = m_page->width() * SCALE;
121  const int full_height = m_page->height() * SCALE;
122 
123  Okular::NormalizedRect nrect = normalizedView();
124 
125  if (m_page && !m_page->hasPixmap( this, full_width, full_height, nrect ))
126  {
127  QLinkedList< Okular::PixmapRequest * > requestedPixmaps;
128 
129  Okular::PixmapRequest *p = new Okular::PixmapRequest( this, m_current, full_width, full_height, PAGEVIEW_PRIO, Okular::PixmapRequest::Asynchronous );
130 
131  if ( m_page->hasTilesManager( this ) ) {
132  p->setTile( true );
133  }
134 
135  // request a little bit bigger rectangle then currently viewed, but not the full scale page
136  const double rect_width = (nrect.right - nrect.left) * 0.5,
137  rect_height = (nrect.bottom - nrect.top) * 0.5;
138 
139  const double top = qMax(nrect.top - rect_height, 0.0);
140  const double bottom = qMin(nrect.bottom + rect_height, 1.0);
141  const double left = qMax(nrect.left - rect_width, 0.0);
142  const double right = qMin(nrect.right + rect_width, 1.0);
143 
144  p->setNormalizedRect( Okular::NormalizedRect(left, top, right, bottom) );
145  requestedPixmaps.push_back( p );
146 
147  m_document->requestPixmaps( requestedPixmaps );
148  }
149 }

158 void MagnifierView::drawTicks( QPainter *p )
159 {
160  p->save();
161 
162  p->setPen(QPen(Qt::black, 0));
163  // the cross
164  p->drawLine(width() / 2, 0, width() / 2, height() - 1);
165  p->drawLine(0, height() / 2, width() - 1, height() / 2);
166 
167  // the borders
168  p->drawLine(0, 0, width() - 1, 0);
169  p->drawLine(width() - 1, 0, width() - 1, height() - 1);
170  p->drawLine(0, height() - 1, width() - 1, height() - 1);
171  p->drawLine(0, height() - 1, 0, 0);
172 
173  // ticks
174  // TODO posibility to switch units (pt, mm, cc, in, printing dots)
175  float ps = (float)SCALE * 5;// how much pixels in widget is one pixel in document * how often
176  int tw = 10; // tick size in pixels
177 
178  for ( float x = 0; x < width(); x += ps )
179  {
180  p->drawLine(x, 1, x, tw);
181  p->drawLine(x, height() - 1, x, height() - tw - 1);
182  p->drawLine(1, x, tw, x);
183  p->drawLine(width() - 1, x, width() - tw - 1, x);
184  }
185 
186  p->restore();
187 }
