105 void dvifile::process_preamble()
106 {
107  command_pointer = dvi_Data();
108 
109  quint8 magic_number = readUINT8();
110  if (magic_number != PRE) {
111  errorMsg = i18n("The DVI file does not start with the preamble.");
112  return;
113  }
114  magic_number = readUINT8();
115  if (magic_number != 2) {
116  errorMsg = i18n("The DVI file contains the wrong version of DVI output for this program. "
117  "Hint: If you use the typesetting system Omega, you have to use a special "
118  "program, such as oxdvi.");
119  return;
120  }
121 
126  quint32 numerator = readUINT32();
127  quint32 denominator = readUINT32();
128  _magnification = readUINT32();
129 
130  cmPerDVIunit = (double(numerator) / double(denominator)) * (double(_magnification) / 1000.0) * (1.0 / 1e5);
131 
132 
133  // Read the generatorString (such as "TeX output ..." from the
134  // DVI-File). The variable "magic_number" holds the length of the
135  // string.
136  char job_id[300];
137  magic_number = readUINT8();
138  strncpy(job_id, (char *)command_pointer, magic_number);
139  job_id[magic_number] = '\0';
140  generatorString = job_id;
141 }

147 void dvifile::find_postamble()
148 {
149  // Move backwards through the TRAILER bytes
150  command_pointer = dvi_Data() + size_of_file - 1;
151  while((*command_pointer == TRAILER) && (command_pointer > dvi_Data()))
152  command_pointer--;
153  if (command_pointer == dvi_Data()) {
154  errorMsg = i18n("The DVI file is badly corrupted. Okular was not able to find the postamble.");
155  return;
156  }
157 
158  // And this is finally the pointer to the beginning of the postamble
159  command_pointer -= 4;
160  beginning_of_postamble = readUINT32();
161  command_pointer = dvi_Data() + beginning_of_postamble;
162 }

165 void dvifile::read_postamble()
166 {
167  quint8 magic_byte = readUINT8();
168  if (magic_byte != POST) {
169  errorMsg = i18n("The postamble does not begin with the POST command.");
170  return;
171  }
172  last_page_offset = readUINT32();
173 
174  // Skip the numerator, denominator and magnification, the largest
175  // box height and width and the maximal depth of the stack. These
176  // are not used at the moment.
177  command_pointer += 4 + 4 + 4 + 4 + 4 + 2;
178 
179  // The number of pages is more interesting for us.
180  total_pages = readUINT16();
181 
182  // As a next step, read the font definitions.
183  quint8 cmnd = readUINT8();
184  while (cmnd >= FNTDEF1 && cmnd <= FNTDEF4) {
185  quint32 TeXnumber = readUINT(cmnd-FNTDEF1+1);
186  quint32 checksum = readUINT32(); // Checksum of the font, as found by TeX in the TFM file
187 
188  // Read scale and design factor, and the name of the font. All
189  // these are explained in section A.4 of the DVI driver standard,
190  // Level 0, published by the TUG DVI driver standards committee
191  quint32 scale = readUINT32();
192  quint32 design = readUINT32();
193  quint16 len = readUINT8() + readUINT8(); // Length of the font name, including the directory name
194  QByteArray fontname((char*)command_pointer, len);
195  command_pointer += len;
196 
197 #ifdef DEBUG_FONTS
198  kDebug(kvs::dvi) << "Postamble: define font \"" << fontname << "\" scale=" << scale << " design=" << design;
199 #endif
200 
201  // According to section A.4 of the DVI driver standard, this font
202  // shall be enlarged by the following factor before it is used.
203  double enlargement_factor = (double(scale) * double(_magnification))/(double(design) * 1000.0);
204 
205  if (font_pool != 0) {
206  TeXFontDefinition *fontp = font_pool->appendx(fontname, checksum, scale, enlargement_factor);
207 
208  // Insert font in dictionary and make sure the dictionary is big
209  // enough.
210  if (tn_table.capacity()-2 <= tn_table.count())
211  // Not quite optimal. The size of the dictionary should be a
212  // prime for optimal performance. I don't care.
213  tn_table.reserve(tn_table.capacity()*2);
214  tn_table.insert(TeXnumber, fontp);
215  }
216 
217  // Read the next command
218  cmnd = readUINT8();
219  }
220 
221  if (cmnd != POSTPOST) {
222  errorMsg = i18n("The postamble contained a command other than FNTDEF.");
223  return;
224  }
225 
226  // Now we remove all those fonts from the memory which are no longer
227  // in use.
228  if (font_pool != 0)
229  font_pool->release_fonts();
230 }

233 void dvifile::prepare_pages()
234 {
235 #ifdef DEBUG_DVIFILE
236  kDebug(kvs::dvi) << "prepare_pages";
237 #endif
238  if (total_pages == 0)
239  return;
240 
241  page_offset.resize(total_pages+1);
242  if (page_offset.size() < (total_pages+1)) {
243  kError(kvs::dvi) << "No memory for page list!" << endl;
244  return;
245  }
246  for(int i=0; i<=total_pages; i++)
247  page_offset[i] = 0;
248 
249  page_offset[int(total_pages)] = beginning_of_postamble;
250  int j = total_pages-1;
251  page_offset[j] = last_page_offset;
252 
253  // Follow back pointers through pages in the DVI file, storing the
254  // offsets in the page_offset table.
255  while (j > 0) {
256  command_pointer = dvi_Data() + page_offset[j--];
257  if (readUINT8() != BOP) {
258  errorMsg = i18n("The page %1 does not start with the BOP command.", j+1);
259  return;
260  }
261  command_pointer += 10 * 4;
262  page_offset[j] = readUINT32();
263  if ((dvi_Data()+page_offset[j] < dvi_Data())||(dvi_Data()+page_offset[j] > dvi_Data()+size_of_file))
264  break;
265  }
266 }

336 void dvifile::renumber()
337 {
338  dviData.detach();
339 
340  // Write the page number to the file, taking good care of byte
341  // orderings.
342  bool bigEndian = (QSysInfo::ByteOrder == QSysInfo::BigEndian);
343 
344  for(int i=1; i<=total_pages; i++) {
345  quint8 *ptr = dviData.data() + page_offset[i-1]+1;
346  quint8 *num = (quint8 *)&i;
347  for(quint8 j=0; j<4; j++)
348  if (bigEndian) {
349  *(ptr++) = num[0];
350  *(ptr++) = num[1];
351  *(ptr++) = num[2];
352  *(ptr++) = num[3];
353  } else {
354  *(ptr++) = num[3];
355  *(ptr++) = num[2];
356  *(ptr++) = num[1];
357  *(ptr++) = num[0];
358  }
359  }

363 QString dvifile::convertPDFtoPS(const QString &PDFFilename, QString *converrorms)
364 {
365  // Check if the PDFFile is known
366  QMap<QString, QString>::Iterator it = convertedFiles.find(PDFFilename);
367  if (it != convertedFiles.end()) {
368  // PDF-File is known. Good.
369  return it.value();
370  }
371 
372  // Get the name of a temporary file.
373  // Must open the QTemporaryFile to access the name.
374  QTemporaryFile tmpfile;
375  tmpfile.open();
376  const QString convertedFileName = tmpfile.fileName();
377  tmpfile.close();
378 
379  // Use pdf2ps to do the conversion
380  QProcess pdf2ps;
381  pdf2ps.setReadChannelMode(QProcess::MergedChannels);
382  pdf2ps.start("pdf2ps",
383  QStringList() << PDFFilename << convertedFileName,
384  QIODevice::ReadOnly|QIODevice::Text);
385 
386  if (!pdf2ps.waitForStarted()) {
387  // Indicates that conversion failed, won't try again.
388  convertedFiles[PDFFilename].clear();
389  if (converrorms != 0 && !have_complainedAboutMissingPDF2PS) {
390  *converrorms = i18n("<qt><p>The external program <strong>pdf2ps</strong> could not be started. As a result, "
391  "the PDF-file %1 could not be converted to PostScript. Some graphic elements in your "
392  "document will therefore not be displayed.</p>"
393  "<p><b>Possible reason:</b> The program <strong>pdf2ps</strong> may not be installed "
394  "on your system, or cannot be found in the current search path.</p>"
395  "<p><b>What you can do:</b> The program <strong>pdf2ps</strong> is normally "
396  "contained in distributions of the ghostscript PostScript interpreter system. If "
397  "ghostscript is not installed on your system, you could install it now. "
398  "If you are sure that ghostscript is installed, try to use <strong>pdf2ps</strong> "
399  "from the command line to check if it really works.</p><p><b>PATH:</b> %2</p></qt>", PDFFilename, getenv("PATH"));
400  have_complainedAboutMissingPDF2PS = true;
401  }
402  return QString();
403  }
404 
405  // We wait here while the external program runs concurrently.
406  pdf2ps.waitForFinished(-1);
407 
408  if (!QFile::exists(convertedFileName) || pdf2ps.exitCode() != 0) {
409  // Indicates that conversion failed, won't try again.
410  convertedFiles[PDFFilename].clear();
411  if (converrorms != 0) {
412  const QString output = pdf2ps.readAll();
413 
414  *converrorms = i18n("<qt><p>The PDF-file %1 could not be converted to PostScript. Some graphic elements in your "
415  "document will therefore not be displayed.</p>"
416  "<p><b>Possible reason:</b> The file %1 might be broken, or might not be a PDF-file at all. "
417  "This is the output of the <strong>pdf2ps</strong> program that Okular used:</p>"
418  "<p><strong>%2</strong></p></qt>", PDFFilename, output);
419  }
420  return QString();
421  }
422  // Save name of converted file to buffer, so PDF file won't be
423  // converted again, and files can be deleted when *this is
424  // deconstructed.
425  convertedFiles[PDFFilename] = convertedFileName;
426 
427  tmpfile.setAutoRemove(false);
428  return convertedFileName;
429 }
