46 void GSRendererThread::run()
47 {
48  while(1)
49  {
50  m_semaphore.acquire();
51  {
52  m_queueMutex.lock();
53  GSRendererThreadRequest req = m_queue.dequeue();
54  m_queueMutex.unlock();
55 
56  spectre_render_context_set_scale(m_renderContext, req.magnify, req.magnify);
57  spectre_render_context_set_use_platform_fonts(m_renderContext, req.platformFonts);
58  spectre_render_context_set_antialias_bits(m_renderContext, req.graphicsAAbits, req.textAAbits);
59  // Do not use spectre_render_context_set_rotation makes some files not render correctly, e.g. bug210499.ps
60  // so we basically do the rendering without any rotation and then rotate to the orientation as needed
61  // spectre_render_context_set_rotation(m_renderContext, req.orientation);
62 
63  unsigned char *data = NULL;
64  int row_length = 0;
65  int wantedWidth = req.request->width();
66  int wantedHeight = req.request->height();
67 
68  if ( req.orientation % 2 )
69  qSwap( wantedWidth, wantedHeight );
70 
71  spectre_page_render(req.spectrePage, m_renderContext, &data, &row_length);
72 
73  // Qt needs the missing alpha of QImage::Format_RGB32 to be 0xff
74  if (data && data[3] != 0xff)
75  {
76  for (int i = 3; i < row_length * wantedHeight; i += 4)
77  data[i] = 0xff;
78  }
79 
80  QImage img;
81  if (row_length == wantedWidth * 4)
82  {
83  img = QImage(data, wantedWidth, wantedHeight, QImage::Format_RGB32);
84  }
85  else
86  {
87  // In case this ends up beign very slow we can try with some memmove
88  QImage aux(data, row_length / 4, wantedHeight, QImage::Format_RGB32);
89  img = QImage(aux.copy(0, 0, wantedWidth, wantedHeight));
90  }
91 
92  switch (req.orientation)
93  {
94  case Okular::Rotation90:
95  {
96  QTransform m;
97  m.rotate(90);
98  img = img.transformed( m );
99  break;
100  }
101 
102  case Okular::Rotation180:
103  {
104  QTransform m;
105  m.rotate(180);
106  img = img.transformed( m );
107  break;
108  }
109  case Okular::Rotation270:
110  {
111  QTransform m;
112  m.rotate(270);
113  img = img.transformed( m );
114  }
115  }
116 
117  QImage *image = new QImage(img.copy());
118  free(data);
119 
120  if (image->width() != req.request->width() || image->height() != req.request->height())
121  {
122  kWarning(4711).nospace() << "Generated image does not match wanted size: "
123  << "[" << image->width() << "x" << image->height() << "] vs requested "
124  << "[" << req.request->width() << "x" << req.request->height() << "]";
125  QImage aux = image->scaled(wantedWidth, wantedHeight);
126  delete image;
127  image = new QImage(aux);
128  }
129  emit imageDone(image, req.request);
130 
131  spectre_page_free(req.spectrePage);
132  }
133  }
134 }
