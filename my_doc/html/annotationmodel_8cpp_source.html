120 void AnnotationModelPrivate::notifyPageChanged( int page, int flags )
121 {
122  // we are strictly interested in annotations
123  if ( !(flags & Okular::DocumentObserver::Annotations ) )
124  return;
125 
126  const QLinkedList< Okular::Annotation* > annots = filterOutWidgetAnnotations( document->page( page )->annotations() );
127  int annItemIndex = -1;
128  AnnItem *annItem = findItem( page, &annItemIndex );
129  // case 1: the page has no more annotations
130  // => remove the branch, if any
131  if ( annots.isEmpty() )
132  {
133  if ( annItem )
134  {
135  q->beginRemoveRows( indexForItem( root ), annItemIndex, annItemIndex );
136  delete root->children.at( annItemIndex );
137  root->children.removeAt( annItemIndex );
138  q->endRemoveRows();
139  }
140  return;
141  }
142  // case 2: no existing branch
143  // => add a new branch, and add the annotations for the page
144  if ( !annItem )
145  {
146  int i = 0;
147  while ( i < root->children.count() && root->children.at( i )->page < page ) ++i;
148 
149  AnnItem *annItem = new AnnItem();
150  annItem->page = page;
151  annItem->parent = root;
152  q->beginInsertRows( indexForItem( root ), i, i );
153  annItem->parent->children.insert( i, annItem );
154  q->endInsertRows();
155  QLinkedList< Okular::Annotation* >::ConstIterator it = annots.begin(), itEnd = annots.end();
156  int newid = 0;
157  for ( ; it != itEnd; ++it, ++newid )
158  {
159  q->beginInsertRows( indexForItem( annItem ), newid, newid );
160  new AnnItem( annItem, *it );
161  q->endInsertRows();
162  }
163  return;
164  }
165  // case 3: existing branch, less annotations than items
166  // => lookup and remove the annotations
167  if ( annItem->children.count() > annots.count() )
168  {
169  for ( int i = annItem->children.count(); i > 0; --i )
170  {
171  Okular::Annotation *ref = annItem->children.at( i - 1 )->annotation;
172  bool found = false;
173  QLinkedList< Okular::Annotation* >::ConstIterator it = annots.begin(), itEnd = annots.end();
174  for ( ; !found && it != itEnd; ++it )
175  {
176  if ( ( *it ) == ref )
177  found = true;
178  }
179  if ( !found )
180  {
181  q->beginRemoveRows( indexForItem( annItem ), i - 1, i - 1 );
182  delete annItem->children.at( i - 1 );
183  annItem->children.removeAt( i - 1 );
184  q->endRemoveRows();
185  }
186  }
187  return;
188  }
189  // case 4: existing branch, less items than annotations
190  // => lookup and add annotations if not in the branch
191  if ( annots.count() > annItem->children.count() )
192  {
193  QLinkedList< Okular::Annotation* >::ConstIterator it = annots.begin(), itEnd = annots.end();
194  for ( ; it != itEnd; ++it )
195  {
196  Okular::Annotation *ref = *it;
197  bool found = false;
198  int count = annItem->children.count();
199  for ( int i = 0; !found && i < count; ++i )
200  {
201  if ( ref == annItem->children.at( i )->annotation )
202  found = true;
203  }
204  if ( !found )
205  {
206  q->beginInsertRows( indexForItem( annItem ), count, count );
207  new AnnItem( annItem, ref );
208  q->endInsertRows();
209  }
210  }
211  return;
212  }
213  // case 5: the data of some annotation changed
214  // TODO: what do we do in this case?
215  // FIXME: for now, update ALL the annotations for that page
216  for ( int i = 0; i < annItem->children.count(); ++i )
217  {
218  QModelIndex index = indexForItem( annItem->children.at( i ) );
219  emit q->dataChanged( index, index );
220  }
221 }

234 void AnnotationModelPrivate::rebuildTree( const QVector< Okular::Page * > &pages )
235 {
236  if ( pages.isEmpty() )
237  return;
238 
239  emit q->layoutAboutToBeChanged();
240  for ( int i = 0; i < pages.count(); ++i )
241  {
242  const QLinkedList< Okular::Annotation* > annots = filterOutWidgetAnnotations( pages.at( i )->annotations() );
243  if ( annots.isEmpty() )
244  continue;
245 
246  AnnItem *annItem = new AnnItem( root, i );
247  QLinkedList< Okular::Annotation* >::ConstIterator it = annots.begin(), itEnd = annots.end();
248  for ( ; it != itEnd; ++it )
249  {
250  new AnnItem( annItem, *it );
251  }
252  }
253  emit q->layoutChanged();
254 }

256 AnnItem* AnnotationModelPrivate::findItem( int page, int *index ) const
257 {
258  for ( int i = 0; i < root->children.count(); ++i )
259  {
260  AnnItem *tmp = root->children.at( i );
261  if ( tmp->page == page )
262  {
263  if ( index )
264  *index = i;
265  return tmp;
266  }
267  }
268  if ( index )
269  *index = -1;
270  return 0;
271 }

296 QVariant AnnotationModel::data( const QModelIndex &index, int role ) const
297 {
298  if ( !index.isValid() )
299  return QVariant();
300 
301  AnnItem *item = static_cast< AnnItem* >( index.internalPointer() );
302  if ( !item->annotation )
303  {
304  if ( role == Qt::DisplayRole )
305  return i18n( "Page %1", item->page + 1 );
306  else if ( role == Qt::DecorationRole )
307  return KIcon( "text-plain" );
308  else if ( role == PageRole )
309  return item->page;
310 
311  return QVariant();
312  }
313  switch ( role )
314  {
315  case Qt::DisplayRole:
316  return GuiUtils::captionForAnnotation( item->annotation );
317  break;
318  case Qt::DecorationRole:
319  return KIcon( "okular" );
320  break;
321  case Qt::ToolTipRole:
322  return GuiUtils::prettyToolTip( item->annotation );
323  break;
324  case AuthorRole:
325  return item->annotation->author();
326  break;
327  case PageRole:
328  return item->page;
329  break;
330  }
331  return QVariant();
332 }
