37 QRect Utils::rotateRect( const QRect & source, int width, int height, int orientation )
38 {
39  QRect ret;
40 
41  // adapt the coordinates of the boxes to the rotation
42  switch ( orientation )
43  {
44  case 1:
45  ret = QRect( width - source.y() - source.height(), source.x(),
46  source.height(), source.width() );
47  break;
48  case 2:
49  ret = QRect( width - source.x() - source.width(), height - source.y() - source.height(),
50  source.width(), source.height() );
51  break;
52  case 3:
53  ret = QRect( source.y(), height - source.x() - source.width(),
54  source.height(), source.width() );
55  break;
56  case 0: // no modifications
57  default: // other cases
58  ret = source;
59  }
60 
61  return ret;
62 }

89 {

93 #if HAVE_LIBKSCREEN

95  KScreen::OutputList outputs = config->outputs();
96  QPoint globalPos = widgetOnScreen->parentWidget() ?
97  widgetOnScreen->mapToGlobal(widgetOnScreen->pos()):
98  widgetOnScreen->pos();
99  QRect widgetRect(globalPos, widgetOnScreen->size());
100 
101  KScreen::Output* selectedOutput = 0;
102  int maxArea = 0;
103  Q_FOREACH(KScreen::Output *output, outputs)
104  {
105  if (output->currentMode())
106  {
107  QRect outputRect(output->pos(),output->currentMode()->size());
108  QRect intersection = outputRect.intersected(widgetRect);
109  int area = intersection.width()*intersection.height();
110  if (area > maxArea)
111  {
112  maxArea = area;
113  selectedOutput = output;
114  }
115  }
116  }
117 
118  if (selectedOutput)
119  {
120  kDebug() << "Found widget at output #" << selectedOutput->id();
121  QRect outputRect(selectedOutput->pos(),selectedOutput->currentMode()->size());
122  QSize szMM = selectedOutput->sizeMm();
123  QSizeF res(static_cast<qreal>(outputRect.width())*25.4/szMM.width(),
124  static_cast<qreal>(outputRect.height())*25.4/szMM.height());
125  kDebug() << "Output DPI is " << res;
126  return res;
127  }
128 #endif
129  }
130  // this is also fallback for LibKScreen branch if KScreen::Output
131  // for particular widget was not found
132  const QDesktopWidget* desktop = QApplication::desktop();

152  static CGDisplayErr GetDisplayDPI( CFDictionaryRef displayModeDict, CGDirectDisplayID displayID,
153  double *horizontalDPI, double *verticalDPI )
154  {
155  CGDisplayErr err = kCGErrorFailure;
156  io_connect_t displayPort;
157  CFDictionaryRef displayDict;
158 
159  // Grab a connection to IOKit for the requested display
160  displayPort = CGDisplayIOServicePort( displayID );
161  if ( displayPort != MACH_PORT_NULL )
162  {
163  // Find out what IOKit knows about this display
164  displayDict = IODisplayCreateInfoDictionary(displayPort, 0);
165  if ( displayDict != NULL )
166  {
167  const double mmPerInch = 25.4;
168  double horizontalSizeInInches =
169  (double)GetIntFromDictionaryForKey(displayDict,
170  CFSTR(kDisplayHorizontalImageSize)) / mmPerInch;
171  double verticalSizeInInches =
172  (double)GetIntFromDictionaryForKey(displayDict,
173  CFSTR(kDisplayVerticalImageSize)) / mmPerInch;
174 
175  // Make sure to release the dictionary we got from IOKit
176  CFRelease(displayDict);
177 
178  // Now we can calculate the actual DPI
179  // with information from the displayModeDict
180  *horizontalDPI =
181  (double)GetIntFromDictionaryForKey( displayModeDict, kCGDisplayWidth )
182  / horizontalSizeInInches;
183  *verticalDPI = (double)GetIntFromDictionaryForKey( displayModeDict,
184  kCGDisplayHeight ) / verticalSizeInInches;
185  err = CGDisplayNoErr;
186  }
187  }
188  return err;
189  }

257 NormalizedRect Utils::imageBoundingBox( const QImage * image )
258 {
259  if ( !image )
260  return NormalizedRect();
261 
262  int width = image->width();
263  int height = image->height();
264  int left, top, bottom, right, x, y;
265 
266 #ifdef BBOX_DEBUG
267  QTime time;
268  time.start();
269 #endif
270 
271  // Scan pixels for top non-white
272  for ( top = 0; top < height; ++top )
273  for ( x = 0; x < width; ++x )
274  if ( !isWhite( image->pixel( x, top ) ) )
275  goto got_top;
276  return NormalizedRect( 0, 0, 0, 0 ); // the image is blank
277 got_top:
278  left = right = x;
279 
280  // Scan pixels for bottom non-white
281  for ( bottom = height-1; bottom >= top; --bottom )
282  for ( x = width-1; x >= 0; --x )
283  if ( !isWhite( image->pixel( x, bottom ) ) )
284  goto got_bottom;
285  Q_ASSERT( 0 ); // image changed?!
286 got_bottom:
287  if ( x < left )
288  left = x;
289  if ( x > right )
290  right = x;
291 
292  // Scan for leftmost and rightmost (we already found some bounds on these):
293  for ( y = top; y <= bottom && ( left > 0 || right < width-1 ); ++y )
294  {
295  for ( x = 0; x < left; ++x )
296  if ( !isWhite( image->pixel( x, y ) ) )
297  left = x;
298  for ( x = width-1; x > right+1; --x )
299  if ( !isWhite( image->pixel( x, y ) ) )
300  right = x;
301  }
302 
303  NormalizedRect bbox( QRect( left, top, ( right - left + 1), ( bottom - top + 1 ) ),
304  image->width(), image->height() );
305 
306 #ifdef BBOX_DEBUG
307  kDebug() << "Computed bounding box" << bbox << "in" << time.elapsed() << "ms";
308 #endif
309 
310  return bbox;
311 }

314 {
315  QByteArray buffer( 65536, '\0' );
316  qint64 read = 0;
317  qint64 written = 0;
318  while ( ( read = from->read( buffer.data(), buffer.size() ) ) > 0 )

326 QTransform Okular::buildRotationMatrix(Rotation rotation)
327 {
328  QTransform matrix;
329  matrix.rotate( (int)rotation * 90 );
330 
331  switch ( rotation )
332  {
333  case Rotation90:

335  break;
