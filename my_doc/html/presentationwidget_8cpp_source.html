80  void recalcGeometry( int width, int height, float screenRatio )
81  {
82  // calculate frame geometry keeping constant aspect ratio
83  float pageRatio = page->ratio();
84  int pageWidth = width,
85  pageHeight = height;
86  if ( pageRatio > screenRatio )
87  pageWidth = (int)( (float)pageHeight / pageRatio );
88  else
89  pageHeight = (int)( (float)pageWidth * pageRatio );
90  geometry.setRect( ( width - pageWidth ) / 2,
91  ( height - pageHeight ) / 2,
92  pageWidth, pageHeight );
93  Q_FOREACH ( VideoWidget *vw, videoWidgets )
94  {
95  const Okular::NormalizedRect r = vw->normGeometry();
96  QRect vwgeom = r.geometry( geometry.width(), geometry.height() );
97  vw->resize( vwgeom.size() );
98  vw->move( geometry.topLeft() + vwgeom.topLeft() );
99  }
100  }

303 void PresentationWidget::notifySetup( const QVector< Okular::Page * > & pageSet, int setupFlags )
304 {
305  // same document, nothing to change - here we assume the document sets up
306  // us with the whole document set as first notifySetup()
307  if ( !( setupFlags & Okular::DocumentObserver::DocumentChanged ) )
308  return;
309 
310  // delete previous frames (if any (shouldn't be))
311  QVector< PresentationFrame * >::iterator fIt = m_frames.begin(), fEnd = m_frames.end();
312  for ( ; fIt != fEnd; ++fIt )
313  delete *fIt;
314  if ( !m_frames.isEmpty() )
315  kWarning() << "Frames setup changed while a Presentation is in progress.";
316  m_frames.clear();
317 
318  // create the new frames
319  QVector< Okular::Page * >::const_iterator setIt = pageSet.begin(), setEnd = pageSet.end();
320  float screenRatio = (float)m_height / (float)m_width;
321  for ( ; setIt != setEnd; ++setIt )
322  {
323  PresentationFrame * frame = new PresentationFrame();
324  frame->page = *setIt;
325  const QLinkedList< Okular::Annotation * > annotations = (*setIt)->annotations();
326  QLinkedList< Okular::Annotation * >::const_iterator aIt = annotations.begin(), aEnd = annotations.end();
327  for ( ; aIt != aEnd; ++aIt )
328  {
329  Okular::Annotation * a = *aIt;
330  if ( a->subType() == Okular::Annotation::AMovie )
331  {
332  Okular::MovieAnnotation * movieAnn = static_cast< Okular::MovieAnnotation * >( a );
333  VideoWidget * vw = new VideoWidget( movieAnn, movieAnn->movie(), m_document, this );
334  frame->videoWidgets.insert( movieAnn->movie(), vw );
335  vw->pageInitialized();
336  }
337  else if ( a->subType() == Okular::Annotation::AScreen )
338  {
339  const Okular::ScreenAnnotation * screenAnn = static_cast< Okular::ScreenAnnotation * >( a );
340  Okular::Movie *movie = GuiUtils::renditionMovieFromScreenAnnotation( screenAnn );
341  if ( movie )
342  {
343  VideoWidget * vw = new VideoWidget( screenAnn, movie, m_document, this );
344  frame->videoWidgets.insert( movie, vw );
345  vw->pageInitialized();
346  }
347  }
348  }
349  frame->recalcGeometry( m_width, m_height, screenRatio );
350  // add the frame to the vector
351  m_frames.push_back( frame );
352  }
353 
354  // get metadata from the document
355  m_metaStrings.clear();
356  const Okular::DocumentInfo * info = m_document->documentInfo();
357  if ( info )
358  {
359  if ( !info->get( "title" ).isNull() )
360  m_metaStrings += i18n( "Title: %1", info->get( "title" ) );
361  if ( !info->get( "author" ).isNull() )
362  m_metaStrings += i18n( "Author: %1", info->get( "author" ) );
363  }
364  m_metaStrings += i18n( "Pages: %1", m_document->pages() );
365  m_metaStrings += i18n( "Click to begin" );
366 
367  m_isSetup = true;
368 }

390 void PresentationWidget::notifyCurrentPageChanged( int previousPage, int currentPage )
391 {
392  if ( previousPage != -1 )
393  {
394  // stop video playback
395  Q_FOREACH ( VideoWidget *vw, m_frames[ previousPage ]->videoWidgets )
396  {
397  vw->stop();
398  vw->pageLeft();
399  }
400 
401  // stop audio playback, if any
402  Okular::AudioPlayer::instance()->stopPlaybacks();
403 
404  // perform the page closing action, if any
405  if ( m_document->page( previousPage )->pageAction( Okular::Page::Closing ) )
406  m_document->processAction( m_document->page( previousPage )->pageAction( Okular::Page::Closing ) );
407 
408  // perform the additional actions of the page's annotations, if any
409  Q_FOREACH ( const Okular::Annotation *annotation, m_document->page( previousPage )->annotations() )
410  {
411  Okular::Action *action = 0;
412 
413  if ( annotation->subType() == Okular::Annotation::AScreen )
414  action = static_cast<const Okular::ScreenAnnotation*>( annotation )->additionalAction( Okular::Annotation::PageClosing );
415  else if ( annotation->subType() == Okular::Annotation::AWidget )
416  action = static_cast<const Okular::WidgetAnnotation*>( annotation )->additionalAction( Okular::Annotation::PageClosing );
417 
418  if ( action )
419  m_document->processAction( action );
420  }
421  }
422 
423  if ( currentPage != -1 )
424  {
425  m_frameIndex = currentPage;
426 
427  // check if pixmap exists or else request it
428  PresentationFrame * frame = m_frames[ m_frameIndex ];
429  int pixW = frame->geometry.width();
430  int pixH = frame->geometry.height();
431 
432  bool signalsBlocked = m_pagesEdit->signalsBlocked();
433  m_pagesEdit->blockSignals( true );
434  m_pagesEdit->setText( QString::number( m_frameIndex + 1 ) );
435  m_pagesEdit->blockSignals( signalsBlocked );
436 
437  // if pixmap not inside the Okular::Page we request it and wait for
438  // notifyPixmapChanged call or else we can proceed to pixmap generation
439  if ( !frame->page->hasPixmap( this, pixW, pixH ) )
440  {
441  requestPixmaps();
442  }
443  else
444  {
445  // make the background pixmap
446  generatePage();
447  }
448 
449  // perform the page opening action, if any
450  if ( m_document->page( m_frameIndex )->pageAction( Okular::Page::Opening ) )
451  m_document->processAction( m_document->page( m_frameIndex )->pageAction( Okular::Page::Opening ) );
452 
453  // perform the additional actions of the page's annotations, if any
454  Q_FOREACH ( const Okular::Annotation *annotation, m_document->page( m_frameIndex )->annotations() )
455  {
456  Okular::Action *action = 0;
457 
458  if ( annotation->subType() == Okular::Annotation::AScreen )
459  action = static_cast<const Okular::ScreenAnnotation*>( annotation )->additionalAction( Okular::Annotation::PageOpening );
460  else if ( annotation->subType() == Okular::Annotation::AWidget )
461  action = static_cast<const Okular::WidgetAnnotation*>( annotation )->additionalAction( Okular::Annotation::PageOpening );
462 
463  if ( action )
464  m_document->processAction( action );
465  }
466 
467  // start autoplay video playback
468  Q_FOREACH ( VideoWidget *vw, m_frames[ m_frameIndex ]->videoWidgets )
469  vw->pageEntered();
470  }

1263 void PresentationWidget::requestPixmaps()
1264 {

1266  int pixW = frame->geometry.width();
1267  int pixH = frame->geometry.height();
1268 
1269  // operation will take long: set busy cursor
1270  QApplication::setOverrideCursor( QCursor( Qt::BusyCursor ) );
1271  // request the pixmap
1272  QLinkedList< Okular::PixmapRequest * > requests;
1273  requests.push_back( new Okular::PixmapRequest( this, m_frameIndex, pixW, pixH, PRESENTATION_PRIO, Okular::PixmapRequest::NoFeature ) );
1274  // restore cursor
1275  QApplication::restoreOverrideCursor();
1276  // ask for next and previous page if not in low memory usage setting
1277  if ( Okular::SettingsCore::memoryLevel() != Okular::SettingsCore::EnumMemoryLevel::Low )
1278  {
1279  int pagesToPreload = 1;
1280 
1281  // If greedy, preload everything
1282  if (Okular::SettingsCore::memoryLevel() == Okular::SettingsCore::EnumMemoryLevel::Greedy)
1283  pagesToPreload = (int)m_document->pages();
1284 
1285  Okular::PixmapRequest::PixmapRequestFeatures requestFeatures = Okular::PixmapRequest::Preload;
1286  requestFeatures |= Okular::PixmapRequest::Asynchronous;
1287

1289  {
1290  int tailRequest = m_frameIndex + j;
1291  if ( tailRequest < (int)m_document->pages() )
1292  {
1293  PresentationFrame *nextFrame = m_frames[ tailRequest ];
1294  pixW = nextFrame->geometry.width();
1295  pixH = nextFrame->geometry.height();

1297  requests.push_back( new Okular::PixmapRequest( this, tailRequest, pixW, pixH, PRESENTATION_PRELOAD_PRIO, requestFeatures ) );
1298  }
1299 
1300  int headRequest = m_frameIndex - j;
1301  if ( headRequest >= 0 )
1302  {
1303  PresentationFrame *prevFrame = m_frames[ headRequest ];
1304  pixW = prevFrame->geometry.width();
1305  pixH = prevFrame->geometry.height();

1307  requests.push_back( new Okular::PixmapRequest( this, headRequest, pixW, pixH, PRESENTATION_PRELOAD_PRIO, requestFeatures ) );
1308  }
1309

1311  if ( headRequest < 0 && tailRequest >= (int)m_document->pages() )

1757 void PresentationWidget::initTransition( const Okular::PageTransition *transition )
1758 {
1759  // if it's just a 'replace' transition, repaint the screen
1760  if ( transition->type() == Okular::PageTransition::Replace )
1761  {
1762  update();
1763  return;
1764  }
1765 
1766  const bool isInward = transition->direction() == Okular::PageTransition::Inward;
1767  const bool isHorizontal = transition->alignment() == Okular::PageTransition::Horizontal;
1768  const float totalTime = transition->duration();
1769 
1770  m_transitionRects.clear();
1771 
1772  switch( transition->type() )
1773  {
1774  // split: horizontal / vertical and inward / outward
1775  case Okular::PageTransition::Split:
1776  {
1777  const int steps = isHorizontal ? 100 : 75;
1778  if ( isHorizontal )
1779  {
1780  if ( isInward )
1781  {
1782  int xPosition = 0;
1783  for ( int i = 0; i < steps; i++ )
1784  {
1785  int xNext = ((i + 1) * m_width) / (2 * steps);
1786  m_transitionRects.push_back( QRect( xPosition, 0, xNext - xPosition, m_height ) );
1787  m_transitionRects.push_back( QRect( m_width - xNext, 0, xNext - xPosition, m_height ) );
1788  xPosition = xNext;
1789  }
1790  }
1791  else
1792  {
1793  int xPosition = m_width / 2;
1794  for ( int i = 0; i < steps; i++ )
1795  {
1796  int xNext = ((steps - (i + 1)) * m_width) / (2 * steps);
1797  m_transitionRects.push_back( QRect( xNext, 0, xPosition - xNext, m_height ) );
1798  m_transitionRects.push_back( QRect( m_width - xPosition, 0, xPosition - xNext, m_height ) );
1799  xPosition = xNext;
1800  }
1801  }
1802  }
1803  else
1804  {
1805  if ( isInward )
1806  {
1807  int yPosition = 0;
1808  for ( int i = 0; i < steps; i++ )
1809  {
1810  int yNext = ((i + 1) * m_height) / (2 * steps);
1811  m_transitionRects.push_back( QRect( 0, yPosition, m_width, yNext - yPosition ) );
1812  m_transitionRects.push_back( QRect( 0, m_height - yNext, m_width, yNext - yPosition ) );
1813  yPosition = yNext;
1814  }
1815  }
1816  else
1817  {
1818  int yPosition = m_height / 2;
1819  for ( int i = 0; i < steps; i++ )
1820  {
1821  int yNext = ((steps - (i + 1)) * m_height) / (2 * steps);
1822  m_transitionRects.push_back( QRect( 0, yNext, m_width, yPosition - yNext ) );
1823  m_transitionRects.push_back( QRect( 0, m_height - yPosition, m_width, yPosition - yNext ) );
1824  yPosition = yNext;
1825  }
1826  }
1827  }
1828  m_transitionMul = 2;
1829  m_transitionDelay = (int)( (totalTime * 1000) / steps );
1830  } break;
1831 
1832  // blinds: horizontal(l-to-r) / vertical(t-to-b)
1833  case Okular::PageTransition::Blinds:
1834  {
1835  const int blinds = isHorizontal ? 8 : 6;
1836  const int steps = m_width / (4 * blinds);
1837  if ( isHorizontal )
1838  {
1839  int xPosition[ 8 ];
1840  for ( int b = 0; b < blinds; b++ )
1841  xPosition[ b ] = (b * m_width) / blinds;
1842 
1843  for ( int i = 0; i < steps; i++ )
1844  {
1845  int stepOffset = (int)( ((float)i * (float)m_width) / ((float)blinds * (float)steps) );
1846  for ( int b = 0; b < blinds; b++ )
1847  {
1848  m_transitionRects.push_back( QRect( xPosition[ b ], 0, stepOffset, m_height ) );
1849  xPosition[ b ] = stepOffset + (b * m_width) / blinds;
1850  }
1851  }
1852  }
1853  else
1854  {
1855  int yPosition[ 6 ];
1856  for ( int b = 0; b < blinds; b++ )
1857  yPosition[ b ] = (b * m_height) / blinds;
1858 
1859  for ( int i = 0; i < steps; i++ )
1860  {
1861  int stepOffset = (int)( ((float)i * (float)m_height) / ((float)blinds * (float)steps) );
1862  for ( int b = 0; b < blinds; b++ )
1863  {
1864  m_transitionRects.push_back( QRect( 0, yPosition[ b ], m_width, stepOffset ) );
1865  yPosition[ b ] = stepOffset + (b * m_height) / blinds;
1866  }
1867  }
1868  }
1869  m_transitionMul = blinds;
1870  m_transitionDelay = (int)( (totalTime * 1000) / steps );
1871  } break;
1872 
1873  // box: inward / outward
