40 void OkularTTS::Private::setupIface()
41 {
42  if ( kspeech )
43  return;
44 
45  // If KTTSD not running, start it.
46  QDBusReply<bool> reply = QDBusConnection::sessionBus().interface()->isServiceRegistered( "org.kde.kttsd" );
47  bool kttsdactive = false;
48  if ( reply.isValid() )
49  kttsdactive = reply.value();
50  if ( !kttsdactive )
51  {
52  QString error;
53  if ( KToolInvocation::startServiceByDesktopName( "kttsd", QStringList(), &error ) )
54  {
55  emit q->errorMessage( i18n( "Starting Jovie Text-to-Speech service Failed: %1", error ) );
56  }
57  else
58  {
59  kttsdactive = true;
60  }
61  }
62  if ( kttsdactive )
63  {
64  // creating the connection to the kspeech interface
65  kspeech = new org::kde::KSpeech( "org.kde.kttsd", "/KSpeech", QDBusConnection::sessionBus() );
66  kspeech->setParent( q );
67  kspeech->setApplicationName( "Okular" );
68  connect( kspeech, SIGNAL(jobStateChanged(QString,int,int)),
69  q, SLOT(slotJobStateChanged(QString,int,int)) );
70  }
71 }

94 void OkularTTS::say( const QString &text )
95 {
96  if ( text.isEmpty() )
97  return;
98 
99  d->setupIface();
100  if ( d->kspeech )
101  {
102  QDBusReply< int > reply = d->kspeech->say( text, KSpeech::soPlainText );
103  if ( reply.isValid() )
104  {
105  d->jobs.insert( reply.value() );
106  emit hasSpeechs( true );
107  }
108  }

127 void OkularTTS::slotJobStateChanged( const QString &appId, int jobNum, int state )
128 {
129  // discard non ours job
130  if ( appId != QDBusConnection::sessionBus().baseService() || !d->kspeech )
131  return;
132 
133  switch ( state )
134  {
135  case KSpeech::jsDeleted:
136  d->jobs.remove( jobNum );
137  emit hasSpeechs( !d->jobs.isEmpty() );
138  break;
139  case KSpeech::jsFinished:
140  d->kspeech->removeJob( jobNum );
141  break;
142  }
143 }
