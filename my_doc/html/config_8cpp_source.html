100  status = 1; /* optimistic */

102 
103  while (1) {
104 
105  ptr = fgets (linebuf, sizeof (linebuf) - 1, fp);
106  if (ptr == NULL)
107  break;
108 
109  line_number += 1;
110  linebuf[strlen (linebuf) - 1] = 0; /* strip newline */
111  if (linebuf[strlen (linebuf) - 1] == '\r')
112  linebuf[strlen (linebuf) - 1] = 0; /* strip carriage return */
113 
114  /* fprintf (stderr, "%s:%d: line is '%s'\n", filename, line_number, linebuf); */
115 
116  linelen = strlen (linebuf);
117  for (buf_index = 0; linebuf[buf_index] != 0; buf_index++)
118  if (!isspace (linebuf[buf_index]))
119  break;
120 
121  if (linebuf[buf_index] == 0)
122  /* blank line */
123  continue;
124 
125  if ((strchr (COMMENT_CHARS, linebuf[0]) != NULL) ||
126  (strncmp (linebuf, "rem", 3) == 0) ||
127  (strncmp (linebuf, "REM", 3) == 0))
128  /* comment */
129  continue;
130 
131  /* At this point we have a valid thing */
132 
133  if (linebuf[buf_index] == SEGMENT_LEAD_CHAR) {
134  if ((str_end =
135  strchr (linebuf + buf_index + 1,
136  SEGMENT_END_CHAR)) == NULL) {
137  /* invalid segment line */
138  _plkr_message ("%s:%d: Invalid segment line '%s'",
139  filename, line_number, linebuf);

141  }
142  str_begin = linebuf + buf_index + 1;
143  for (charptr = str_begin; charptr < str_end; charptr++)
144  *charptr = tolower (*charptr);
145  *str_end = 0;
146  current_segment = GetOrCreateSegment (str_begin);
147  /* fprintf (stderr, "Current segment is now %p (%s)\n", current_segment, str_begin); */
148  if (current_option)
149  free (current_option);
150  current_option = NULL;
151 
152  }
153  else if ((linebuf[0] == ' ' || linebuf[0] == '\t')
154  && current_option != NULL) {
155  /* continuation line */
156  str_begin =
157  (char *) _plkr_RemoveFromTable (current_segment,
158  current_option);
159  for (str_end = linebuf + strlen (linebuf) - 1;
160  str_end > linebuf && isspace (*str_end); str_end--);
161  charptr =

164  strcpy (charptr, str_begin);

167  strncpy (charptr + len2 + 1, linebuf + buf_index,
168  str_end - (linebuf + buf_index));
169  charptr[len2 + (str_end - (linebuf + buf_index)) + 1] = '\0';
170  _plkr_AddToTable (current_segment, current_option, charptr);
171  free (str_begin);
172 
173  }
174  else if ((int)strcspn (linebuf, OPTION_SEPARATOR_CHARS) < linelen) {
175  /* possible option line */
176 
177  for (str_begin = linebuf + buf_index, ptr = str_begin;
178  isalnum (*ptr) || (*ptr == '.') || (*ptr == '_')
179  || (*ptr == '-'); ptr++);
180  if (ptr == str_begin) {
181  _plkr_message ("%s:%d: Invalid option line '%s'",
182  filename, line_number, linebuf);
183  goto error_exit;
184  }
185 
186  for (charptr = str_begin; charptr < ptr; charptr++)
187  *charptr = tolower (*charptr);
188  str_end = ptr;
189 
190  while (isspace (*ptr) && (*ptr != '\0'))
191  ptr++;
192 
193  if (strchr (OPTION_SEPARATOR_CHARS, *ptr) != NULL)
194  ptr++;
195  else {
196  _plkr_message ("%s:%d: Invalid option line '%s'",
197  filename, line_number, linebuf);
198  goto error_exit;
199  }
200 
201  while (isspace (*ptr) && (*ptr != '\0'))
202  ptr++;
203 
204  if (*ptr == 0) {
205  _plkr_message ("%s:%d: Invalid option line '%s'",
206  filename, line_number, linebuf);
207  goto error_exit;
208  }
209 
210  if (current_option)
211  free (current_option);
212  current_option =
213  _plkr_strndup (str_begin, str_end - str_begin);
214 
215  option_value = _plkr_strndup (ptr, strlen (ptr));
216 
217  ptr =
218  (char *) _plkr_RemoveFromTable (current_segment,
219  current_option);
220  if (ptr)
221  free (ptr);
222  _plkr_AddToTable (current_segment, current_option,
223  option_value);
224  /* fprintf (stderr, "Added value '%s' for option '%p:%s'\n", option_value, current_segment, current_option); */
225 
226  }
227  else {
228  _plkr_message ("%s:%d: Bad line '%s'", filename, line_number,
229  linebuf);
230  goto error_exit;
231  }
232  }
233 
234  good_exit:
235 
236  if (current_option)
237  free (current_option);
238  fclose (fp);
239  return status;
240 
241  error_exit:
242  status = 0;
243  goto good_exit;
244 }
