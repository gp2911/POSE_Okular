329 void dviRenderer::prescan_ParsePSSpecial(const QString& cp)
330 {
331 #ifdef DEBUG_PRESCAN
332  kDebug(kvs::dvi) << "PostScript-special, direct PostScript " << cp;
333 #endif
334 
335  // Unfortunately, in some TeX distribution the hyperref package uses
336  // the dvips driver by default, rather than the hypertex driver. As
337  // a result, the DVI files produced are full of PostScript that
338  // specifies links and anchors, and KDVI would call the ghostscript
339  // interpreter for every page which makes it really slow. This is a
340  // major nuisance, so that we try to filter and interpret the
341  // hypertex generated PostScript here.
342  if (cp.startsWith("ps:SDict begin")) {
343  // We suspect this may be hyperref generated nonsense. Let's check
344  // for some known code that hyperref generates.
345  if (cp == "ps:SDict begin H.S end")
346  return; // start of hyperref rectangle
347  if (cp == "ps:SDict begin H.R end")
348  return; // end of hyperref rectangle
349  if (cp.endsWith("H.A end"))
350  return; // end of hyperref anchor
351  if (cp.endsWith("H.L end"))
352  return; // end of hyperref link
353  if (cp.startsWith("ps:SDict begin /product where{pop product(Distiller)"))
354  return; // hyperref tries to work around Distiller bug
355  if (cp.startsWith("ps:SDict begin [") && cp.endsWith(" pdfmark end")) { // hyperref definition of link/anchor/bookmark/etc
356  if (cp.contains("/DEST")) { // The PostScript code defines an anchor
357  QString anchorName = cp.section('(', 1, 1).section(')', 0, 0);
358  Length l;
359  l.setLength_in_inch(currinf.data.dvi_v/(resolutionInDPI*shrinkfactor));
360  anchorList[anchorName] = Anchor(current_page+1, l);
361  }
362  // The PostScript code defines a bookmark
363  if (cp.contains("/Dest") && cp.contains("/Title")) {
364  const QString childrenNumberAndMoreStuff = cp.section('-', 1, 1); // Contains from the - symbol to the end of cp, effectively containing the number of children and some stuff after it
365  int indexOfFirstNonDigit = 0;
366  foreach(const QChar &c, childrenNumberAndMoreStuff) {
367  if (c.isDigit()) ++indexOfFirstNonDigit;
368  else break;
369  }
370  prebookmarks.append(PreBookmark(PDFencodingToQString(cp.section('(', 2, 2).section(')', 0, 0)),
371  cp.section('(', 1, 1).section(')', 0, 0),
372  childrenNumberAndMoreStuff.left(indexOfFirstNonDigit).toUInt()
373  ));
374  }
375  return;
376  }
377  }
378 
379  double PS_H = (currinf.data.dvi_h*300.0)/(65536*1200)-300;
380  double PS_V = (currinf.data.dvi_v*300.0)/1200 - 300;
381 
382  if (cp.indexOf("ps::[begin]", 0, Qt::CaseInsensitive) == 0) {
383  PostScriptOutPutString->append( QString(" %1 %2 moveto\n").arg(PS_H).arg(PS_V) );
384  PostScriptOutPutString->append( QString(" %1\n").arg(cp.mid(11)) );
385  } else {
386  if (cp.indexOf("ps::[end]", 0, Qt::CaseInsensitive) == 0) {
387  PostScriptOutPutString->append( QString(" %1\n").arg(cp.mid(9)) );
388  } else {
389  if (cp.indexOf("ps::", 0, Qt::CaseInsensitive) == 0) {
390  PostScriptOutPutString->append( QString(" %1\n").arg(cp.mid(4)) );
391  } else {
392  PostScriptOutPutString->append( QString(" %1 %2 moveto\n").arg(PS_H).arg(PS_V) );
393  PostScriptOutPutString->append( QString(" %1\n").arg(cp.mid(3)) );
394  }
395  }
396  }
397 }

522

524  // prescan phase, and NOT during rendering.
525

527  if (strncasecmp(cp, "papersize", 9) == 0) {
528  prescan_ParsePapersizeSpecial(special_command.mid(9));
529  return;

531

534  prescan_ParseBackgroundSpecial(special_command.mid(10));
535  return;
536  }
537

540  prescan_ParseHTMLAnchorSpecial(special_command.mid(14));

542  }
543

545  if (strncasecmp(cp, "header=", 7) == 0) {
546  prescan_ParsePSHeaderSpecial(special_command.mid(7));
547  return;
548  }
549

551  if (cp[0] == '!') {
552  prescan_ParsePSBangSpecial(special_command.mid(1));
553  return;

557  if (cp[0] == '"') {

559  return;
