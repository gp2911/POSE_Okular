102 void TextFormatProperty::apply( QTextCharFormat *format ) const
103 {
104  if ( !mFontName.isEmpty() ) {
105  if ( mStyleInformation ) {
106  const FontFormatProperty property = mStyleInformation->fontProperty( mFontName );
107  property.apply( format );
108  }
109  }
110 
111  if ( mFontWeight != -1 ) {
112  QFont font = format->font();
113  font.setWeight( mFontWeight );
114  format->setFont( font );
115  }
116 
117  if ( mHasFontSize ) {
118  QFont font = format->font();
119  font.setPointSize( mFontSize );
120  format->setFont( font );
121  }
122 
123  if ( mFontStyle != -1 ) {
124  QFont font = format->font();
125  font.setStyle( (QFont::Style)mFontStyle );
126  format->setFont( font );
127  }
128 
129  if ( mColor.isValid() )
130  format->setForeground( mColor );
131 
132  if ( mBackgroundColor.isValid() )
133  format->setBackground( mBackgroundColor );
