99 void PixmapPreviewSelector::iconComboChanged( const QString& icon )
100 {
101  int id = m_comboItems->findText( icon, Qt::MatchFixedString );
102  if ( id >= 0 )
103  {
104  m_icon = m_comboItems->itemData( id ).toString();
105  }
106  else
107  {
108  m_icon = icon;
109  }
110 
111  QPixmap pixmap = GuiUtils::loadStamp( m_icon, QSize(), m_previewSize );
112  const QRect cr = m_iconLabel->contentsRect();
113  if ( pixmap.width() > cr.width() || pixmap.height() > cr.height() )
114  pixmap = pixmap.scaled( cr.size(), Qt::KeepAspectRatio, Qt::SmoothTransformation );
115  m_iconLabel->setPixmap( pixmap );
116 
117  emit iconChanged( m_icon );
118 }

121 AnnotationWidget * AnnotationWidgetFactory::widgetFor( Okular::Annotation * ann )
122 {
123  switch ( ann->subType() )
124  {
125  case Okular::Annotation::AStamp:
126  return new StampAnnotationWidget( ann );
127  break;
128  case Okular::Annotation::AText:
129  return new TextAnnotationWidget( ann );
130  break;
131  case Okular::Annotation::ALine:
132  return new LineAnnotationWidget( ann );
133  break;
134  case Okular::Annotation::AHighlight:
135  return new HighlightAnnotationWidget( ann );
136  break;
137  case Okular::Annotation::AInk:
138  return new InkAnnotationWidget( ann );
139  break;
140  case Okular::Annotation::AGeom:
141  return new GeomAnnotationWidget( ann );
142  break;
143  case Okular::Annotation::AFileAttachment:
144  return new FileAttachmentAnnotationWidget( ann );
145  break;
146  case Okular::Annotation::ACaret:
147  return new CaretAnnotationWidget( ann );
148  break;
149  // shut up gcc
150  default:
151  ;
152  }
153  // cases not covered yet: return a generic widget
154  return new AnnotationWidget( ann );
155 }
