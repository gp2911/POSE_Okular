34 void AnnotatorEngine::decodeEvent( const QMouseEvent * mouseEvent, EventType * eventType, Button * button )
35 {
36  *eventType = AnnotatorEngine::Press;
37  if ( mouseEvent->type() == QEvent::MouseMove )
38  *eventType = AnnotatorEngine::Move;
39  else if ( mouseEvent->type() == QEvent::MouseButtonRelease )
40  *eventType = AnnotatorEngine::Release;
41 
42  *button = AnnotatorEngine::None;
43  const Qt::MouseButtons buttonState = ( *eventType == AnnotatorEngine::Move ) ? mouseEvent->buttons() : mouseEvent->button();
44  if ( buttonState == Qt::LeftButton )
45  *button = AnnotatorEngine::Left;
46  else if ( buttonState == Qt::RightButton )
47  *button = AnnotatorEngine::Right;
48 }

50 void AnnotatorEngine::decodeEvent( const QTabletEvent * tabletEvent, EventType * eventType, Button * button )
51 {
52  switch ( tabletEvent->type() )
53  {
54  case QEvent::TabletPress:
55  // Tablet press event is equivalent to pressing the left mouse button
56  *button = AnnotatorEngine::Left;
57  *eventType = AnnotatorEngine::Press;
58  break;
59  case QEvent::TabletRelease:
60  // Tablet release event is equivalent to releasing the left mouse button
61  *button = AnnotatorEngine::Left;
62  *eventType = AnnotatorEngine::Release;
63  break;
64  case QEvent::TabletMove:
65  // Tablet events are only routed if the pen is down so
66  // this is equivalent to the left mouse button being pressed
67  *button = AnnotatorEngine::Left;
68  *eventType = AnnotatorEngine::Move;
69  break;
70  default:
71  Q_ASSERT(false);
72  break;
73  }
74 }

98 {
99  // only proceed if pressing left button

101  return QRect();
102

104  if ( type == Press && points.isEmpty() )
105  {
106  lastPoint.x = nX;

108  totalRect.left = totalRect.right = lastPoint.x;
109  totalRect.top = totalRect.bottom = lastPoint.y;

111  }
112  // add a point to the path
113  else if ( type == Move && points.count() > 0 )
114  {
115  //double dist = hypot( nX - points.last().x, nY - points.last().y );
116  //if ( dist > 0.0001 )
117  //{
118  // append mouse position (as normalized point) to the list
119  Okular::NormalizedPoint nextPoint = Okular::NormalizedPoint( nX, nY );
120  points.append( nextPoint );
121  // update total rect
122  double dX = 2.0 / (double)xScale;
123  double dY = 2.0 / (double)yScale;
124  totalRect.left = qMin( totalRect.left, nX - dX );
125  totalRect.top = qMin( totalRect.top, nY - dY );
126  totalRect.right = qMax( nX + dX, totalRect.right );
127  totalRect.bottom = qMax( nY + dY, totalRect.bottom );
128  // paint the difference to previous full rect
129  Okular::NormalizedRect incrementalRect;
130  incrementalRect.left = qMin( nextPoint.x, lastPoint.x ) - dX;
131  incrementalRect.right = qMax( nextPoint.x, lastPoint.x ) + dX;
132  incrementalRect.top = qMin( nextPoint.y, lastPoint.y ) - dY;
133  incrementalRect.bottom = qMax( nextPoint.y, lastPoint.y ) + dY;
134  lastPoint = nextPoint;
135  return incrementalRect.geometry( (int)xScale, (int)yScale );
136  //}
137  }
138  // terminate process
139  else if ( type == Release && points.count() > 0 )
140  {
141  if ( points.count() < 2 )
142  points.clear();
143  else
144  m_creationCompleted = true;
145  return totalRect.geometry( (int)xScale, (int)yScale );

179 QList< Okular::Annotation* > SmoothPathEngine::end()
180 {
181  m_creationCompleted = false;
182 
183  // find out annotation's description node
184  if ( m_annotElement.isNull() )
185  return QList< Okular::Annotation* >();
186 
187  // find out annotation's type
188  Okular::Annotation * ann = 0;
189  QString typeString = m_annotElement.attribute( "type" );
190 
191  // create InkAnnotation from path
192  if ( typeString == "Ink" )
193  {
194  Okular::InkAnnotation * ia = new Okular::InkAnnotation();
195  ann = ia;
196  if ( m_annotElement.hasAttribute( "width" ) )
197  ann->style().setWidth( m_annotElement.attribute( "width" ).toDouble() );
198  // fill points
199  QList< QLinkedList<Okular::NormalizedPoint> > list = ia->inkPaths();
200  list.append( points );
201  ia->setInkPaths( list );
202  // set boundaries
203  ia->setBoundingRectangle( totalRect );
204  }
205 
206  // safety check
207  if ( !ann )
208  return QList< Okular::Annotation* >();
209 
210  // set common attributes
211  ann->style().setColor( m_annotElement.hasAttribute( "color" ) ?
212  m_annotElement.attribute( "color" ) : m_engineColor );
213  if ( m_annotElement.hasAttribute( "opacity" ) )
214  ann->style().setOpacity( m_annotElement.attribute( "opacity", "1.0" ).toDouble() );
215 
216  // return annotation
217  return QList< Okular::Annotation* >() << ann;
218 }
