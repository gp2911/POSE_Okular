296 static void
297 FillTable(struct tabent *T, int Size, const struct proto *P, int State)
298 {
299  int limit = 1 << Size;
300 
301  while (P->val) {
302  int width = P->val & 15;
303  int param = P->val >> 4;
304  int incr = 1 << width;
305  int code;
306  for (code = P->code; code < limit; code += incr) {
307  struct tabent *E = T+code;
308  E->State = State;
309  E->Width = width;
310  E->Param = param;
311  }
312  P++;
313  }
314 }
