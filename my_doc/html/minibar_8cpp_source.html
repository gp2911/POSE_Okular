74 void MiniBarLogic::notifySetup( const QVector< Okular::Page * > & pageVector, int setupFlags )
75 {
76  // only process data when document changes
77  if ( !( setupFlags & Okular::DocumentObserver::DocumentChanged ) )
78  return;
79 
80  // if document is closed or has no pages, hide widget
81  const int pages = pageVector.count();
82  if ( pages < 1 )
83  {
84  foreach ( MiniBar *miniBar, m_miniBars )
85  {
86  miniBar->setEnabled( false );
87  }
88  return;
89  }
90 
91  bool labelsDiffer = false;
92  foreach(const Okular::Page * page, pageVector)
93  {
94  if (!page->label().isEmpty())
95  {
96  if (page->label().toInt() != (page->number() + 1))
97  {
98  labelsDiffer = true;
99  }
100  }
101  }
102 
103  const QString pagesString = QString::number( pages );
104 
105  foreach ( MiniBar *miniBar, m_miniBars )
106  {
107  // resize width of widgets
108  miniBar->resizeForPage( pages );
109 
110  // update child widgets
111  miniBar->m_pageLabelEdit->setPageLabels( pageVector );
112  miniBar->m_pageNumberEdit->setPagesNumber( pages );
113  miniBar->m_pagesButton->setText( pagesString );
114  miniBar->m_prevButton->setEnabled( false );
115  miniBar->m_nextButton->setEnabled( false );
116  miniBar->m_pageLabelEdit->setVisible( labelsDiffer );
117  miniBar->m_pageNumberLabel->setVisible( labelsDiffer );
118  miniBar->m_pageNumberEdit->setVisible( !labelsDiffer );
119 
120  miniBar->resize( miniBar->minimumSizeHint() );
121 
122  miniBar->setEnabled( true );
123  }
124 }

387 void ProgressWidget::paintEvent( QPaintEvent * e )
388 {
389  QPainter p( this );
390 
391  if ( m_progressPercentage < 0.0 )
392  {
393  p.fillRect( rect(), palette().color( QPalette::Active, QPalette::HighlightedText ) );
394  return;
395  }
396 
397  // find out the 'fill' and the 'clear' rectangles
398  int w = width(),
399  h = height(),
400  l = (int)( (float)w * m_progressPercentage );
401  QRect cRect = ( QApplication::isRightToLeft() ? QRect( 0, 0, w - l, h ) : QRect( l, 0, w - l, h ) ).intersect( e->rect() );
402  QRect fRect = ( QApplication::isRightToLeft() ? QRect( w - l, 0, l, h ) : QRect( 0, 0, l, h ) ).intersect( e->rect() );
403 
404  QPalette pal = palette();
405  // paint clear rect
406  if ( cRect.isValid() )
407  p.fillRect( cRect, pal.color( QPalette::Active, QPalette::HighlightedText ) );
408  // draw a frame-like outline

503 void PagesEdit::setText( const QString & newText )
504 {
505  // call default handler if hasn't focus
506  if ( !hasFocus() )
507  {
508  KLineEdit::setText( newText );
509  }
510  // else preserve existing selection
511  else
512  {
513  // save selection and adapt it to the new text length
514  int selectionLength = selectedText().length();
515  const bool allSelected = ( selectionLength == text().length() );
516  if ( allSelected )
517  {
518  KLineEdit::setText( newText );
519  selectAll();
520  }
521  else
522  {
523  int newSelectionStart = newText.length() - text().length() + selectionStart();
524  if ( newSelectionStart < 0 )
525  {
526  // the new text is shorter than the old one, and the front part, which is "cut off", is selected
527  // shorten the selection accordingly
528  selectionLength += newSelectionStart;
529  newSelectionStart = 0;
530  }
531  KLineEdit::setText( newText );
532  setSelection( newSelectionStart, selectionLength );
533  }
534  }
535 }
