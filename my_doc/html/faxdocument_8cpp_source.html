21 /* rearrange input bits into t16bits lsb-first chunks */
22 static void normalize( struct pagenode *pn, int revbits, int swapbytes, size_t length )
23 {
24  t32bits *p = (t32bits *) pn->data;
25 
26  switch ( (revbits<<1) | swapbytes )
27  {
28  case 0:
29  break;
30  case 1:
31  for ( ; length; length -= 4)
32  {
33  t32bits t = *p;
34  *p++ = ((t & 0xff00ff00) >> 8) | ((t & 0x00ff00ff) << 8);
35  }
36  break;
37  case 2:
38  for ( ; length; length -= 4)
39  {
40  t32bits t = *p;
41  t = ((t & 0xf0f0f0f0) >> 4) | ((t & 0x0f0f0f0f) << 4);
42  t = ((t & 0xcccccccc) >> 2) | ((t & 0x33333333) << 2);
43  *p++ = ((t & 0xaaaaaaaa) >> 1) | ((t & 0x55555555) << 1);
44  }
45  break;
46  case 3:
47  for ( ; length; length -= 4)
48  {
49  t32bits t = *p;
50  t = ((t & 0xff00ff00) >> 8) | ((t & 0x00ff00ff) << 8);
51  t = ((t & 0xf0f0f0f0) >> 4) | ((t & 0x0f0f0f0f) << 4);
52  t = ((t & 0xcccccccc) >> 2) | ((t & 0x33333333) << 2);
53  *p++ = ((t & 0xaaaaaaaa) >> 1) | ((t & 0x55555555) << 1);
54  }
55  }
56 }

70 /* get compressed data into memory */
71 static unsigned char* getstrip( pagenode *pn, int strip )
72 {
73  size_t offset, roundup;
74  unsigned char *data;
75 
76  union { t16bits s; unsigned char b[2]; } so;
77 #define ShortOrder so.b[1]
78  so.s = 1; /* XXX */
79 
80  QFile file( pn->filename );
81  if ( !file.open( QIODevice::ReadOnly ) )
82  return 0;
83 
84  if ( pn->strips == 0 )
85  {
86  offset = 0;
87  pn->length = file.size();
88  }
89  else if ( strip < pn->nstrips )
90  {
91  offset = pn->strips[ strip ].offset;
92  pn->length = pn->strips[ strip ].size;
93  }
94  else
95  return 0;
96 
97  /* round size to full boundary plus t32bits */
98  roundup = (pn->length + 7) & ~3;
99 
100  data = new uchar[ roundup ];
101  /* clear the last 2 t32bits, to force the expander to terminate
102  even if the file ends in the middle of a fax line */
103  *((t32bits *) data + roundup/4 - 2) = 0;
104  *((t32bits *) data + roundup/4 - 1) = 0;
105 
106  /* we expect to get it in one gulp... */
107  if ( !file.seek(offset) || (size_t) file.read( (char *)data, pn->length ) != pn->length )
108  {
109  delete [] data;
110  return 0;
111  }
112  file.close();
113 
114  pn->data = (t16bits *)data;
115 
116  if ( pn->strips == 0 && memcmp( data, FAXMAGIC, sizeof( FAXMAGIC ) - 1 ) == 0 )
117  {
118  /* handle ghostscript / PC Research fax file */
119  pn->length -= 64;
120  pn->vres = data[29];
121  pn->data += 32;
122  roundup -= 64;
123  }
124 
125  normalize( pn, !pn->lsbfirst, ShortOrder, roundup );
126  if ( pn->size.height() == 0 )
127  pn->size.setHeight( G3count( pn, pn->expander == g32expand ) );
128 
129  if ( pn->size.height() == 0 )
130  {
131  delete [] data;
132  pn->data = 0;
133  return 0;
134  }
135 
136  if ( pn->strips == 0 )
137  pn->rowsperstrip = pn->size.height();
138 
139  pn->dataOrig = (t16bits *)data;
140 
141  return data;
142 }

144 static void draw_line( pixnum *run, int lineNum, pagenode *pn )
145 {
146  t32bits *p, *p1; /* p - current line, p1 - low-res duplicate */
147  pixnum *r; /* pointer to run-lengths */
148  t32bits pix; /* current pixel value */
149  t32bits acc; /* pixel accumulator */
150  int nacc; /* number of valid bits in acc */
151  int tot; /* total pixels in line */
152  int n;
153 
154  lineNum += pn->stripnum * pn->rowsperstrip;
155  if ( lineNum >= pn->size.height() )
156  return;
157 
158  p = (t32bits *)(pn->imageData + lineNum*(2-pn->vres)*pn->bytes_per_line);
159  p1 =(t32bits *)(pn->vres ? 0 : p + pn->bytes_per_line/sizeof(*p));
160 
161  r = run;
162  acc = 0;
163  nacc = 0;
164  pix = pn->inverse ? ~0 : 0;
165  tot = 0;
166  while ( tot < pn->size.width() )
167  {
168  n = *r++;
169  tot += n;
170  /* Watch out for buffer overruns, e.g. when n == 65535. */
171  if ( tot > pn->size.width() )
172  break;
173  if ( pix )
174  acc |= (~(t32bits)0 >> nacc);
175  else if ( nacc )
176  acc &= (~(t32bits)0 << (32 - nacc));
177  else
178  acc = 0;
179  if ( nacc + n < 32 )
180  {
181  nacc += n;
182  pix = ~pix;
183  continue;
184  }
185  *p++ = acc;
186  if ( p1 )
187  *p1++ = acc;
188  n -= 32 - nacc;
189  while ( n >= 32 )
190  {
191  n -= 32;
192  *p++ = pix;
193  if ( p1 )
194  *p1++ = pix;
195  }
196  acc = pix;
197  nacc = n;
198  pix = ~pix;
199  }
200  if ( nacc )
201  {
202  *p++ = acc;
203  if ( p1 )
204  *p1++ = acc;
205  }
206 }

263 bool FaxDocument::load()
264 {
265  fax_init_tables();
266 
267  bool ok = get_image( &(d->mPageNode) );
268  if ( !ok )
269  return false;
270 
271  // byte-swapping the image
272  int height = d->mPageNode.size.height();
273  int bytes_per_line = d->mPageNode.size.width()/8;
274 
275  QByteArray bytes( height * bytes_per_line, 0 );
276  for ( int y= height - 1; y >= 0; --y )
277  {
278  quint32 offset = y * bytes_per_line;
279  quint32 *source = (quint32 *) (d->mPageNode.imageData + offset);
280  quint32 *dest = (quint32 *) (bytes.data() + offset);
281  for ( int x= (bytes_per_line/4) - 1; x >= 0; --x )
282  {
283  quint32 dv = 0, sv = *source;
284  for ( int bit = 32; bit > 0; --bit )
285  {
286  dv <<= 1;
287  dv |= sv&1;
288  sv >>= 1;
289  }
290  *dest = dv;
291  ++dest;
292  ++source;
293  }
294  }
295 
296  // convert it into a QImage
297  QImage img( (uchar*)bytes.data(), d->mPageNode.size.width(), d->mPageNode.size.height(), QImage::Format_MonoLSB );
298  img.setColor( 0, qRgb( 255, 255, 255 ) );
299  img.setColor( 1, qRgb( 0, 0, 0 ) );
300 
301  d->mPageNode.image = img.copy().scaled( img.width(), img.height() * 1.5 );
302 
303  return true;
304 }
