122 void SidebarDelegate::paint( QPainter *painter, const QStyleOptionViewItem &option, const QModelIndex &index ) const
123 {
124  QBrush backBrush;
125  QColor foreColor;
126  bool disabled = false;
127  bool hover = false;
128 
129  if ( !( option.state & QStyle::State_Enabled ) )
130  {
131  backBrush = m_windowBackground->brush(QPalette::Disabled);
132  foreColor = m_windowForeground->brush(QPalette::Disabled).color();
133  disabled = true;
134  }
135  else if ( option.state & ( QStyle::State_HasFocus | QStyle::State_Selected ) )
136  {
137  backBrush = m_selectionBackground->brush(option.palette);
138  foreColor = m_selectionForeground->brush(option.palette).color();
139  }
140  else if ( option.state & QStyle::State_MouseOver )
141  {
142  backBrush = m_selectionBackground->brush(option.palette).color().light( 115 );
143  foreColor = m_selectionForeground->brush(option.palette).color();
144  hover = true;
145  }
146  else /*if ( option.state & QStyle::State_Enabled )*/
147  {
148  backBrush = m_windowBackground->brush(option.palette);
149  foreColor = m_windowForeground->brush(option.palette).color();
150  }
151  QStyle *style = QApplication::style();
152  QStyleOptionViewItemV4 opt( option );
153  // KStyle provides an "hover highlight" effect for free;
154  // but we want that for non-KStyle-based styles too
155  if ( !style->inherits( "KStyle" ) && hover )
156  {
157  Qt::BrushStyle bs = opt.backgroundBrush.style();
158  if ( bs > Qt::NoBrush && bs < Qt::TexturePattern )
159  opt.backgroundBrush = opt.backgroundBrush.color().light( 115 );
160  else
161  opt.backgroundBrush = backBrush;
162  }
163  painter->save();
164  style->drawPrimitive( QStyle::PE_PanelItemViewItem, &opt, painter, 0 );
165  painter->restore();
166  QIcon icon = index.data( Qt::DecorationRole ).value< QIcon >();
167  if ( !icon.isNull() )
168  {
169  QPoint iconpos(
170  ( option.rect.width() - option.decorationSize.width() ) / 2,
171  ITEM_MARGIN_TOP
172  );
173  iconpos += option.rect.topLeft();
174  QIcon::Mode iconmode = disabled ? QIcon::Disabled : QIcon::Normal;
175  painter->drawPixmap( iconpos, icon.pixmap( option.decorationSize, iconmode ) );
176  }
177 
178  if ( m_showText )
179  {
180  QString text = index.data( Qt::DisplayRole ).toString();
181  QRect fontBoundaries = QFontMetrics( option.font ).boundingRect( text );
182  QPoint textPos(
183  ITEM_MARGIN_LEFT + ( option.rect.width() - ITEM_MARGIN_LEFT - ITEM_MARGIN_RIGHT - fontBoundaries.width() ) / 2,
184  ITEM_MARGIN_TOP + option.decorationSize.height() + ITEM_PADDING
185  );
186  fontBoundaries.translate( -fontBoundaries.topLeft() );
187  fontBoundaries.translate( textPos );
188  fontBoundaries.translate( option.rect.topLeft() );
189  painter->setPen( foreColor );
190  painter->drawText( fontBoundaries, Qt::AlignCenter, text );
191  }
192 }

195 {

331 QModelIndex SidebarListWidget::moveCursor( QAbstractItemView::CursorAction cursorAction, Qt::KeyboardModifiers modifiers )
332 {
333  Q_UNUSED( modifiers )
334  QModelIndex oldindex = currentIndex();
335  QModelIndex newindex = oldindex;
336  switch ( cursorAction )
337  {
338  case MoveUp:
339  case MovePrevious:
340  {
341  int row = oldindex.row() - 1;
342  while ( row > -1 && !( model()->index( row, 0 ).flags() & Qt::ItemIsSelectable ) ) --row;
343  if ( row > -1 )
344  newindex = model()->index( row, 0 );
345  break;
346  }
347  case MoveDown:
348  case MoveNext:
349  {
350  int row = oldindex.row() + 1;
351  int max = model()->rowCount();
352  while ( row < max && !( model()->index( row, 0 ).flags() & Qt::ItemIsSelectable ) ) ++row;
353  if ( row < max )
354  newindex = model()->index( row, 0 );
355  break;
356  }
357  case MoveHome:
358  case MovePageUp:
359  {
360  int row = 0;
361  while ( row < oldindex.row() && !( model()->index( row, 0 ).flags() & Qt::ItemIsSelectable ) ) ++row;
362  if ( row < oldindex.row() )
363  newindex = model()->index( row, 0 );
364  break;
365  }
366  case MoveEnd:
367  case MovePageDown:
368  {
369  int row = model()->rowCount() - 1;
370  while ( row > oldindex.row() && !( model()->index( row, 0 ).flags() & Qt::ItemIsSelectable ) ) --row;
371  if ( row > oldindex.row() )
372  newindex = model()->index( row, 0 );
373  break;
374  }
375  // no navigation possible for these
376  case MoveLeft:
377  case MoveRight:
378  break;
379  }
380 
381  // dirty hack to change item when the key cursor changes item
382  if ( oldindex != newindex )
383  {
384  emit itemClicked( itemFromIndex( newindex ) );
385  }
386  return newindex;
387 }

416 void Sidebar::Private::adjustListSize( bool recalc, bool expand )
417 {
418  QRect bottomElemRect(
419  QPoint( 0, 0 ),
420  list->sizeHintForIndex( list->model()->index( list->count() - 1, 0 ) )
421  );
422  if ( recalc )
423  {
424  int w = 0;
425  for ( int i = 0; i < list->count(); ++i )
426  {
427  QSize s = list->sizeHintForIndex( list->model()->index( i, 0 ) );
428  if ( s.width() > w )
429  w = s.width();
430  }
431  bottomElemRect.setWidth( w );
432  }
433  bottomElemRect.translate( 0, bottomElemRect.height() * ( list->count() - 1 ) );
434  itemsHeight = bottomElemRect.height() * list->count();
435  list->setMinimumHeight( itemsHeight + list->frameWidth() * 2 );
436  int curWidth = list->minimumWidth();
437  int newWidth = expand
438  ? qMax( bottomElemRect.width() + list->frameWidth() * 2, curWidth )
439  : qMin( bottomElemRect.width() + list->frameWidth() * 2, curWidth );
440  list->setFixedWidth( newWidth );
441 }
