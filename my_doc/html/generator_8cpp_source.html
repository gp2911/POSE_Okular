84 void GeneratorPrivate::pixmapGenerationFinished()
85 {
86  Q_Q( Generator );
87  PixmapRequest *request = mPixmapGenerationThread->request();
88  mPixmapGenerationThread->endGeneration();
89 
90  QMutexLocker locker( threadsLock() );
91  mPixmapReady = true;
92 
93  if ( m_closing )
94  {
95  delete request;
96  if ( mTextPageReady )
97  {
98  locker.unlock();
99  m_closingLoop->quit();
100  }
101  return;
102  }
103 
104  const QImage& img = mPixmapGenerationThread->image();
105  request->page()->setPixmap( request->observer(), new QPixmap( QPixmap::fromImage( img ) ), request->normalizedRect() );
106  const int pageNumber = request->page()->number();
107 
108  if ( mPixmapGenerationThread->calcBoundingBox() )
109  q->updatePageBoundingBox( pageNumber, mPixmapGenerationThread->boundingBox() );
110  q->signalPixmapRequestDone( request );
111 }

183 bool Generator::closeDocument()

188 
189  d->threadsLock()->lock();
190  if ( !( d->mPixmapReady && d->mTextPageReady ) )
191  {

193  d->m_closingLoop = &loop;

219 void Generator::generatePixmap( PixmapRequest *request )
220 {
221  Q_D( Generator );
222  d->mPixmapReady = false;
223 
224  const bool calcBoundingBox = !request->isTile() && !request->page()->isBoundingBoxKnown();
225 
226  if ( request->asynchronous() && hasFeature( Threaded ) )
227  {
228  d->pixmapGenerationThread()->startGeneration( request, calcBoundingBox );
229 
234  if ( hasFeature( TextExtraction ) && !request->page()->hasTextPage() && canGenerateTextPage() && !d->m_closing ) {
235  d->mTextPageReady = false;
236  d->textPageGenerationThread()->startGeneration( request->page() );
237  }
238 
239  return;
240  }
241 
242  const QImage& img = image( request );
243  request->page()->setPixmap( request->observer(), new QPixmap( QPixmap::fromImage( img ) ), request->normalizedRect() );
244  const int pageNumber = request->page()->number();
245 
246  d->mPixmapReady = true;
247 
248  signalPixmapRequestDone( request );
249  if ( calcBoundingBox )
250  updatePageBoundingBox( pageNumber, Utils::imageBoundingBox( &img ) );
251 }
