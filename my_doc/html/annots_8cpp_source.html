86 void PopplerAnnotationProxy::notifyAddition( Okular::Annotation *okl_ann, int page )
87 {
88 #ifdef HAVE_POPPLER_0_20
89  // Export annotation to DOM
90  QDomDocument doc;
91  QDomElement dom_ann = doc.createElement( "root" );
92  Okular::AnnotationUtils::storeAnnotation( okl_ann, dom_ann, doc );
93 
94  QMutexLocker ml(mutex);
95 
96  // Create poppler annotation
97  Poppler::Annotation *ppl_ann = Poppler::AnnotationUtils::createAnnotation( dom_ann );
98 
99  // Poppler doesn't render StampAnnotations yet
100  if ( ppl_ann->subType() != Poppler::Annotation::AStamp )
101  okl_ann->setFlags( okl_ann->flags() | Okular::Annotation::ExternallyDrawn );
102 
103  // Poppler stores highlight points in swapped order
104  if ( ppl_ann->subType() == Poppler::Annotation::AHighlight )
105  {
106  Poppler::HighlightAnnotation * hlann = static_cast<Poppler::HighlightAnnotation*>( ppl_ann );
107  QList<Poppler::HighlightAnnotation::Quad> quads = hlann->highlightQuads();
108  QMutableListIterator<Poppler::HighlightAnnotation::Quad> it( quads );
109  while ( it.hasNext() )
110  {
111  Poppler::HighlightAnnotation::Quad &q = it.next();
112  QPointF t;
113  t = q.points[3];
114  q.points[3] = q.points[0];
115  q.points[0] = t;
116  t = q.points[2];
117  q.points[2] = q.points[1];
118  q.points[1] = t;
119  }
120  hlann->setHighlightQuads( quads );
121  }
122 
123  // Bind poppler object to page
124  Poppler::Page *ppl_page = ppl_doc->page( page );
125  ppl_page->addAnnotation( ppl_ann );
126  delete ppl_page;
127 
128  // Set pointer to poppler annotation as native Id
129  okl_ann->setNativeId( qVariantFromValue( ppl_ann ) );
130  okl_ann->setDisposeDataFunction( disposeAnnotation );
131 
132  kDebug(PDFGenerator::PDFDebug) << okl_ann->uniqueName();
133 #endif
134 }

136 void PopplerAnnotationProxy::notifyModification( const Okular::Annotation *okl_ann, int page, bool appearanceChanged )
137 {
138 #ifdef HAVE_POPPLER_0_20
139  Q_UNUSED( page );
140  Q_UNUSED( appearanceChanged );
141 
142  Poppler::Annotation *ppl_ann = qvariant_cast<Poppler::Annotation*>( okl_ann->nativeId() );
143 
144  if ( !ppl_ann ) // Ignore non-native annotations
145  return;
146 
147  QMutexLocker ml(mutex);
148 
149  if ( okl_ann->flags() & Okular::Annotation::BeingMoved )
150  {
151  // Okular ui already renders the annotation on its own
152  ppl_ann->setFlags( Poppler::Annotation::Hidden );
153  return;
154  }
155 
156  // Set basic properties
157  // Note: flags and boundary must be set first in order to correctly handle
158  // FixedRotation annotations.
159  ppl_ann->setFlags(maskExportedFlags( okl_ann->flags() ));
160  ppl_ann->setBoundary(normRectToRectF( okl_ann->boundingRectangle() ));
161 
162  ppl_ann->setAuthor( okl_ann->author() );
163  ppl_ann->setContents( okl_ann->contents() );
164 
165  // Set style
166  Poppler::Annotation::Style s;
167  s.setColor( okl_ann->style().color() );
168  s.setWidth( okl_ann->style().width() );
169  s.setOpacity( okl_ann->style().opacity() );
170  ppl_ann->setStyle( s );
171 
172  // Set type-specific properties (if any)
173  switch ( ppl_ann->subType() )
174  {
175  case Poppler::Annotation::AText:
176  {
177  const Okular::TextAnnotation * okl_txtann = static_cast<const Okular::TextAnnotation*>(okl_ann);
178  Poppler::TextAnnotation * ppl_txtann = static_cast<Poppler::TextAnnotation*>(ppl_ann);
179  ppl_txtann->setTextIcon( okl_txtann->textIcon() );
180  ppl_txtann->setTextFont( okl_txtann->textFont() );
181  ppl_txtann->setInplaceAlign( okl_txtann->inplaceAlignment() );
182  ppl_txtann->setCalloutPoints( QVector<QPointF>() );
183  ppl_txtann->setInplaceIntent( (Poppler::TextAnnotation::InplaceIntent)okl_txtann->inplaceIntent() );
184  break;
185  }
186  case Poppler::Annotation::ALine:
187  {
188  const Okular::LineAnnotation * okl_lineann = static_cast<const Okular::LineAnnotation*>(okl_ann);
189  Poppler::LineAnnotation * ppl_lineann = static_cast<Poppler::LineAnnotation*>(ppl_ann);
190  QLinkedList<QPointF> points;
191  foreach ( const Okular::NormalizedPoint &p, okl_lineann->linePoints() )
192  points.append(normPointToPointF( p ));
193  ppl_lineann->setLinePoints( points );
194  ppl_lineann->setLineStartStyle( (Poppler::LineAnnotation::TermStyle)okl_lineann->lineStartStyle() );
195  ppl_lineann->setLineEndStyle( (Poppler::LineAnnotation::TermStyle)okl_lineann->lineEndStyle() );
196  ppl_lineann->setLineClosed( okl_lineann->lineClosed() );
197  ppl_lineann->setLineInnerColor( okl_lineann->lineInnerColor() );
198  ppl_lineann->setLineLeadingForwardPoint( okl_lineann->lineLeadingForwardPoint() );
199  ppl_lineann->setLineLeadingBackPoint( okl_lineann->lineLeadingBackwardPoint() );
200  ppl_lineann->setLineShowCaption( okl_lineann->showCaption() );
201  ppl_lineann->setLineIntent( (Poppler::LineAnnotation::LineIntent)okl_lineann->lineIntent() );
202  break;
203  }
204  case Poppler::Annotation::AGeom:
205  {
206  const Okular::GeomAnnotation * okl_geomann = static_cast<const Okular::GeomAnnotation*>(okl_ann);
207  Poppler::GeomAnnotation * ppl_geomann = static_cast<Poppler::GeomAnnotation*>(ppl_ann);
208  ppl_geomann->setGeomType( (Poppler::GeomAnnotation::GeomType)okl_geomann->geometricalType() );
209  ppl_geomann->setGeomInnerColor( okl_geomann->geometricalInnerColor() );
210  break;
211  }
212  case Poppler::Annotation::AHighlight:
213  {
214  const Okular::HighlightAnnotation * okl_hlann = static_cast<const Okular::HighlightAnnotation*>(okl_ann);
215  Poppler::HighlightAnnotation * ppl_hlann = static_cast<Poppler::HighlightAnnotation*>(ppl_ann);
216  ppl_hlann->setHighlightType( (Poppler::HighlightAnnotation::HighlightType)okl_hlann->highlightType() );
217  break;
218  }
219  case Poppler::Annotation::AStamp:
220  {
221  const Okular::StampAnnotation * okl_stampann = static_cast<const Okular::StampAnnotation*>(okl_ann);
222  Poppler::StampAnnotation * ppl_stampann = static_cast<Poppler::StampAnnotation*>(ppl_ann);
223  ppl_stampann->setStampIconName( okl_stampann->stampIconName() );
224  break;
225  }
226  case Poppler::Annotation::AInk:
227  {
228  const Okular::InkAnnotation * okl_inkann = static_cast<const Okular::InkAnnotation*>(okl_ann);
229  Poppler::InkAnnotation * ppl_inkann = static_cast<Poppler::InkAnnotation*>(ppl_ann);
230  QList< QLinkedList<QPointF> > paths;
231  foreach ( const QLinkedList<Okular::NormalizedPoint> &path, okl_inkann->inkPaths() )
232  {
233  QLinkedList<QPointF> points;
234  foreach ( const Okular::NormalizedPoint &p, path )
235  points.append(normPointToPointF( p ));
236  paths.append( points );
237  }
238  ppl_inkann->setInkPaths( paths );
239  break;
240  }
241  default:
242  kDebug() << "Type-specific property modification is not implemented for this annotation type";
243  break;
244  }
245 
246  kDebug(PDFGenerator::PDFDebug) << okl_ann->uniqueName();
247 #endif
248 }

271 Okular::Annotation* createAnnotationFromPopplerAnnotation( Poppler::Annotation *ann, bool *doDelete )
272 {
273  Okular::Annotation *annotation = 0;
274  *doDelete = true;
275  bool tieToOkularAnn = false;
276  bool externallyDrawn = false;
277  switch ( ann->subType() )
278  {
279  case Poppler::Annotation::AFileAttachment:
280  {
281  Poppler::FileAttachmentAnnotation * attachann = static_cast< Poppler::FileAttachmentAnnotation * >( ann );
282  Okular::FileAttachmentAnnotation * f = new Okular::FileAttachmentAnnotation();
283  annotation = f;
284  tieToOkularAnn = true;
285  *doDelete = false;
286 
287  f->setFileIconName( attachann->fileIconName() );
288  f->setEmbeddedFile( new PDFEmbeddedFile( attachann->embeddedFile() ) );
289 
290  break;
291  }
292  case Poppler::Annotation::ASound:
293  {
294  Poppler::SoundAnnotation * soundann = static_cast< Poppler::SoundAnnotation * >( ann );
295  Okular::SoundAnnotation * s = new Okular::SoundAnnotation();
296  annotation = s;
297 
298  s->setSoundIconName( soundann->soundIconName() );
299  s->setSound( createSoundFromPopplerSound( soundann->sound() ) );
300 
301  break;
302  }
303  case Poppler::Annotation::AMovie:
304  {
305  Poppler::MovieAnnotation * movieann = static_cast< Poppler::MovieAnnotation * >( ann );
306  Okular::MovieAnnotation * m = new Okular::MovieAnnotation();
307  annotation = m;
308  tieToOkularAnn = true;
309  *doDelete = false;
310 
311  m->setMovie( createMovieFromPopplerMovie( movieann->movie() ) );
312 
313  break;
314  }
315 #ifdef HAVE_POPPLER_0_22
316  case Poppler::Annotation::AWidget:
317  {
318  annotation = new Okular::WidgetAnnotation();
319  break;
320  }
321 #endif
322 #ifdef HAVE_POPPLER_0_20
323  case Poppler::Annotation::AScreen:
324  {
325 #ifdef HAVE_POPPLER_0_22
326  Okular::ScreenAnnotation * m = new Okular::ScreenAnnotation();
327  annotation = m;
328  tieToOkularAnn = true;
329  *doDelete = false;
330 #else
331  Poppler::ScreenAnnotation * screenann = static_cast< Poppler::ScreenAnnotation * >( ann );
332  Okular::MovieAnnotation * m = new Okular::MovieAnnotation();
333  annotation = m;
334 
335  m->setMovie( createMovieFromPopplerScreen( screenann->action() ) );
336 #endif
337  break;
338  }
339  case Poppler::Annotation::AText:
340  case Poppler::Annotation::ALine:
341  case Poppler::Annotation::AGeom:
342  case Poppler::Annotation::AHighlight:
343  case Poppler::Annotation::AInk:
344  {
345  externallyDrawn = true;
346  /* fallback */
347  }
348  case Poppler::Annotation::AStamp:
349  {
350  tieToOkularAnn = true;
351  *doDelete = false;
352  /* fallback */
353  }
354 #endif
355  default:
356  {
357  // this is uber ugly but i don't know a better way to do it without introducing a poppler::annotation dependency on core
358  QDomDocument doc;
359  QDomElement root = doc.createElement( "root" );
360  doc.appendChild( root );
361  Poppler::AnnotationUtils::storeAnnotation( ann, root, doc );
362  annotation = Okular::AnnotationUtils::createAnnotation( root );
363  break;
364  }
365  }
366  if ( annotation )
367  {
368  // the Contents field might have lines separated by \r
369  QString contents = ann->contents();
370  contents.replace( QLatin1Char( '\r' ), QLatin1Char( '\n' ) );
371 
372  annotation->setAuthor( ann->author() );
373  annotation->setContents( contents );
374  annotation->setUniqueName( ann->uniqueName() );
375  annotation->setModificationDate( ann->modificationDate() );
376  annotation->setCreationDate( ann->creationDate() );
377  annotation->setFlags( ann->flags() | Okular::Annotation::External );
378  annotation->setBoundingRectangle( Okular::NormalizedRect::fromQRectF( ann->boundary() ) );
379 
380  if (externallyDrawn)
381  annotation->setFlags( annotation->flags() | Okular::Annotation::ExternallyDrawn );
382 
383  // Poppler stores highlight points in swapped order
384  if ( annotation->subType() == Okular::Annotation::AHighlight )
385  {
386  Okular::HighlightAnnotation * hlann = static_cast<Okular::HighlightAnnotation*>( annotation );
387  QList<Okular::HighlightAnnotation::Quad> &quads = hlann->highlightQuads();
388  for (QList<Okular::HighlightAnnotation::Quad>::iterator it = quads.begin(); it != quads.end(); ++it)
389  {
390  Okular::NormalizedPoint t;
391  t = it->point( 3 );
392  it->setPoint( it->point(0), 3 );
393  it->setPoint( t, 0 );
394  t = it->point( 2 );
395  it->setPoint( it->point(1), 2 );
396  it->setPoint( t, 1 );
397  }
398  }
399 
400  if ( annotation->subType() == Okular::Annotation::AText )
401  {
402  Okular::TextAnnotation * txtann = static_cast<Okular::TextAnnotation*>( annotation );
403 
404  if ( txtann->textType() == Okular::TextAnnotation::Linked )
405  {
406  Poppler::TextAnnotation * ppl_txtann = static_cast<Poppler::TextAnnotation*>( ann );
407 
408  // Poppler and Okular assume a different default icon name in XML
409  // We re-read it via getter, which always tells the right one
410  txtann->setTextIcon( ppl_txtann->textIcon() );
411  }
412  }
413 
414  // TODO clone style
415  // TODO clone window
416  // TODO clone revisions
417  if ( tieToOkularAnn )
418  {
419  annotation->setNativeId( qVariantFromValue( ann ) );
420  annotation->setDisposeDataFunction( disposeAnnotation );
421  }
422  }
423  return annotation;
