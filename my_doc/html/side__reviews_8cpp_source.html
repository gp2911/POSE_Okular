38 class TreeView : public QTreeView
39 {

41  TreeView( Okular::Document *document, QWidget *parent = 0 )
42  : QTreeView( parent ), m_document( document )
43  {
44  }

46  protected:
47  virtual void paintEvent( QPaintEvent *event )
48  {
49  bool hasAnnotations = false;
50  for ( uint i = 0; i < m_document->pages(); ++i )
51  if ( m_document->page( i )->hasAnnotations() ) {
52  hasAnnotations = true;
53  break;
54  }
55  if ( !hasAnnotations ) {
56  QPainter p( viewport() );
57  p.setRenderHint( QPainter::Antialiasing, true );
58  p.setClipRect( event->rect() );
59 
60  QTextDocument document;
61  document.setHtml( i18n( "<div align=center><h3>No annotations</h3>"
62  "To create new annotations press F6 or select <i>Tools -&gt; Review</i>"
63  " from the menu.</div>" ) );
64  document.setTextWidth( width() - 50 );
65 
66  const uint w = document.size().width() + 20;
67  const uint h = document.size().height() + 20;
68 
69  p.setBrush( palette().background() );
70  p.translate( 0.5, 0.5 );
71  p.drawRoundRect( 15, 15, w, h, (8*200)/w, (8*200)/h );
72  p.translate( 20, 20 );
73  document.drawContents( &p );
74 
75  } else {
76  QTreeView::paintEvent( event );
77  }
78  }
79 
80  private:
81  Okular::Document *m_document;

206 void Reviews::activated( const QModelIndex &index )
207 {
208  const QModelIndex authorIndex = m_authorProxy->mapToSource( index );
209  const QModelIndex filterIndex = m_groupProxy->mapToSource( authorIndex );
210  const QModelIndex annotIndex = m_filterProxy->mapToSource( filterIndex );
211 
212  Okular::Annotation *annotation = m_model->annotationForIndex( annotIndex );
213  if ( !annotation )
214  return;
215 
216  int pageNumber = m_model->data( annotIndex, AnnotationModel::PageRole ).toInt();
217  const Okular::Page * page = m_document->page( pageNumber );
218 
219  // calculating the right coordinates to center the view on the annotation
220  QRect rect = Okular::AnnotationUtils::annotationGeometry( annotation, page->width(), page->height() );
221  Okular::NormalizedRect nr( rect, (int)page->width(), (int)page->height() );
222  // set the viewport parameters
223  Okular::DocumentViewport vp;
224  vp.pageNumber = pageNumber;
225  vp.rePos.enabled = true;
226  vp.rePos.pos = Okular::DocumentViewport::Center;
227  vp.rePos.normalizedX = ( nr.right + nr.left ) / 2.0;
228  vp.rePos.normalizedY = ( nr.bottom + nr.top ) / 2.0;
229  // setting the viewport
230  m_document->setViewport( vp, 0, true );
231 }

233 QModelIndexList Reviews::retrieveAnnotations(const QModelIndex& idx) const
234 {
235  QModelIndexList ret;
236  if ( idx.isValid() )
237  {
238  if ( idx.model()->hasChildren( idx ) )
239  {
240  int rowCount = idx.model()->rowCount( idx );
241  for ( int i = 0; i < rowCount; i++ )
242  {
243  ret += retrieveAnnotations( idx.child( i, idx.column() ) );
244  }
245  }
246  else
247  {
248  ret += idx;
249  }
250  }
251 
252  return ret;
253 }

255 void Reviews::contextMenuRequested( const QPoint &pos )
256 {
257  AnnotationPopup popup( m_document, AnnotationPopup::SingleAnnotationMode, this );
258  connect( &popup, SIGNAL(openAnnotationWindow(Okular::Annotation*,int)),
259  this, SIGNAL(openAnnotationWindow(Okular::Annotation*,int)) );
260 
261  QModelIndexList indexes = m_view->selectionModel()->selectedIndexes();
262  Q_FOREACH ( const QModelIndex &index, indexes )
263  {
264  QModelIndexList annotations = retrieveAnnotations(index);
265  Q_FOREACH ( const QModelIndex &idx, annotations )
266  {
267  const QModelIndex authorIndex = m_authorProxy->mapToSource( idx );
268  const QModelIndex filterIndex = m_groupProxy->mapToSource( authorIndex );
269  const QModelIndex annotIndex = m_filterProxy->mapToSource( filterIndex );
270  Okular::Annotation *annotation = m_model->annotationForIndex( annotIndex );
271  if ( annotation )
272  {
273  const int pageNumber = m_model->data( annotIndex, AnnotationModel::PageRole ).toInt();
274  popup.addAnnotation( annotation, pageNumber );
275  }
276  }
277  }
278 
279  popup.exec( m_view->viewport()->mapToGlobal( pos ) );
