174 void Shell::openUrl( const KUrl & url )
175 {
176  const int activeTab = m_tabWidget->currentIndex();
177  if ( activeTab < m_tabs.size() )
178  {
179  KParts::ReadWritePart* const activePart = m_tabs[activeTab].part;
180  if( !activePart->url().isEmpty() )
181  {
182  if( m_unique )
183  {
184  KMessageBox::error(this, i18n("Can't open more than one document in the unique Okular instance."));
185  }
186  else
187  {
188  if( dynamic_cast<Okular::ViewerInterface *>(activePart)->openNewFilesInTabs() )
189  {
190  openNewTab( url );
191  setActiveTab( m_tabs.size()-1 );
192  }
193  else
194  {
195  Shell* newShell = new Shell();
196  newShell->openUrl( url );
197  newShell->show();
198  }
199  }
200  }
201  else
202  {
203  m_tabWidget->setTabText( activeTab, url.fileName() );
204  if ( m_args ){
205  KDocumentViewer* const doc = qobject_cast<KDocumentViewer*>(activePart);
206  if ( doc && m_args->isSet( "presentation" ) )
207  doc->startPresentation();
208  if ( m_args->isSet( "print" ) )
209  QMetaObject::invokeMethod( activePart, "enableStartWithPrint" );
210  }
211  bool openOk = activePart->openUrl( url );
212  const bool isstdin = url.fileName( KUrl::ObeyTrailingSlash ) == QLatin1String( "-" );
213  if ( !isstdin )
214  {
215  if ( openOk )
216  {
217 #ifdef KActivities_FOUND
218  if ( !m_activityResource )
219  m_activityResource = new KActivities::ResourceInstance( window()->winId(), this );
220 
221  m_activityResource->setUri( url );
222 #endif
223  m_recent->addUrl( url );
224  }
225  else
226  m_recent->removeUrl( url );
227  }
228  }
229  }
230 }

315 QStringList Shell::fileFormats() const
316 {
317  QStringList supportedPatterns;
318 
319  QString constraint( "(Library == 'okularpart')" );
320  QLatin1String basePartService( "KParts/ReadOnlyPart" );
321  KService::List offers = KServiceTypeTrader::self()->query( basePartService, constraint );
322  KService::List::ConstIterator it = offers.constBegin(), itEnd = offers.constEnd();
323  for ( ; it != itEnd; ++it )
324  {
325  KService::Ptr service = *it;
326  QStringList mimeTypes = service->serviceTypes();
327  foreach ( const QString& mimeType, mimeTypes )
328  if ( mimeType != basePartService )
329  supportedPatterns.append( mimeType );
330  }
331 
332  return supportedPatterns;
333 }

335 void Shell::fileOpen()
336 {
337  // this slot is called whenever the File->Open menu is selected,
338  // the Open shortcut is pressed (usually CTRL+O) or the Open toolbar
339  // button is clicked
340  const int activeTab = m_tabWidget->currentIndex();
341  if ( !m_fileformatsscanned )
342  {
343  const KDocumentViewer* const doc = qobject_cast<KDocumentViewer*>(m_tabs[activeTab].part);
344  if ( doc )
345  m_fileformats = doc->supportedMimeTypes();
346 
347  if ( m_fileformats.isEmpty() )
348  m_fileformats = fileFormats();
349 
350  m_fileformatsscanned = true;
351  }
352 
353  QString startDir;
354  const KParts::ReadWritePart* const curPart = m_tabs[activeTab].part;
355  if ( curPart->url().isLocalFile() )
356  startDir = curPart->url().toLocalFile();
357  KFileDialog dlg( startDir, QString(), this );
358  dlg.setOperationMode( KFileDialog::Opening );
359 
360  // A directory may be a document. E.g. comicbook generator.
361  if ( m_fileformats.contains( "inode/directory" ) )
362  dlg.setMode( dlg.mode() | KFile::Directory );
363 
364  if ( m_fileformatsscanned && m_fileformats.isEmpty() )
365  dlg.setFilter( i18n( "*|All Files" ) );
366  else
367  dlg.setMimeFilter( m_fileformats );
368  dlg.setCaption( i18n( "Open Document" ) );
369  if ( !dlg.exec() )
370  return;
371  KUrl url = dlg.selectedUrl();
372  if ( !url.isEmpty() )
373  {
374  openUrl( url );
375  }
376 }
