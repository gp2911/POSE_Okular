231 void ThumbnailList::notifySetup( const QVector< Okular::Page * > & pages, int setupFlags )
232 {
233  // if there was a widget selected, save its pagenumber to restore
234  // its selection (if available in the new set of pages)
235  int prevPage = -1;
236  if ( !( setupFlags & Okular::DocumentObserver::DocumentChanged ) && d->m_selected )
237  {
238  prevPage = d->m_selected->page()->number();
239  } else
240  prevPage = d->m_document->viewport().pageNumber;
241 
242  // delete all the Thumbnails
243  QVector<ThumbnailWidget *>::const_iterator tIt = d->m_thumbnails.constBegin(), tEnd = d->m_thumbnails.constEnd();
244  for ( ; tIt != tEnd; ++tIt )
245  delete *tIt;
246  d->m_thumbnails.clear();
247  d->m_visibleThumbnails.clear();
248  d->m_selected = 0;
249  d->m_mouseGrabItem = 0;
250 
251  if ( pages.count() < 1 )
252  {
253  widget()->resize( 0, 0 );
254  return;
255  }
256 
257  // show pages containing hilighted text or bookmarked ones
258  //RESTORE THIS int flags = Okular::Settings::filterBookmarks() ? Okular::Page::Bookmark : Okular::Page::Highlight;
259 
260  // if no page matches filter rule, then display all pages
261  QVector< Okular::Page * >::const_iterator pIt = pages.constBegin(), pEnd = pages.constEnd();
262  bool skipCheck = true;
263  for ( ; pIt != pEnd ; ++pIt )
264  //if ( (*pIt)->attributes() & flags )
265  if ( (*pIt)->hasHighlights( SW_SEARCH_ID ) )
266  skipCheck = false;
267 
268  // generate Thumbnails for the given set of pages
269  const int width = viewport()->width();
270  int height = 0;
271  int centerHeight = 0;
272  for ( pIt = pages.constBegin(); pIt != pEnd ; ++pIt )
273  //if ( skipCheck || (*pIt)->attributes() & flags )
274  if ( skipCheck || (*pIt)->hasHighlights( SW_SEARCH_ID ) )
275  {
276  ThumbnailWidget * t = new ThumbnailWidget( d, *pIt );
277  t->move(0, height);
278  // add to the internal queue
279  d->m_thumbnails.push_back( t );
280  // update total height (asking widget its own height)
281  t->resizeFitWidth( width );
282  // restoring the previous selected page, if any
283  if ( (*pIt)->number() < prevPage )
284  {
285  centerHeight = height + t->height() + KDialog::spacingHint()/2;
286  }
287  if ( (*pIt)->number() == prevPage )
288  {
289  d->m_selected = t;
290  d->m_selected->setSelected( true );
291  centerHeight = height + t->height() / 2;
292  }
293  height += t->height() + KDialog::spacingHint();
294  }
295 
296  // update scrollview's contents size (sets scrollbars limits)
297  height -= KDialog::spacingHint();
298  widget()->resize( width, height );
299 
300  // enable scrollbar when there's something to scroll
301  verticalScrollBar()->setEnabled( viewport()->height() < height );
302  verticalScrollBar()->setValue(centerHeight - viewport()->height() / 2);
303 
304  // request for thumbnail generation
305  d->delayedRequestVisiblePixmaps( 200 );
306 }

309 {
310  Q_UNUSED( previousPage )
311 
312  // skip notifies for the current page (already selected)
313  if ( d->m_selected && d->m_selected->pageNumber() == currentPage )
314  return;
315 
316  // deselect previous thumbnail
317  if ( d->m_selected )
318  d->m_selected->setSelected( false );
319  d->m_selected = 0;

321  // select the page with viewport and ensure it's centered in the view
322  d->m_vectorIndex = 0;

414 int ThumbnailListPrivate::getNewPageOffset(int n, ThumbnailListPrivate::ChangePageDirection dir) const
415 {
416  int reason = 1;
417  int facingFirst = 0; //facingFirstCentered cornercase
418  if ( Okular::Settings::viewMode() == Okular::Settings::EnumViewMode::Facing )
419  reason = 2;
420  else if (Okular::Settings::viewMode() == Okular::Settings::EnumViewMode::FacingFirstCentered )
421  {
422  facingFirst = 1;
423  reason = 2;
424  }
425  else if ( Okular::Settings::viewMode() == Okular::Settings::EnumViewMode::Summary )
426  reason = 3;
427  if ( dir == ThumbnailListPrivate::Up )
428  {
429  if ( facingFirst && n == 1 )
430  return -1;
431  return -reason;
432  }
433  if ( dir == ThumbnailListPrivate::Down )
434  return reason;
435  if ( dir == ThumbnailListPrivate::Left && reason > 1 && (n + facingFirst) % reason )
436  return -1;
437  if ( dir == ThumbnailListPrivate::Right && reason > 1 && (n + 1 + facingFirst) % reason )
438  return 1;
439  return 0;
440 }

462 ThumbnailListPrivate::ChangePageDirection ThumbnailListPrivate::forwardTrack(const QPoint &point, const QSize &r )
463 {
464  Okular::DocumentViewport vp = m_document->viewport();
465  const double deltaX = (double)point.x() / r.width(),
466  deltaY = (double)point.y() / r.height();
467  vp.rePos.normalizedX -= deltaX;
468  vp.rePos.normalizedY -= deltaY;
469  if ( vp.rePos.normalizedY > 1.0 )
470  return ThumbnailListPrivate::Down;
471  if ( vp.rePos.normalizedY < 0.0 )

474  return ThumbnailListPrivate::Right;
475  if ( vp.rePos.normalizedX < 0.0 )
476  return ThumbnailListPrivate::Left;
477  vp.rePos.enabled = true;
478  m_document->setViewport( vp );
479  return ThumbnailListPrivate::Null;

512  }

519  }
520  else if ( keyEvent->key() == Qt::Key_PageUp )
521  verticalScrollBar()->triggerAction( QScrollBar::SliderPageStepSub );
522  else if ( keyEvent->key() == Qt::Key_PageDown )
523  verticalScrollBar()->triggerAction( QScrollBar::SliderPageStepAdd );
524  else if ( keyEvent->key() == Qt::Key_Home )
525  nextPage = d->m_thumbnails[ 0 ]->pageNumber();
526  else if ( keyEvent->key() == Qt::Key_End )
527  nextPage = d->m_thumbnails[ d->m_thumbnails.count() - 1 ]->pageNumber();
528 
529  if ( nextPage == -1 )
530  return keyEvent->ignore();
531 
532  keyEvent->accept();

554 void ThumbnailListPrivate::viewportResizeEvent( QResizeEvent * e )
555 {
556  if ( m_thumbnails.count() < 1 || width() < 1 )
557  return;
558 
559  // if width changed resize all the Thumbnails, reposition them to the
560  // right place and recalculate the contents area
561  if ( e->size().width() != e->oldSize().width() )
562  {
563  // runs the timer avoiding a thumbnail regeneration by 'contentsMoving'
564  delayedRequestVisiblePixmaps( 2000 );
565 
566  // resize and reposition items
567  const int newWidth = q->viewport()->width();
568  int newHeight = 0;
569  QVector<ThumbnailWidget *>::const_iterator tIt = m_thumbnails.constBegin(), tEnd = m_thumbnails.constEnd();
570  for ( ; tIt != tEnd; ++tIt )
571  {
572  ThumbnailWidget *t = *tIt;
573  t->move(0, newHeight);
574  t->resizeFitWidth( newWidth );
575  newHeight += t->height() + KDialog::spacingHint();
576  }
577 
578  // update scrollview's contents size (sets scrollbars limits)
579  newHeight -= KDialog::spacingHint();
580  const int oldHeight = q->widget()->height();
581  const int oldYCenter = q->verticalScrollBar()->value() + q->viewport()->height() / 2;
582  q->widget()->resize( newWidth, newHeight );
583 
584  // enable scrollbar when there's something to scroll
585  q->verticalScrollBar()->setEnabled( q->viewport()->height() < newHeight );
586 
587  // ensure that what was visibile before remains visible now
588  q->ensureVisible( 0, int( (qreal)oldYCenter * q->widget()->height() / oldHeight ), 0, q->viewport()->height() / 2 );
589  }
590  else if ( e->size().height() <= e->oldSize().height() )
591  return;
592 
593  // invalidate the bookmark overlay
594  if ( m_bookmarkOverlay )
595  {
596  delete m_bookmarkOverlay;
597  m_bookmarkOverlay = 0;
598  }
599 
600  // update Thumbnails since width has changed or height has increased
601  delayedRequestVisiblePixmaps( 500 );
602 }

621 void ThumbnailListPrivate::slotRequestVisiblePixmaps( int /*newContentsY*/ )
622 {
623  // if an update is already scheduled or the widget is hidden, don't proceed
624  if ( ( m_delayTimer && m_delayTimer->isActive() ) || q->isHidden() )
625  return;
626 
627  // scroll from the top to the last visible thumbnail
628  m_visibleThumbnails.clear();
629  QLinkedList< Okular::PixmapRequest * > requestedPixmaps;
630  QVector<ThumbnailWidget *>::const_iterator tIt = m_thumbnails.constBegin(), tEnd = m_thumbnails.constEnd();
631  const QRect viewportRect = q->viewport()->rect().translated( q->horizontalScrollBar()->value(), q->verticalScrollBar()->value() );
632  for ( ; tIt != tEnd; ++tIt )
633  {
634  ThumbnailWidget * t = *tIt;

780 void ThumbnailListPrivate::mouseMoveEvent( QMouseEvent * e )
781 {
782  if ( e->buttons() == Qt::NoButton )
783  return e->ignore();
784  // no item under the mouse or previously selected
785  if ( !m_mouseGrabItem )
786  return e->ignore();
787  const QRect r = m_mouseGrabItem->rect();
788  if ( !m_mouseGrabPos.isNull() )
789  {
790  const QPoint mousePos = e->pos();
791  const QPoint delta = m_mouseGrabPos - mousePos;
792  m_mouseGrabPos = e->pos();
793  // don't handle the mouse move, forward it to the thumbnail list
794  ThumbnailListPrivate::ChangePageDirection direction;
795  if (( direction = forwardTrack( delta, r.size() ))
796  != ThumbnailListPrivate::Null )
797  {
798  // Changing the selected page
799  const int offset = getNewPageOffset( m_pageCurrentlyGrabbed, direction );
800  const ThumbnailWidget *newThumb = getThumbnailbyOffset( m_pageCurrentlyGrabbed, offset );
801  if ( !newThumb )
802  return;
803  int newPageOn = newThumb->pageNumber();
804  if ( newPageOn == m_pageCurrentlyGrabbed || newPageOn < 0 ||
805  newPageOn >= (int)m_document->pages() )
806  {
807  return;
808  }
809  Okular::DocumentViewport vp = m_document->viewport();
810  const float origNormalX = vp.rePos.normalizedX;
811  const float origNormalY = vp.rePos.normalizedY;
812  vp = Okular::DocumentViewport( newPageOn );
813  vp.rePos.normalizedX = origNormalX;
814  vp.rePos.normalizedY = origNormalY;
815  if ( direction == ThumbnailListPrivate::Up )
816  {
817  vp.rePos.normalizedY = 1.0;
818  if ( Okular::Settings::viewMode() ==
819  Okular::Settings::EnumViewMode::FacingFirstCentered
820  && !newPageOn)
821  {
822  if ( m_pageCurrentlyGrabbed == 1 )
823  vp.rePos.normalizedX = origNormalX - 0.5;
824  else
825  vp.rePos.normalizedX = origNormalX + 0.5;
826  if ( vp.rePos.normalizedX < 0.0 )
827  vp.rePos.normalizedX = 0.0;
828  else if ( vp.rePos.normalizedX > 1.0 )
829  vp.rePos.normalizedX = 1.0;
830  }
831  }
832  else if ( direction == ThumbnailListPrivate::Down )
833  {
834  vp.rePos.normalizedY = 0.0;
835  if ( Okular::Settings::viewMode() ==
836  Okular::Settings::EnumViewMode::FacingFirstCentered
837  && !m_pageCurrentlyGrabbed)
838  {
839  if ( origNormalX < 0.5 )
840  {
841  vp = Okular::DocumentViewport( --newPageOn );
842  vp.rePos.normalizedX = origNormalX + 0.5;
843  }
844  else
845  vp.rePos.normalizedX = origNormalX - 0.5;
846  if ( vp.rePos.normalizedX < 0.0 )
847  vp.rePos.normalizedX = 0.0;
848  else if ( vp.rePos.normalizedX > 1.0 )
849  vp.rePos.normalizedX = 1.0;
850  }
851  }
852  else if ( Okular::Settings::viewMode() != Okular::Settings::EnumViewMode::Single )
853  {
854  if ( direction == ThumbnailListPrivate::Left )
855  vp.rePos.normalizedX = 1.0;
856  else
857  vp.rePos.normalizedX = 0.0;
858  }
859  vp.rePos.pos = Okular::DocumentViewport::Center;
860  vp.rePos.enabled = true;
861  m_document->setViewport( vp );
862  m_mouseGrabPos.setX( 0 );
863  m_mouseGrabPos.setY( 0 );
864  m_pageCurrentlyGrabbed = newPageOn;
865  m_mouseGrabItem = getPageByNumber( m_pageCurrentlyGrabbed );
866  }
867  // wrap mouse from top to bottom
868  const QRect mouseContainer = KGlobalSettings::desktopGeometry( this );
869  QPoint currentMousePos = QCursor::pos();
870  if ( currentMousePos.y() <= mouseContainer.top() + 4 )
871  {
872  currentMousePos.setY( mouseContainer.bottom() - 5 );
873  QCursor::setPos( currentMousePos );
874  m_mouseGrabPos.setX( 0 );
875  m_mouseGrabPos.setY( 0 );
876  }
877  // wrap mouse from bottom to top
878  else if ( currentMousePos.y() >= mouseContainer.bottom() - 4 )
879  {
880  currentMousePos.setY( mouseContainer.top() + 5 );
881  QCursor::setPos( currentMousePos );
882  m_mouseGrabPos.setX( 0 );
883  m_mouseGrabPos.setY( 0 );
884  }
885  }
886  else
887  {
888  setCursor( Qt::ClosedHandCursor );
889  m_mouseGrabPos = e->pos();
890  }
891 }

921 void ThumbnailWidget::paint( QPainter &p, const QRect &_clipRect )
922 {
923  const int width = m_pixmapWidth + m_margin;
924  QRect clipRect = _clipRect;
925  const QPalette pal = m_parent->palette();
926 
927  // draw the bottom label + highlight mark
928  const QColor fillColor = m_selected ? pal.color( QPalette::Active, QPalette::Highlight ) : pal.color( QPalette::Active, QPalette::Base );
929  p.fillRect( clipRect, fillColor );
930  p.setPen( m_selected ? pal.color( QPalette::Active, QPalette::HighlightedText ) : pal.color( QPalette::Active, QPalette::Text ) );
931  p.drawText( 0, m_pixmapHeight + (m_margin - 3), width, m_labelHeight, Qt::AlignCenter, QString::number( m_labelNumber ) );
932 
933  // draw page outline and pixmap
934  if ( clipRect.top() < m_pixmapHeight + m_margin )
935  {
936  // if page is bookmarked draw a colored border
937  const bool isBookmarked = m_parent->m_document->bookmarkManager()->isBookmarked( pageNumber() );
938  // draw the inner rect
939  p.setPen( isBookmarked ? QColor( 0xFF8000 ) : Qt::black );
940  p.drawRect( m_margin/2 - 1, m_margin/2 - 1, m_pixmapWidth + 1, m_pixmapHeight + 1 );
941  // draw the clear rect
942  p.setPen( isBookmarked ? QColor( 0x804000 ) : pal.color( QPalette::Active, QPalette::Base ) );
943  // draw the bottom and right shadow edges
944  if ( !isBookmarked )
945  {
946  int left, right, bottom, top;
947  left = m_margin/2 + 1;
948  right = m_margin/2 + m_pixmapWidth + 1;
949  bottom = m_pixmapHeight + m_margin/2 + 1;
950  top = m_margin/2 + 1;
951  p.setPen( Qt::gray );
952  p.drawLine( left, bottom, right, bottom );
953  p.drawLine( right, top, right, bottom );
954  }
955 
956  // draw the page using the shared PagePainter class
957  p.translate( m_margin/2, m_margin/2 );
958  clipRect.translate( -m_margin/2, -m_margin/2 );
959  clipRect = clipRect.intersect( QRect( 0, 0, m_pixmapWidth, m_pixmapHeight ) );
960  if ( clipRect.isValid() )
961  {
962  int flags = PagePainter::Accessibility | PagePainter::Highlights |
963  PagePainter::Annotations;
964  PagePainter::paintPageOnPainter( &p, m_page, m_parent->q, flags, m_pixmapWidth, m_pixmapHeight, clipRect );
965  }
966 
967  if ( !m_visibleRect.isNull() )
968  {
969  p.save();
970  p.setPen( QColor( 255, 255, 0, 200 ) );
971  p.setBrush( QColor( 0, 0, 0, 100 ) );
972  p.drawRect( m_visibleRect.geometry( m_pixmapWidth, m_pixmapHeight ).adjusted( 0, 0, -1, -1 ) );
973  p.restore();
974  }
975 
976  // draw the bookmark overlay on the top-right corner
977  const QPixmap * bookmarkPixmap = m_parent->getBookmarkOverlay();
978  if ( isBookmarked && bookmarkPixmap )
979  {
980  int pixW = bookmarkPixmap->width(),
981  pixH = bookmarkPixmap->height();
982  clipRect = clipRect.intersect( QRect( m_pixmapWidth - pixW, 0, pixW, pixH ) );
983  if ( clipRect.isValid() )
984  p.drawPixmap( m_pixmapWidth - pixW, -pixH/8, *bookmarkPixmap );
985  }
986  }
987 }
