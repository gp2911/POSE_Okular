193 void fontPool::locateFonts()
194 {
195  kpsewhichOutput.clear();
196 
197  // First, we try and find those fonts which exist on disk
198  // already. If virtual fonts are found, they will add new fonts to
199  // the list of fonts whose font files need to be located, so that we
200  // repeat the lookup.
201  bool vffound;
202  do {
203  vffound = false;
204  locateFonts(false, false, &vffound);
205  } while(vffound);
206 
207  // If still not all fonts are found, look again, this time with
208  // on-demand generation of PK fonts enabled.
209  if (!areFontsLocated())
210  locateFonts(true, false);
211 
212  // If still not all fonts are found, we look for TFM files as a last
213  // resort, so that we can at least draw filled rectangles for
214  // characters.
215  if (!areFontsLocated())
216  locateFonts(false, true);
217 
218  // If still not all fonts are found, we give up. We mark all fonts
219  // as 'located', so that we won't look for them any more, and
220  // present an error message to the user.
221  if (!areFontsLocated()) {
222  markFontsAsLocated();
223  emit error(i18n("Okular was not able to locate all the font files "
224  "which are necessary to display the current DVI file. "
225  "Your document might be unreadable.\n"
226  "PATH: %1\nkpswhich: %2", getenv("PATH"), kpsewhichOutput), -1);
227  }
228 }

231 void fontPool::locateFonts(bool makePK, bool locateTFMonly, bool *virtualFontsFound)
232 {
233  // Set up the kpsewhich process. If pass == 0, look for vf-fonts and
234  // disable automatic font generation as vf-fonts can't be
235  // generated. If pass == 0, ennable font generation, if it was
236  // enabled globally.
237 
238  // Now generate the command line for the kpsewhich
239  // program. Unfortunately, this can be rather long and involved...
240  QStringList kpsewhich_args;
241  kpsewhich_args << "--dpi" << "1200"
242  << "--mode" << "lexmarks";
243 
244  // Disable automatic pk-font generation.
245  kpsewhich_args << (makePK ? "--mktex" : "--no-mktex") << "pk";
246 
247  // Names of fonts that shall be located
248  quint16 numFontsInJob = 0;
249  QList<TeXFontDefinition*>::const_iterator cit_fontp = fontList.constBegin();
250  for (; cit_fontp != fontList.constEnd(); ++cit_fontp) {
251  TeXFontDefinition *fontp = *cit_fontp;
252  if (!fontp->isLocated()) {
253  numFontsInJob++;
254 
255  if (locateTFMonly == true)
256  kpsewhich_args << QString("%1.tfm").arg(fontp->fontname);
257  else {
258 #ifdef HAVE_FREETYPE
259  if (FreeType_could_be_loaded == true) {
260  const QString &filename = fontsByTeXName.findFileName(fontp->fontname);
261  if (!filename.isEmpty())
262  kpsewhich_args << QString("%1").arg(filename);
263  }
264 #endif
265  kpsewhich_args << QString("%1.vf").arg(fontp->fontname)
266  << QString("%1.1200pk").arg(fontp->fontname);
267  }
268  }
269  }
270 
271  if (numFontsInJob == 0)
272  return;
273 
274  progress.setTotalSteps(numFontsInJob, &kpsewhich_);
275 
276  // Now run... kpsewhich. In case of error, kick up a fuss.
277  // This string is not going to be quoted, as it might be were it
278  // a real command line, but who cares?
279  const QString kpsewhich_exe = "kpsewhich";
280  kpsewhichOutput +=
281  "<p><b>" + kpsewhich_exe + ' ' + kpsewhich_args.join(" ") + "</b></p>";
282 
283  const QString importanceOfKPSEWHICH =
284  i18n("<p>Okular relies on the <b>kpsewhich</b> program to locate font files "
285  "on your hard disk and to generate PK fonts, if necessary.</p>");
286 
287  kpsewhich_.start(kpsewhich_exe, kpsewhich_args,
288  QIODevice::ReadOnly|QIODevice::Text);
289  if (!kpsewhich_.waitForStarted()) {
290  QApplication::restoreOverrideCursor();
291  emit error(i18n("There were problems running 'kpsewhich'. As a result, "
292  "some font files could not be located, and your document might be unreadable.\n"
293  "Possible reason: The kpsewhich program is perhaps not installed on your system, or it "
294  "cannot be found in the current search path.\n"
295  "PATH: %1\nkpswhich: %2", getenv("PATH"), kpsewhichOutput), -1);
296 
297  // This makes sure the we don't try to run kpsewhich again
298  markFontsAsLocated();
299  return;
300  }
301 
302  // We wait here while the external program runs concurrently.
303  // Every 200ms we call processEvents() to keep the GUI updated.
304  // This is important, e.g. for the progress dialog that is
305  // shown when PK fonts are generated by MetaFont.
306  while(!kpsewhich_.waitForFinished(200))
307  qApp->processEvents();
308  progress.hide();
309 
310  // Handle fatal errors.
311  int const kpsewhich_exit_code = kpsewhich_.exitCode();
312  if (kpsewhich_exit_code < 0) {
313  emit warning(i18n("The font generation by 'kpsewhich' was aborted (exit code %1, error %2). As a "
314  "result, some font files could not be located, and your document might be unreadable.",
315  kpsewhich_exit_code, kpsewhich_.errorString()), -1);
316 
317  // This makes sure the we don't try to run kpsewhich again
318  if (makePK == false)
319  markFontsAsLocated();
320  }
321 
322  // Create a list with all filenames found by the kpsewhich program.
323  const QStringList fileNameList =
324  QString(kpsewhich_.readAll()).split('\n', QString::SkipEmptyParts);
325 
326  // Now associate the file names found with the fonts
327  QList<TeXFontDefinition*>::iterator it_fontp = fontList.begin();
328  for (; it_fontp != fontList.end(); ++it_fontp) {
329  TeXFontDefinition *fontp = *it_fontp;
330 
331  if (fontp->filename.isEmpty() == true) {
332  QStringList matchingFiles;
333 #ifdef HAVE_FREETYPE
334  const QString &fn = fontsByTeXName.findFileName(fontp->fontname);
335  if (!fn.isEmpty())
336  matchingFiles = fileNameList.filter(fn);
337 #endif
338  if (matchingFiles.isEmpty() == true)
339  matchingFiles += fileNameList.filter("/"+fontp->fontname+".");
340 
341  if (matchingFiles.isEmpty() != true) {
342 #ifdef DEBUG_FONTPOOL
343  kDebug(kvs::dvi) << "Associated " << fontp->fontname << " to " << matchingFiles.first();
344 #endif
345  QString fname = matchingFiles.first();
346  fontp->fontNameReceiver(fname);
347  fontp->flags |= TeXFontDefinition::FONT_KPSE_NAME;
348  if (fname.endsWith(".vf")) {
349  if (virtualFontsFound != 0)
350  *virtualFontsFound = true;
351  // Constructing a virtual font will most likely insert other
352  // fonts into the fontList. After that, fontList.next() will
353  // no longer work. It is therefore safer to start over.
354  it_fontp=fontList.begin();
355  continue;
356  }
357  }
358  } // of if (fontp->filename.isEmpty() == true)
359  }
360 }

457 void fontPool::mf_output_receiver()
458 {
459  const QString output_data =
460  QString::fromLocal8Bit(kpsewhich_.readAllStandardError());
461 
462  kpsewhichOutput.append(output_data);
463  MetafontOutput.append(output_data);
464 
465  // We'd like to print only full lines of text.
466  int numleft;
467  while( (numleft = MetafontOutput.indexOf('\n')) != -1) {
468  QString line = MetafontOutput.left(numleft+1);
469 #ifdef DEBUG_FONTPOOL
470  kDebug(kvs::dvi) << "MF OUTPUT RECEIVED: " << line;
471 #endif
472  // If the Output of the kpsewhich program contains a line starting
473  // with "kpathsea:", this means that a new MetaFont-run has been
474  // started. We filter these lines out and update the display
475  // accordingly.
476  int startlineindex = line.indexOf("kpathsea:");
477  if (startlineindex != -1) {
478  int endstartline = line.indexOf("\n",startlineindex);
479  QString startLine = line.mid(startlineindex,endstartline-startlineindex);
480 
481  // The last word in the startline is the name of the font which we
482  // are generating. The second-to-last word is the resolution in
483  // dots per inch. Display this info in the text label below the
484  // progress bar.
485  int lastblank = startLine.lastIndexOf(' ');
486  QString fontName = startLine.mid(lastblank+1);
487  int secondblank = startLine.lastIndexOf(' ',lastblank-1);
488  QString dpi = startLine.mid(secondblank+1,lastblank-secondblank-1);
489 
490  progress.show();
491  progress.increaseNumSteps( i18n("Currently generating %1 at %2 dpi", fontName, dpi) );
492  }
493  MetafontOutput = MetafontOutput.remove(0,numleft+1);
494  }
