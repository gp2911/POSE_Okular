63 void FilePrinterPreviewPrivate::getPart()
64 {
65  if (previewPart) {
66  kDebug(500) << "already got a part";
67  return;
68  }
69  kDebug(500) << "querying trader for application/ps service";
70 
71  KPluginFactory *factory(0);
72  KService::List offers =
73  KMimeTypeTrader::self()->query("application/postscript", "KParts/ReadOnlyPart");
74 
75  KService::List::ConstIterator it = offers.constBegin();
76  while (!factory && it != offers.constEnd()) {
77  KPluginLoader loader(**it);
78  factory = loader.factory();
79  if (!factory) {
80  kDebug(500) << "Loading failed:" << loader.errorString();
81  }
82  ++it;
83  }
84  if (factory) {
85  kDebug(500) << "Trying to create a part";
86  previewPart = factory->create<KParts::ReadOnlyPart>(q, (QVariantList() << "Print/Preview"));
87  if (!previewPart) {
88  kDebug(500) << "Part creation failed";
89  }
90  }
91 }

93 bool FilePrinterPreviewPrivate::doPreview()
94 {
95  if (!QFile::exists(filename)) {
96  kWarning() << "Nothing was produced to be previewed";
97  return false;
98  }
99 
100  getPart();
101  if (!previewPart) {
102  //TODO: error dialog
103  kWarning() << "Could not find a PS viewer for the preview dialog";
104  fail();
105  return false;
106  } else {
107  q->setMainWidget(previewPart->widget());
108  return previewPart->openUrl(filename);
109  }
110 }
