82 void KTreeViewSearchLine::Private::rowsInserted( const QModelIndex & parentIndex, int start, int end ) const
83 {
84  QAbstractItemModel* model = qobject_cast<QAbstractItemModel*>( parent->sender() );
85  if ( !model )
86  return;
87 
88  QTreeView* widget = 0L;
89  foreach ( QTreeView* tree, treeViews )
90  if ( tree->model() == model ) {
91  widget = tree;
92  break;
93  }
94 
95  if ( !widget )
96  return;
97 
98  for ( int i = start; i <= end; ++i ) {
99  widget->setRowHidden( i, parentIndex, !parent->itemMatches( parentIndex, i, parent->text() ) );
100  }

109 void KTreeViewSearchLine::Private::slotColumnActivated( QAction *action )
110 {
111  if ( !action )
112  return;
113 
114  bool ok;
115  int column = action->data().toInt( &ok );
116 
117  if ( !ok )
118  return;
119 
120  if ( action->isChecked() ) {
121  if ( !searchColumns.isEmpty() ) {
122  if ( !searchColumns.contains( column ) )
123  searchColumns.append( column );
124 
125  if ( searchColumns.count() == treeViews.first()->header()->count() - treeViews.first()->header()->hiddenSectionCount() )
126  searchColumns.clear();
127 
128  } else {
129  searchColumns.append( column );
130  }
131  } else {
132  if ( searchColumns.isEmpty() ) {
133  QHeaderView* const header = treeViews.first()->header();
134 
135  for ( int i = 0; i < header->count(); i++ ) {
136  if ( i != column && !header->isSectionHidden( i ) )
137  searchColumns.append( i );
138  }
139 
140  } else if ( searchColumns.contains( column ) ) {
141  searchColumns.removeAll( column );
142  }
143  }
144 
145  parent->updateSearch();
146 }

210 bool KTreeViewSearchLine::Private::checkItemParentsVisible( QTreeView *treeView, const QModelIndex &index )
211 {
212  bool childMatch = false;
213  const int rowcount = treeView->model()->rowCount( index );
214  for ( int i = 0; i < rowcount; ++i )
215  childMatch |= checkItemParentsVisible( treeView, treeView->model()->index( i, 0, index ) );
216 
217  // Should this item be shown? It should if any children should be, or if it matches.
218  const QModelIndex parentindex = index.parent();
219  if ( childMatch || parent->itemMatches( parentindex, index.row(), search ) ) {
220  treeView->setRowHidden( index.row(), parentindex, false );
221  return true;
222  }
223 
224  treeView->setRowHidden( index.row(), parentindex, true );
225 
226  return false;

341 void KTreeViewSearchLine::updateSearch( QTreeView *treeView )
342 {
343  if ( !treeView || !treeView->model()->rowCount() )
344  return;
345 
346 
347  // If there's a selected item that is visible, make sure that it's visible
348  // when the search changes too (assuming that it still matches).
349 
350  QModelIndex currentIndex = treeView->currentIndex();
351 
352  bool wasUpdateEnabled = treeView->updatesEnabled();
353  treeView->setUpdatesEnabled( false );
354  if ( d->keepParentsVisible )
355  for ( int i = 0; i < treeView->model()->rowCount(); ++i )
356  d->checkItemParentsVisible( treeView, treeView->rootIndex() );
357  else
358  d->checkItemParentsNotVisible( treeView );
359  treeView->setUpdatesEnabled( wasUpdateEnabled );
360 
361  if ( currentIndex.isValid() )
362  treeView->scrollTo( currentIndex );
363 }

422 bool KTreeViewSearchLine::itemMatches( const QModelIndex &index, int row, const QString &pattern ) const
423 {
424  if ( pattern.isEmpty() )
425  return true;
426 
427  if ( !index.isValid() )
428  return false;
429 
430  // Contruct a regular expression object with the right options.
431  QRegExp expression = QRegExp( pattern,
432  d->caseSensitive ? Qt::CaseSensitive : Qt::CaseInsensitive,
433  d->regularExpression ? QRegExp::RegExp : QRegExp::FixedString );
434 
435  // If the search column list is populated, search just the columns
436  // specifified. If it is empty default to searching all of the columns.
437 
438  const int columncount = index.model()->columnCount( index );
439  if ( !d->searchColumns.isEmpty() ) {
440  QList<int>::ConstIterator it = d->searchColumns.constBegin();
441  for ( ; it != d->searchColumns.constEnd(); ++it ) {
442  if ( *it < columncount &&
443  expression.indexIn( index.child( row, *it ).data( Qt::DisplayRole ).toString() ) >= 0 )
444  return true;
445  }
446  } else {
447  for ( int i = 0; i < columncount; ++i) {
448  if ( expression.indexIn( index.child( row, i ).data( Qt::DisplayRole ).toString() ) >= 0 )
449  return true;
450  }
451  }
452 
453  return false;
454 }

534 bool KTreeViewSearchLine::canChooseColumnsCheck()
535 {
536  // This is true if either of the following is true:
537 
538  // there are no listviews connected
539  if ( d->treeViews.isEmpty() )
540  return false;
541 
542  const QTreeView *first = d->treeViews.first();
543 
544  const int numcols = first->model()->columnCount();
545  // the listviews have only one column,
546  if ( numcols < 2 )
547  return false;
548 
549  QStringList headers;
550  for ( int i = 0; i < numcols; ++i )
551  headers.append( first->header()->model()->headerData( i, Qt::Horizontal, Qt::DisplayRole ).toString() );
552 
553  QList<QTreeView *>::ConstIterator it = d->treeViews.constBegin();
554  for ( ++it /* skip the first one */; it != d->treeViews.constEnd(); ++it ) {
555  // the listviews have different numbers of columns,
556  if ( (*it)->model()->columnCount() != numcols )
557  return false;
558 
559  // the listviews differ in column labels.
560  QStringList::ConstIterator jt;
561  int i;
562  for ( i = 0, jt = headers.constBegin(); i < numcols; ++i, ++jt ) {
563  Q_ASSERT( jt != headers.constEnd() );
564 
565  if ( (*it)->header()->model()->headerData( i, Qt::Horizontal, Qt::DisplayRole ).toString() != *jt )
566  return false;
567  }
568  }
569 
570  return true;
571 }
