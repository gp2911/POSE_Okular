135 bool ProtocolMSITS::parseLoadAndLookup ( const KUrl& url, QString& abspath )
136 {
137  kDebug() << "ProtocolMSITS::parseLoadAndLookup (const KUrl&) " << url.path();
138 
139  int pos = url.path().indexOf ("::");
140 
141  if ( pos == -1 )
142  {
143  error( KIO::ERR_MALFORMED_URL, url.prettyUrl() );
144  return false;
145  }
146 
147  QString filename = url.path().left (pos);
148  abspath = url.path().mid (pos + 2); // skip ::
149 
150  // Some buggy apps add ms-its:/ to the path as well
151  if ( abspath.startsWith( "ms-its:" ) )
152  abspath = abspath.mid( 7 );
153 
154  kDebug() << "ProtocolMSITS::parseLoadAndLookup: filename " << filename << ", path " << abspath;
155 
156  if ( filename.isEmpty() )
157  {
158  error( KIO::ERR_MALFORMED_URL, url.prettyUrl() );
159  return false;
160  }
161 
162  // If the file has been already loaded, nothing to do.
163  if ( m_chmFile && filename == m_openedFile )
164  return true;
165 
166  kDebug() << "Opening a new CHM file " << QFile::encodeName( QDir::toNativeSeparators( filename ) );
167 
168  // First try to open a temporary file
169  chmFile * tmpchm;
170 
171  if( (tmpchm = chm_open ( QFile::encodeName( QDir::toNativeSeparators( filename) ) ) ) == 0 )
172  {
173  error( KIO::ERR_COULD_NOT_READ, url.prettyUrl() );
174  return false;
175  }
176 
177  // Replace an existing file by a new one
178  if ( m_chmFile )
179  chm_close (m_chmFile);
180 
181  m_chmFile = tmpchm;
182  m_openedFile = filename;
183 
184  kDebug() << "A CHM file " << filename << " has beed opened successfully";
185  return true;
186 }

278

284  chmlib_enumerator,

288  return;
289  }

292  int striplength = filepath.length();
293

308  finished();
