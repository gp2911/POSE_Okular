147 void ghostscript_interface::gs_generate_graphics_file(const PageNumber& page, const QString& filename, long magnification) {
148 #ifdef DEBUG_PSGS
149  kDebug(kvs::dvi) << "ghostscript_interface::gs_generate_graphics_file( " << page << ", " << filename << " )";
150 #endif
151 
152  if (knownDevices.isEmpty()) {
153  kError(kvs::dvi) << "No known devices found" << endl;
154  return;
155  }
156 
157  pageInfo *info = pageList.value(page);
158 
159  // Generate a PNG-file
160  // Step 1: Write the PostScriptString to a File
161  KTemporaryFile PSfile;
162  PSfile.setAutoRemove(false);
163  PSfile.setSuffix(".ps");
164  PSfile.open();
165  const QString PSfileName = PSfile.fileName();
166 
167  QTextStream os(&PSfile);
168  os << "%!PS-Adobe-2.0\n"
169  << "%%Creator: kdvi\n"
170  << "%%Title: KDVI temporary PostScript\n"
171  << "%%Pages: 1\n"
172  << "%%PageOrder: Ascend\n"
173  // HSize and VSize in 1/72 inch
174  << "%%BoundingBox: 0 0 "
175  << (qint32)(72*(pixel_page_w/resolution)) << ' '
176  << (qint32)(72*(pixel_page_h/resolution)) << '\n'
177  << "%%EndComments\n"
178  << "%!\n"
179  << psheader
180  << "TeXDict begin "
181  // HSize in (1/(65781.76*72))inch
182  << (qint32)(72*65781*(pixel_page_w/resolution)) << ' '
183  // VSize in (1/(65781.76*72))inch
184  << (qint32)(72*65781*(pixel_page_h/resolution)) << ' '
185  // Magnification
186  << (qint32)(magnification)
187  // dpi and vdpi
188  << " 300 300"
189  // Name
190  << " (test.dvi)"
191  << " @start end\n"
192  << "TeXDict begin\n"
193  // Start page
194  << "1 0 bop 0 0 a \n";
195 
196  if (!PostScriptHeaderString->toLatin1().isNull())
197  os << PostScriptHeaderString->toLatin1();
198 
199  if (info->background != Qt::white) {
200  QString colorCommand = QString("gsave %1 %2 %3 setrgbcolor clippath fill grestore\n").
201  arg(info->background.red()/255.0).
202  arg(info->background.green()/255.0).
203  arg(info->background.blue()/255.0);
204  os << colorCommand.toLatin1();
205  }
206 
207  if (!info->PostScriptString->isNull())
208  os << *(info->PostScriptString);
209 
210  os << "end\n"
211  << "showpage \n";
212 
213  PSfile.close();
214 
215  // Step 2: Call GS with the File
216  QFile::remove(filename.toAscii());
217  KProcess proc;
218  proc.setOutputChannelMode(KProcess::SeparateChannels);
219  QStringList argus;
220  argus << "gs";
221  argus << "-dSAFER" << "-dPARANOIDSAFER" << "-dDELAYSAFER" << "-dNOPAUSE" << "-dBATCH";
222  argus << QString("-sDEVICE=%1").arg(*gsDevice);
223  argus << QString("-sOutputFile=%1").arg(filename);
224  argus << QString("-sExtraIncludePath=%1").arg(includePath);
225  argus << QString("-g%1x%2").arg(pixel_page_w).arg(pixel_page_h); // page size in pixels
226  argus << QString("-r%1").arg(resolution); // resolution in dpi
227  argus << "-dTextAlphaBits=4 -dGraphicsAlphaBits=2"; // Antialiasing
228  argus << "-c" << "<< /PermitFileReading [ ExtraIncludePath ] /PermitFileWriting [] /PermitFileControl [] >> setuserparams .locksafe";
229  argus << "-f" << PSfileName;
230 
231 #ifdef DEBUG_PSGS
232  kDebug(kvs::dvi) << argus.join(" ");
233 #endif
234 
235  proc << argus;
236  int res = proc.execute();
237 
238  if ( res ) {
239  // Starting ghostscript did not work.
240  // TODO: Issue error message, switch PS support off.
241  kError(kvs::dvi) << "ghostview could not be started" << endl;
242  }
243 
244  PSfile.remove();
245 
246  // Check if gs has indeed produced a file.
247  if (QFile::exists(filename) == false) {
248  kError(kvs::dvi) << "GS did not produce output." << endl;
249 
250  // No. Check is the reason is that the device is not compiled into
251  // ghostscript. If so, try again with another device.
252  QString GSoutput;
253  proc.setReadChannel(QProcess::StandardOutput);
254  while(proc.canReadLine()) {
255  GSoutput = QString::fromLocal8Bit(proc.readLine());
256  if (GSoutput.contains("Unknown device")) {
257  kDebug(kvs::dvi) << QString("The version of ghostview installed on this computer does not support "
258  "the '%1' ghostview device driver.").arg(*gsDevice) << endl;
259  knownDevices.erase(gsDevice);
260  gsDevice = knownDevices.begin();
261  if (knownDevices.isEmpty())
262  // TODO: show a requestor of some sort.
263  emit error(i18n("The version of Ghostview that is installed on this computer does not contain "
264  "any of the Ghostview device drivers that are known to Okular. PostScript "
265  "support has therefore been turned off in Okular."), -1);
266 #if 0
267  i18n("<qt><p>The Ghostview program, which Okular uses internally to display the "
268  "PostScript graphics that is included in this DVI file, is generally able to "
269  "write its output in a variety of formats. The sub-programs that Ghostview uses "
270  "for these tasks are called 'device drivers'; there is one device driver for "
271  "each format that Ghostview is able to write. Different versions of Ghostview "
272  "often have different sets of device drivers available. It seems that the "
273  "version of Ghostview that is installed on this computer does not contain "
274  "<strong>any</strong> of the device drivers that are known to Okular.</p>"
275  "<p>It seems unlikely that a regular installation of Ghostview would not contain "

281  "drivers. Note that Okular needs to be restarted to re-enable PostScript support."
282  "</p></qt>"));
283 #endif
284  else {
285  kDebug(kvs::dvi) << QString("Okular will now try to use the '%1' device driver.").arg(*gsDevice);
286  gs_generate_graphics_file(page, filename, magnification);
287  }
288  return;
289  }
290  }

295 void ghostscript_interface::graphics(const PageNumber& page, double dpi, long magnification, QPainter* paint) {
296 #ifdef DEBUG_PSGS
297  kDebug(kvs::dvi) << "ghostscript_interface::graphics( " << page << ", " << dpi << ", ... ) called.";
298 #endif
299 
300  if (paint == 0) {
301  kError(kvs::dvi) << "ghostscript_interface::graphics(PageNumber page, double dpi, long magnification, QPainter *paint) called with paint == 0" << endl;
302  return;
303  }
304 
305  resolution = dpi;
306 
307  pixel_page_w = paint->viewport().width();
308  pixel_page_h = paint->viewport().height();
309 
310  pageInfo *info = pageList.value(page);
311 
312  // No PostScript? Then return immediately.
313  if ((info == 0) || (info->PostScriptString->isEmpty())) {
314 #ifdef DEBUG_PSGS
315  kDebug(kvs::dvi) << "No PostScript found. Not drawing anything.";
316 #endif
317  return;
318  }
319 
320  QTemporaryFile gfxFile;
321  gfxFile.open();
322  const QString gfxFileName = gfxFile.fileName();
323  // We are want the filename, not the file.
324  gfxFile.close();
325 
326  gs_generate_graphics_file(page, gfxFileName, magnification);
327 
328  QImage MemoryCopy(gfxFileName);
329  paint->drawImage(0, 0, MemoryCopy);
330  return;
331 }

334 QString ghostscript_interface::locateEPSfile(const QString &filename, const KUrl &base)
335 {
336  // If the base URL indicates that the DVI file is local, try to find
337  // the graphics file in the directory where the DVI file resides
338  if (base.isLocalFile()) {
339  QString path = base.path(); // -> "/bar/foo.dvi"
340  QFileInfo fi1(path);
341  QFileInfo fi2(fi1.dir(),filename);
342  if (fi2.exists())
343  return fi2.absoluteFilePath();
344  }
345 
346  // Otherwise, use kpsewhich to find the eps file.
347  KProcess proc;
348  proc << "kpsewhich" << filename;
349  proc.execute();
350  return QString::fromLocal8Bit(proc.readLine().trimmed());
351 }
