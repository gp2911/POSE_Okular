333 void BookmarkList::rebuildTree( bool filter )
334 {
335  // disconnect and reconnect later, otherwise we'll get many itemChanged()
336  // signals for all the current items
337  disconnect( m_tree, SIGNAL(itemChanged(QTreeWidgetItem*,int)), this, SLOT(slotChanged(QTreeWidgetItem*)) );
338 
339  m_currentDocumentItem = 0;
340  m_tree->clear();
341 
342  KUrl::List urls = m_document->bookmarkManager()->files();
343  if ( filter )
344  {
345  if ( m_document->isOpened() )
346  {
347  foreach ( const KUrl& url, urls )
348  {
349  if ( url == m_document->currentDocument() )
350  {
351  m_tree->addTopLevelItems( createItems( url, m_document->bookmarkManager()->bookmarks( url ) ) );
352  m_currentDocumentItem = m_tree->invisibleRootItem();
353  break;
354  }
355  }
356  }
357  }
358  else
359  {
360  QTreeWidgetItem * currenturlitem = 0;
361  foreach ( const KUrl& url, urls )
362  {
363  QList<QTreeWidgetItem*> subitems = createItems( url, m_document->bookmarkManager()->bookmarks( url ) );
364  if ( !subitems.isEmpty() )
365  {
366  FileItem * item = new FileItem( url, m_tree, m_document );
367  item->addChildren( subitems );
368  if ( !currenturlitem && url == m_document->currentDocument() )
369  {
370  currenturlitem = item;
371  }
372  }
373  }
374  if ( currenturlitem )
375  {
376  currenturlitem->setExpanded( true );
377  currenturlitem->setIcon( 0, KIcon( "bookmarks" ) );
378  m_tree->scrollToItem( currenturlitem, QAbstractItemView::PositionAtTop );
379  m_currentDocumentItem = currenturlitem;
380  }
381  }
382 
383  m_tree->sortItems( 0, Qt::AscendingOrder );
384 
385  connect( m_tree, SIGNAL(itemChanged(QTreeWidgetItem*,int)), this, SLOT(slotChanged(QTreeWidgetItem*)) );
386 }

401 void BookmarkList::selectiveUrlUpdate( const KUrl& url, QTreeWidgetItem*& item )
402 {
403  disconnect( m_tree, SIGNAL(itemChanged(QTreeWidgetItem*,int)), this, SLOT(slotChanged(QTreeWidgetItem*)) );
404 
405  const KBookmark::List urlbookmarks = m_document->bookmarkManager()->bookmarks( url );
406  if ( urlbookmarks.isEmpty() )
407  {
408  if ( item != m_tree->invisibleRootItem() )
409  {
410  m_tree->invisibleRootItem()->removeChild( item );
411  item = 0;
412  }
413  else if ( item )
414  {
415  for ( int i = item->childCount(); i >= 0; --i )
416  {
417  item->removeChild( item->child( i ) );
418  }
419  }
420  }
421  else
422  {
423  bool fileitem_created = false;
424 
425  if ( item )
426  {
427  for ( int i = item->childCount() - 1; i >= 0; --i )
428  {
429  item->removeChild( item->child( i ) );
430  }
431  }
432  else
433  {
434  item = new FileItem( url, m_tree, m_document );
435  fileitem_created = true;
436  }
437  if ( m_document->isOpened() && url == m_document->currentDocument() )
438  {
439  item->setIcon( 0, KIcon( "bookmarks" ) );
440  item->setExpanded( true );
441  }
442  item->addChildren( createItems( url, urlbookmarks ) );
443 
444  if ( fileitem_created )
445  {
446  // we need to sort also the parent of the new file item,
447  // so it can be properly shown in the correct place
448  m_tree->invisibleRootItem()->sortChildren( 0, Qt::AscendingOrder );
449  }
450  item->sortChildren( 0, Qt::AscendingOrder );
451  }
452 
453  connect( m_tree, SIGNAL(itemChanged(QTreeWidgetItem*,int)), this, SLOT(slotChanged(QTreeWidgetItem*)) );
454 }
