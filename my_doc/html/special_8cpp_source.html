209 {

211 
212  QString command = cp.section(' ', 0, 0);
213 
214  if (command == "pop") {
215  // Take color off the stack
216  if (colorStack.isEmpty())
217  printErrorMsgForSpecials( i18n("Error in DVIfile '%1', page %2. Color pop command issued when the color stack is empty." ,
218  dviFile->filename, current_page));
219  else
220  colorStack.pop();
221  return;
222  }
223 
224  if (command == "push") {
225  // Get color specification
226  const QColor col = parseColorSpecification(cp.section(' ', 1));
227  // Set color
228  if (col.isValid())
229  colorStack.push(col);
230  else
231  colorStack.push(Qt::black);
232  return;
233  }
234 
235  // Get color specification and set the color for the rest of this
236  // page
237  QColor col = parseColorSpecification(cp);
238  // Set color
239  if (col.isValid())
240  globalColor = col;
241  else
242  globalColor = Qt::black;
243  return;

505 void dviRenderer::applicationDoSpecial(char *cp)
506 {
507  QString special_command(cp);
508 
509  // First come specials which is only interpreted during rendering,
510  // and NOT during the prescan phase
511 
512  // font color specials
513  if (strncasecmp(cp, "color", 5) == 0) {
514  color_special(special_command.mid(5));
515  return;
516  }
517 
518  // HTML reference
519  if (strncasecmp(cp, "html:<A href=", 13) == 0) {
520  html_href_special(special_command.mid(14));
521  return;
522  }
523 
524  // HTML anchor end
525  if (strncasecmp(cp, "html:</A>", 9) == 0) {
526  html_anchor_end();
527  return;
528  }
529 
530  // TPIC specials
531  if (strncasecmp(cp, "pn", 2) == 0) {
532  TPIC_setPen_special(special_command.mid(2));
533  return;
534  }
535  if (strncasecmp(cp, "pa ", 3) == 0) {
536  TPIC_addPath_special(special_command.mid(3));
537  return;
538  }
539  if (strncasecmp(cp, "fp", 2) == 0) {
540  TPIC_flushPath_special();
541  return;
542  }
543 
544  // Encapsulated Postscript File
545  if (strncasecmp(cp, "PSfile=", 7) == 0) {
546  epsf_special(special_command.mid(7));
547  return;
548  }
549 
550  // source special
551  if (strncasecmp(cp, "src:", 4) == 0) {
552  source_special(special_command.mid(4));
553  return;
554  }
555 
556  // Unfortunately, in some TeX distribution the hyperref package uses
557  // the dvips driver by default, rather than the hypertex driver. As
558  // a result, the DVI files produced are full of PostScript that
559  // specifies links and anchors, and KDVI would call the ghostscript
560  // interpreter for every page which makes it really slow. This is a
561  // major nuisance, so that we try to filter and interpret the
562  // hypertex generated PostScript here.
563  if (special_command.startsWith("ps:SDict begin")) {
564 
565  // Hyperref: start of hyperref rectangle. At this stage it is not
566  // yet clear if the rectangle will conain a hyperlink, an anchor,
567  // or another type of object. We suspect that this rectangle will
568  // define a hyperlink, allocate a QString and set HTML_href to
569  // point to this string. The string contains the name of the
570  // destination which ---due to the nature of the PostScript
571  // language--- will be defined only after characters are drawn and
572  // the hyperref rectangle has been closed. We use "glopglyph" as a
573  // temporary name. Since the pointer HTML_href is not NULL, the
574  // chracter drawing routines will now underline all characters in
575  // blue to point out that they correspond to a hyperlink. Also, as
576  // soon as characters are drawn, the drawing routines will
577  // allocate a Hyperlink and add it to the top of the vector
578  // currentlyDrawnPage->hyperLinkList.
579  if (special_command == "ps:SDict begin H.S end") {
580  // At this stage, the vector 'hyperLinkList' should not contain
581  // links with unspecified destinations (i.e. destination set to
582  // 'glopglyph'). As a protection against bad DVI files, we make
583  // sure to remove all link rectangles which point to
584  // 'glopglyph'.
585  while (!currentlyDrawnPage->hyperLinkList.isEmpty())
586  if (currentlyDrawnPage->hyperLinkList.last().linkText == "glopglyph")
587  currentlyDrawnPage->hyperLinkList.pop_back();
588  else
589  break;
590 
591  HTML_href = new QString("glopglyph");
592  return;
593  }
594 
595  // Hyperref: end of hyperref rectangle of unknown type or hyperref
596  // link rectangle. In these cases we set HTML_href to NULL, which
597  // causes the character drawing routines to stop drawing
598  // characters underlined in blue. Note that the name of the
599  // destination is still set to "glopglyph". In a well-formed DVI
600  // file, this special command is immediately followed by another
601  // special, where the destination is specified. This special is
602  // treated below.
603  if ((special_command == "ps:SDict begin H.R end") || special_command.endsWith("H.L end")) {
604  if (HTML_href != NULL) {
605  delete HTML_href;
606  HTML_href = NULL;
607  }
608  return; // end of hyperref rectangle
609  }
610 
611  // Hyperref: end of anchor rectangle. If this special is
612  // encountered, the rectangle, which was started with "ps:SDict
613  // begin H.S end" does not contain a link, but an anchor for a
614  // link. Anchors, however, have already been dealt with in the
615  // prescan phase and will not be considered here. Thus, we set
616  // HTML_href to NULL so that character drawing routines will no
617  // longer underline hyperlinks in blue, and remove the link from
618  // the hyperLinkList. NOTE: in a well-formed DVI file, the "H.A"
619  // special comes directly after the "H.S" special. A
620  // hyperlink-anchor rectangle therefore never contains characters,
621  // so no character will by accidentally underlined in blue.
622  if (special_command.endsWith("H.A end")) {
623  if (HTML_href != NULL) {
624  delete HTML_href;
625  HTML_href = NULL;
626  }
627  while (!currentlyDrawnPage->hyperLinkList.isEmpty())
628  if (currentlyDrawnPage->hyperLinkList.last().linkText == "glopglyph")
629  currentlyDrawnPage->hyperLinkList.pop_back();
630  else
631  break;
632  return; // end of hyperref anchor
633  }
634 
635  // Hyperref: specification of a hyperref link rectangle's
636  // destination. As mentioned above, the destination of a hyperlink
637  // is specified only AFTER the rectangle has been specified. We
638  // will therefore go through the list of rectangles stored in
639  // currentlyDrawnPage->hyperLinkList, find those whose destination
640  // is open and fill in the value found here. NOTE: the character
641  // drawing routines sometimes split a single hyperlink rectangle
642  // into several rectangles (e.g. if the font changes, or when a
643  // line break is encountered)
644  if (special_command.startsWith("ps:SDict begin [") && special_command.endsWith(" pdfmark end")) {
645  if (!currentlyDrawnPage->hyperLinkList.isEmpty()) {
646  QString targetName = special_command.section('(', 1, 1).section(')', 0, 0);
647  QVector<Hyperlink>::iterator it;
648  for( it = currentlyDrawnPage->hyperLinkList.begin(); it != currentlyDrawnPage->hyperLinkList.end(); ++it )
649  if (it->linkText == "glopglyph")
650  it->linkText = targetName;
651  }
652  return; // hyperref definition of link/anchor/bookmark/etc
653  }
654  }
655 
656  // Detect text rotation specials that are included by the graphicx
657  // package. If one of these specials is found, the state of the
658  // painter is saved, and the coordinate system is rotated
659  // accordingly
660  if (special_command.startsWith("ps: gsave currentpoint currentpoint translate ") &&
661  special_command.endsWith(" neg rotate neg exch neg exch translate") ) {
662  bool ok;
663  double angle = special_command.section(' ', 5, 5).toDouble(&ok);
664  if (ok == true) {
665  int x = ((int) ((currinf.data.dvi_h) / (shrinkfactor * 65536)));
666  int y = currinf.data.pxl_v;
667 
668  foreGroundPainter->save();
669  // Rotate about the current point
670  foreGroundPainter->translate(x,y);
671  foreGroundPainter->rotate(-angle);
672  foreGroundPainter->translate(-x,-y);
673  } else
674  printErrorMsgForSpecials( i18n("Error in DVIfile '%1', page %2. Could not interpret angle in text rotation special." ,
675  dviFile->filename, current_page));
676  }
677 
678  // The graphicx package marks the end of rotated text with this
679  // special. The state of the painter is restored.
680  if (special_command == "ps: currentpoint grestore moveto") {
681  foreGroundPainter->restore();
682  }
683 
684  // The following special commands are not used here; they are of
685  // interest only during the prescan phase. We recognize them here
686  // anyway, to make sure that KDVI doesn't complain about
687  // unrecognized special commands.
688  if ((cp[0] == '!') ||
689  (cp[0] == '"') ||
690  (strncasecmp(cp, "html:<A name=", 13) == 0) ||
691  (strncasecmp(cp, "ps:", 3) == 0) ||
692  (strncasecmp(cp, "papersize", 9) == 0) ||
693  (strncasecmp(cp, "header", 6) == 0) ||
694  (strncasecmp(cp, "background", 10) == 0) )
695  return;
696 
697  printErrorMsgForSpecials(i18n("The special command '%1' is not implemented.", special_command));
698  return;
699 }
