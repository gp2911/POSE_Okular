162 QPixmap loadStamp( const QString& _name, const QSize& size, int iconSize )
163 {
164  const QString name = _name.toLower();
165  QSvgRenderer * r = 0;
166  if ( ( r = s_data->svgStamps() ) && r->elementExists( name ) )
167  {
168  const QRectF stampElemRect = r->boundsOnElement( name );
169  const QRectF stampRect( size.isValid() ? QRectF( QPointF( 0, 0 ), size ) : stampElemRect );
170  QPixmap pixmap( stampRect.size().toSize() );
171  pixmap.fill( Qt::transparent );
172  QPainter p( &pixmap );
173  r->render( &p, name );
174  p.end();
175  return pixmap;
176  }
177  QPixmap pixmap;
178  const KIconLoader * il = iconLoader();
179  QString path;
180  const int minSize = iconSize > 0 ? iconSize : qMin( size.width(), size.height() );
181  pixmap = il->loadIcon( name, KIconLoader::User, minSize, KIconLoader::DefaultState, QStringList(), &path, true );
182  if ( path.isEmpty() )
183  pixmap = il->loadIcon( name, KIconLoader::NoGroup, minSize );
184  return pixmap;
185 }

237 void colorizeImage( QImage & grayImage, const QColor & color, unsigned int destAlpha )
238 {
239  // Make sure that the image is Format_ARGB32_Premultiplied
240  if ( grayImage.format() != QImage::Format_ARGB32_Premultiplied )
241  grayImage = grayImage.convertToFormat( QImage::Format_ARGB32_Premultiplied );
242 
243  // iterate over all pixels changing the alpha component value
244  unsigned int * data = (unsigned int *)grayImage.bits();
245  unsigned int pixels = grayImage.width() * grayImage.height();
246  int red = color.red(),
247  green = color.green(),
248  blue = color.blue();
249 
250  int source, sourceSat, sourceAlpha;
251  for( register unsigned int i = 0; i < pixels; ++i )
252  { // optimize this loop keeping byte order into account
253  source = data[i];
254  sourceSat = qRed( source );
255  int newR = qt_div_255( sourceSat * red ),
256  newG = qt_div_255( sourceSat * green ),
257  newB = qt_div_255( sourceSat * blue );
258  if ( (sourceAlpha = qAlpha( source )) == 255 )
259  {
260  // use destAlpha
261  data[i] = qRgba( newR, newG, newB, destAlpha );
262  }
263  else
264  {
265  // use destAlpha * sourceAlpha product
266  if ( destAlpha < 255 )
267  sourceAlpha = qt_div_255( destAlpha * sourceAlpha );
268  data[i] = qRgba( newR, newG, newB, sourceAlpha );
269  }
270  }
271 }
