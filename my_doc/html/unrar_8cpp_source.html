42 static UnrarFlavour* detectUnrar( const QString &unrarPath )
43 {
44  UnrarFlavour* kind = 0;
45  QProcess proc;
46  proc.start( unrarPath, QStringList() << "--version" );
47  bool ok = proc.waitForFinished( -1 );
48  Q_UNUSED( ok )
49  const QStringList lines = QString::fromLocal8Bit( proc.readAllStandardOutput() ).split( '\n', QString::SkipEmptyParts );
50  if ( !lines.isEmpty() )
51  {
52  if ( lines.first().startsWith( "UNRAR " ) )
53  kind = new NonFreeUnrarFlavour();
54  else if ( lines.first().startsWith( "RAR " ) )
55  kind = new NonFreeUnrarFlavour();
56  else if ( lines.first().startsWith( "unrar " ) )
57  kind = new FreeUnrarFlavour();
58  }
59  return kind;
60 }

214 int Unrar::startSyncProcess( const QStringList &args )
215 {
216  int ret = 0;
217 
218 #if defined(Q_OS_WIN)
219  mProcess = new QProcess( this );
220 #else
221  mProcess = new KPtyProcess( this );
222  mProcess->setOutputChannelMode( KProcess::SeparateChannels );
223 #endif
224 
225  connect( mProcess, SIGNAL(readyReadStandardOutput()), SLOT(readFromStdout()) );
226  connect( mProcess, SIGNAL(readyReadStandardError()), SLOT(readFromStderr()) );
227  connect( mProcess, SIGNAL(finished(int,QProcess::ExitStatus)), SLOT(finished(int,QProcess::ExitStatus)) );
228 
229 #if defined(Q_OS_WIN)
230  mProcess->start( helper->unrarPath, args, QIODevice::ReadWrite | QIODevice::Unbuffered );
231  ret = mProcess->waitForFinished( -1 ) ? 0 : 1;
232 #else
233  mProcess->setProgram( helper->unrarPath, args );
234  mProcess->setNextOpenMode( QIODevice::ReadWrite | QIODevice::Unbuffered );
235  mProcess->start();
236  QEventLoop loop;
237  mLoop = &loop;
238  ret = loop.exec( QEventLoop::WaitForMoreEvents | QEventLoop::ExcludeUserInputEvents );
239  mLoop = 0;
240 #endif
241 
242  delete mProcess;
243  mProcess = 0;
244 
245  return ret;
246 }
