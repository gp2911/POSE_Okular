103 bool GSGenerator::print( QPrinter& printer )
104 {
105  bool result = false;
106 
107  // Create tempfile to write to
108  KTemporaryFile tf;
109  tf.setSuffix( ".ps" );
110 
111  // Get list of pages to print
112  QList<int> pageList = Okular::FilePrinter::pageList( printer,
113  spectre_document_get_n_pages( m_internalDocument ),
114  document()->currentPage() + 1,
115  document()->bookmarkedPageList() );
116 
117  // Default to Postscript export, but if printing to PDF use that instead
118  SpectreExporterFormat exportFormat = SPECTRE_EXPORTER_FORMAT_PS;
119  if ( printer.outputFileName().right(3) == "pdf" )
120  {
121  exportFormat = SPECTRE_EXPORTER_FORMAT_PDF;
122  tf.setSuffix(".pdf");
123  }
124 
125  if ( !tf.open() )
126  return false;
127 
128  SpectreExporter *exporter = spectre_exporter_new( m_internalDocument, exportFormat );
129  SpectreStatus exportStatus = spectre_exporter_begin( exporter, tf.fileName().toAscii() );
130 
131  int i = 0;
132  while ( i < pageList.count() && exportStatus == SPECTRE_STATUS_SUCCESS )
133  {
134  exportStatus = spectre_exporter_do_page( exporter, pageList.at( i ) - 1 );
135  i++;
136  }
137 
138  SpectreStatus endStatus = SPECTRE_STATUS_EXPORTER_ERROR;
139  if (exportStatus == SPECTRE_STATUS_SUCCESS)
140  endStatus = spectre_exporter_end( exporter );
141 
142  spectre_exporter_free( exporter );
143 
144  const QString fileName = tf.fileName();
145  tf.close();
146 
147  if ( exportStatus == SPECTRE_STATUS_SUCCESS && endStatus == SPECTRE_STATUS_SUCCESS )
148  {
149  tf.setAutoRemove( false );
150  int ret = Okular::FilePrinter::printFile( printer, fileName, document()->orientation(),
151  Okular::FilePrinter::SystemDeletesFiles,
152  Okular::FilePrinter::ApplicationSelectsPages,
153  document()->bookmarkedPageRange() );
154  if ( ret >= 0 ) result = true;
155  }
156 
157  return result;
158 }

160 bool GSGenerator::loadDocument( const QString & fileName, QVector< Okular::Page * > & pagesVector )
161 {
162  cache_AAtext = documentMetaData("TextAntialias", true).toBool();
163  cache_AAgfx = documentMetaData("GraphicsAntialias", true).toBool();
164 
165  m_internalDocument = spectre_document_new();
166  spectre_document_load(m_internalDocument, QFile::encodeName(fileName));
167  const SpectreStatus loadStatus = spectre_document_status(m_internalDocument);
168  if (loadStatus != SPECTRE_STATUS_SUCCESS)
169  {
170  kDebug(4711) << "ERR:" << spectre_status_to_string(loadStatus);
171  spectre_document_free(m_internalDocument);
172  m_internalDocument = 0;
173  return false;
174  }
175  pagesVector.resize( spectre_document_get_n_pages(m_internalDocument) );
176  kDebug(4711) << "Page count:" << pagesVector.count();
177  return loadPages(pagesVector);
178 }

191 void GSGenerator::slotImageGenerated(QImage *img, Okular::PixmapRequest *request)
192 {
193  // This can happen as GSInterpreterCMD is a singleton and on creation signals all the slots
194  // of all the generators attached to it
195  if (request != m_request) return;
196 
197  if ( !request->page()->isBoundingBoxKnown() )
198  updatePageBoundingBox( request->page()->number(), Okular::Utils::imageBoundingBox( img ) );
199 
200  m_request = 0;
201  QPixmap *pix = new QPixmap(QPixmap::fromImage(*img));
202  delete img;
203  request->page()->setPixmap( request->observer(), pix );
204  signalPixmapRequestDone( request );
205 }

207 bool GSGenerator::loadPages( QVector< Okular::Page * > & pagesVector )
208 {
209  for (uint i = 0; i < spectre_document_get_n_pages(m_internalDocument); i++)
210  {
211  SpectrePage *page;
212  int width = 0, height = 0;
213  SpectreOrientation pageOrientation = SPECTRE_ORIENTATION_PORTRAIT;
214  page = spectre_document_get_page (m_internalDocument, i);
215  if (spectre_document_status (m_internalDocument)) {
216  kDebug(4711) << "Error getting page" << i << spectre_status_to_string(spectre_document_status(m_internalDocument));
217  } else {
218  spectre_page_get_size(page, &width, &height);
219  pageOrientation = spectre_page_get_orientation(page);
220  }
221  spectre_page_free(page);
222  if (pageOrientation % 2 == 1) qSwap(width, height);
223  pagesVector[i] = new Okular::Page(i, width, height, orientation(pageOrientation));
224  }
225  return pagesVector.count() > 0;
226 }

228 void GSGenerator::generatePixmap( Okular::PixmapRequest * req )
229 {
230  kDebug(4711) << "receiving" << *req;
231 
232  SpectrePage *page = spectre_document_get_page(m_internalDocument, req->pageNumber());
233 
234  GSRendererThread *renderer = GSRendererThread::getCreateRenderer();
235 
236  GSRendererThreadRequest gsreq(this);
237  gsreq.spectrePage = page;
238  gsreq.platformFonts = GSSettings::platformFonts();
239  int graphicsAA = 1;
240  int textAA = 1;
241  if (cache_AAgfx) graphicsAA = 4;
242  if (cache_AAtext) textAA = 4;
243  gsreq.textAAbits = textAA;
244  gsreq.graphicsAAbits = graphicsAA;
245 
246  gsreq.orientation = req->page()->orientation();
247  if (req->page()->rotation() == Okular::Rotation90 ||
248  req->page()->rotation() == Okular::Rotation270)
249  {
250  gsreq.magnify = qMax( (double)req->height() / req->page()->width(),
251  (double)req->width() / req->page()->height() );
252  }
253  else
254  {
255  gsreq.magnify = qMax( (double)req->width() / req->page()->width(),
256  (double)req->height() / req->page()->height() );
257  }
258  gsreq.request = req;
259  m_request = req;
260  renderer->addRequest(gsreq);
261 }
