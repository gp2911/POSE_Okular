64 void EditFormsTest::init()
65 {
66  const QString testFile = KDESRCDIR "data/formSamples.pdf";
67  const KMimeType::Ptr mime = KMimeType::findByPath( testFile );
68  m_document->openDocument( testFile, KUrl(), mime );
69 
70  // Undo and Redo should be unavailable when docuemnt is first opened.
71  QVERIFY( !m_document->canUndo() );
72  QVERIFY( !m_document->canRedo() );
73 
74  const Okular::Page* page = m_document->page( 0 );
75  QLinkedList<Okular::FormField*> pageFields = page->formFields();
76 
77  // Clear lists
78  m_checkBoxForms.clear();
79  m_radioButtonForms.clear();
80 
81  // Collect forms of the various types
82  QLinkedList< Okular::FormField * >::const_iterator ffIt = pageFields.constBegin(), ffEnd = pageFields.constEnd();
83  for ( ; ffIt != ffEnd; ++ffIt )
84  {
85  Okular::FormField * ff = *ffIt;
86  ff->type();
87 
88  switch ( ff->type() )
89  {
90  case Okular::FormField::FormButton:
91  {
92  Okular::FormFieldButton * ffb = static_cast< Okular::FormFieldButton * >( ff );
93  switch ( ffb->buttonType() )
94  {
95  case Okular::FormFieldButton::Push:
96  break;
97  case Okular::FormFieldButton::CheckBox:
98  m_checkBoxForms.append(ffb);
99  break;
100  case Okular::FormFieldButton::Radio:
101  m_radioButtonForms.append(ffb);
102  break;
103  default: ;
104  }
105  break;
106  }
107  case Okular::FormField::FormText:
108  {
109  Okular::FormFieldText * fft = static_cast< Okular::FormFieldText * >( ff );
110  switch ( fft->textType() )
111  {
112  case Okular::FormFieldText::Multiline:
113  m_textAreaForm = fft;
114  break;
115  case Okular::FormFieldText::Normal:
116  m_textLineForm = fft;
117  break;
118  case Okular::FormFieldText::FileSelect:
119  m_fileEditForm = fft;
120  break;
121  }
122  break;
123  }
124  case Okular::FormField::FormChoice:
125  {
126  Okular::FormFieldChoice * ffc = static_cast< Okular::FormFieldChoice * >( ff );
127  switch ( ffc->choiceType() )
128  {
129  case Okular::FormFieldChoice::ListBox:
130  if ( ffc->multiSelect() )
131  {
132  m_listMultiEdit = ffc;
133  }
134  else
135  {
136  m_listSingleEdit = ffc;
137  }
138  break;
139  case Okular::FormFieldChoice::ComboBox:
140  m_comboEdit = ffc;
141  break;
142  }
143  break;
144  }
145  default: ;
146  }
147  }
148 }

386 void EditFormsTest::verifyTextForm( Okular::FormFieldText* form )
387 {
388  // Text in form is initially empty
389  QCOMPARE( form->text(), QString("") );
390 
391  // Insert the string "Hello" into the form
392  m_document->editFormText(0, form, QString("Hello"), 5, 0, 0);
393  QCOMPARE( form->text(), QString("Hello") );
394  QVERIFY( m_document->canUndo() );
395  QVERIFY( !m_document->canRedo() );
396 
397  // Undo the insertion and verify that form is empty again
398  m_document->undo();
399  QCOMPARE( form->text(), QString("") );
400  QVERIFY( !m_document->canUndo() );
401  QVERIFY( m_document->canRedo() );
402 
403  // Redo the insertion of "Hello"
404  m_document->redo();
405  QCOMPARE( form->text(), QString("Hello") );
406  QVERIFY( m_document->canUndo() );
407  QVERIFY( !m_document->canRedo() );
408 
409  // Type "_World" after "Hello"
410  m_document->editFormText(0, form, QString("Hello_"), 6, 5, 5);
411  m_document->editFormText(0, form, QString("Hello_W"), 7, 6, 6);
412  m_document->editFormText(0, form, QString("Hello_Wo"), 8, 7, 7);
413  m_document->editFormText(0, form, QString("Hello_Wor"), 9, 8, 8);
414  m_document->editFormText(0, form, QString("Hello_Worl"), 10, 9, 9);
415  m_document->editFormText(0, form, QString("Hello_World"), 11, 10, 10);
416 
417  // Verify that character insertion operations were merged together into a single undo command
418  m_document->undo();
419  QCOMPARE( form->text(), QString("Hello") );
420  QVERIFY( m_document->canUndo() );
421  QVERIFY( m_document->canRedo() );
422 
423  // Verify that one more undo gets us back to the original state (empty form)
424  m_document->undo();
425  QCOMPARE( form->text(), QString("") );
426  QVERIFY( !m_document->canUndo() );
427  QVERIFY( m_document->canRedo() );
428 }
