495 bool PDFGenerator::init(QVector<Okular::Page*> & pagesVector, const QString &walletKey)
496 {
497  // if the file didn't open correctly it might be encrypted, so ask for a pass
498  bool firstInput = true;
499  bool triedWallet = false;
500  KWallet::Wallet * wallet = 0;
501  bool keep = true;
502  while ( pdfdoc && pdfdoc->isLocked() )
503  {
504  QString password;
505 
506  // 1.A. try to retrieve the first password from the kde wallet system
507  if ( !triedWallet && !walletKey.isNull() )
508  {
509  QString walletName = KWallet::Wallet::NetworkWallet();
510  WId parentwid = 0;
511  if ( document() && document()->widget() )
512  parentwid = document()->widget()->effectiveWinId();
513  wallet = KWallet::Wallet::openWallet( walletName, parentwid );
514  if ( wallet )
515  {
516  // use the KPdf folder (and create if missing)
517  if ( !wallet->hasFolder( "KPdf" ) )
518  wallet->createFolder( "KPdf" );
519  wallet->setFolder( "KPdf" );
520 
521  // look for the pass in that folder
522  QString retrievedPass;
523  if ( !wallet->readPassword( walletKey, retrievedPass ) )
524  password = retrievedPass;
525  }
526  triedWallet = true;
527  }
528 
529  // 1.B. if not retrieved, ask the password using the kde password dialog
530  if ( password.isNull() )
531  {
532  QString prompt;
533  if ( firstInput )
534  prompt = i18n( "Please enter the password to read the document:" );
535  else
536  prompt = i18n( "Incorrect password. Try again:" );
537  firstInput = false;
538 
539  // if the user presses cancel, abort opening
540  KPasswordDialog dlg( document()->widget(), wallet ? KPasswordDialog::ShowKeepPassword : KPasswordDialog::KPasswordDialogFlags() );
541  dlg.setCaption( i18n( "Document Password" ) );
542  dlg.setPrompt( prompt );
543  if( !dlg.exec() )
544  break;
545  password = dlg.password();
546  if ( wallet )
547  keep = dlg.keepPassword();
548  }
549 
550  // 2. reopen the document using the password
551  pdfdoc->unlock( password.toLatin1(), password.toLatin1() );
552 
553  // 3. if the password is correct and the user chose to remember it, store it to the wallet
554  if ( !pdfdoc->isLocked() && wallet && /*safety check*/ wallet->isOpen() && keep )
555  {
556  wallet->writePassword( walletKey, password );
557  }
558  }
559  if ( !pdfdoc || pdfdoc->isLocked() )
560  {
561  delete pdfdoc;
562  pdfdoc = 0;
563  return false;
564  }
565 
566  // build Pages (currentPage was set -1 by deletePages)
567  int pageCount = pdfdoc->numPages();
568  if (pageCount < 0) {
569  delete pdfdoc;
570  pdfdoc = 0;
571  return false;
572  }
573  pagesVector.resize(pageCount);
574  rectsGenerated.fill(false, pageCount);
575 
576  annotationsHash.clear();
577 
578  loadPages(pagesVector, 0, false);
579 
580  // update the configuration
581  reparseConfig();
582 
583  // create annotation proxy
584  annotProxy = new PopplerAnnotationProxy( pdfdoc, userMutex() );
585 
586  // the file has been loaded correctly
587  return true;
588 }

616 void PDFGenerator::loadPages(QVector<Okular::Page*> &pagesVector, int rotation, bool clear)
617 {
618  // TODO XPDF 3.01 check
619  const int count = pagesVector.count();
620  double w = 0, h = 0;
621  for ( int i = 0; i < count ; i++ )
622  {
623  // get xpdf page
624  Poppler::Page * p = pdfdoc->page( i );
625  Okular::Page * page;
626  if (p)
627  {
628  const QSizeF pSize = p->pageSizeF();
629  w = pSize.width() / 72.0 * dpi().width();
630  h = pSize.height() / 72.0 * dpi().height();
631  Okular::Rotation orientation = Okular::Rotation0;
632  switch (p->orientation())
633  {
634  case Poppler::Page::Landscape: orientation = Okular::Rotation90; break;
635  case Poppler::Page::UpsideDown: orientation = Okular::Rotation180; break;
636  case Poppler::Page::Seascape: orientation = Okular::Rotation270; break;
637  case Poppler::Page::Portrait: orientation = Okular::Rotation0; break;
638  }
639  if (rotation % 2 == 1)
640  qSwap(w,h);
641  // init a Okular::page, add transition and annotation information
642  page = new Okular::Page( i, w, h, orientation );
643  addTransition( p, page );
644  if ( true ) //TODO real check
645  addAnnotations( p, page );
646  Poppler::Link * tmplink = p->action( Poppler::Page::Opening );
647  if ( tmplink )
648  {
649  page->setPageAction( Okular::Page::Opening, createLinkFromPopplerLink( tmplink ) );
650  }
651  tmplink = p->action( Poppler::Page::Closing );
652  if ( tmplink )
653  {
654  page->setPageAction( Okular::Page::Closing, createLinkFromPopplerLink( tmplink ) );
655  }
656  page->setDuration( p->duration() );
657  page->setLabel( p->label() );
658 
659  addFormFields( p, page );
660 // kWarning(PDFDebug).nospace() << page->width() << "x" << page->height();
661 
662 #ifdef PDFGENERATOR_DEBUG
663  kDebug(PDFDebug) << "load page" << i << "with rotation" << rotation << "and orientation" << orientation;
664 #endif
665  delete p;
666 
667  if (clear && pagesVector[i])
668  delete pagesVector[i];
669  }
670  else
671  {
672  page = new Okular::Page( i, defaultPageWidth, defaultPageHeight, Okular::Rotation0 );
673  }

675  pagesVector[i] = page;
