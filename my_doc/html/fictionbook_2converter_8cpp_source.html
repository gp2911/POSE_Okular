66 {

68  if ( !fbDocument.open() ) {
69  emit error( fbDocument.lastErrorString(), -1 );

71  }
72 
73  mTextDocument = new QTextDocument;
74  mCursor = new QTextCursor( mTextDocument );
75  mSectionCounter = 0;
76  mLocalLinks.clear();
77  mSectionMap.clear();

80 
84  mTextDocument->setPageSize( QSizeF( 600, 800 ) );

95  const QDomElement documentElement = document.documentElement();

98  emit error( i18n( "Document is not a valid FictionBook" ), -1 );
99  delete mCursor;
100  return 0;
101  }
102 
106  QDomElement element = documentElement.firstChildElement();
107  while ( !element.isNull() ) {
108  if ( element.tagName() == QLatin1String( "binary" ) ) {
109  if ( !convertBinary( element ) ) {
110  delete mCursor;
111  return 0;
112  }
113  }
114 
115  element = element.nextSiblingElement();
116  }
117 
121  element = documentElement.firstChildElement();
122  while ( !element.isNull() ) {
123  if ( element.tagName() == QLatin1String( "description" ) ) {
124  if ( !convertDescription( element ) ) {
125  delete mCursor;
126  return 0;
127  }
128  } else if ( element.tagName() == QLatin1String( "body" ) ) {
129  if ( !mTitleInfo->mCoverPage.isNull() ) {
130  convertCover( mTitleInfo->mCoverPage );
131  mCursor->insertBlock();
132  }
133 
134  QTextFrame *topFrame = mCursor->currentFrame();
135 
136  QTextFrameFormat frameFormat;
137  frameFormat.setBorder( 2 );
138  frameFormat.setPadding( 8 );
139  frameFormat.setBackground( Qt::lightGray );
140 
141  if ( !mTitleInfo->mTitle.isEmpty() ) {
142  mCursor->insertFrame( frameFormat );
143 
144  QTextCharFormat charFormat;
145  charFormat.setFontPointSize( 22 );
146  charFormat.setFontWeight( QFont::Bold );
147  mCursor->insertText( mTitleInfo->mTitle, charFormat );
148 
149  mCursor->setPosition( topFrame->lastPosition() );
150  }
151 
152  if ( !mTitleInfo->mAuthor.isEmpty() ) {
153  frameFormat.setBorder( 1 );
154  mCursor->insertFrame( frameFormat );
155 
156  QTextCharFormat charFormat;
157  charFormat.setFontPointSize( 10 );
158  mCursor->insertText( mTitleInfo->mAuthor, charFormat );
159

162  }
163 
164  mCursor->insertBlock();
165 
166  if ( !convertBody( element ) ) {
167  delete mCursor;
168  return 0;
169  }
170  }
171 
172  element = element.nextSiblingElement();
173  }
174 
178  if ( mTitleInfo ) {
179  if ( !mTitleInfo->mTitle.isEmpty() )
180  emit addMetaData( Okular::DocumentInfo::Title, mTitleInfo->mTitle );
181 
182  if ( !mTitleInfo->mAuthor.isEmpty() )
183  emit addMetaData( Okular::DocumentInfo::Author, mTitleInfo->mAuthor );
184  }
185 
186  if ( mDocumentInfo ) {
187  if ( !mDocumentInfo->mProducer.isEmpty() )
188  emit addMetaData( Okular::DocumentInfo::Producer, mDocumentInfo->mProducer );
189 
190  if ( mDocumentInfo->mDate.isValid() )
191  emit addMetaData( Okular::DocumentInfo::CreationDate,
192  KGlobal::locale()->formatDate( mDocumentInfo->mDate, KLocale::ShortDate ) );
193  }
194 
195  QMapIterator<QString, QPair<int, int> > it( mLocalLinks );
196  while ( it.hasNext() ) {
197  it.next();

202

204 
205  Okular::GotoAction *action = new Okular::GotoAction( QString(), viewport );

207  emit addAction( action, it.value().first, it.value().second );
208  }
209 
210  delete mCursor;
211 
212  return mTextDocument;

258 bool Converter::convertTitleInfo( const QDomElement &element )
259 {
260  delete mTitleInfo;
261  mTitleInfo = new TitleInfo;
262 
263  QDomElement child = element.firstChildElement();
264  while ( !child.isNull() ) {
265  if ( child.tagName() == QLatin1String( "genre" ) ) {
266  QString genre;
267  if ( !convertTextNode( child, genre ) )
268  return false;
269 
270  if ( !genre.isEmpty() )
271  mTitleInfo->mGenres.append( genre );
272  } else if ( child.tagName() == QLatin1String( "author" ) ) {
273  QString firstName, middleName, lastName, dummy;
274 
275  if ( !convertAuthor( child, firstName, middleName, lastName, dummy, dummy ) )
276  return false;
277 
278  mTitleInfo->mAuthor = QString( "%1 %2 %3" ).arg( firstName, middleName, lastName );
279  } else if ( child.tagName() == QLatin1String( "book-title" ) ) {
280  if ( !convertTextNode( child, mTitleInfo->mTitle ) )
281  return false;
282  } else if ( child.tagName() == QLatin1String( "keywords" ) ) {
283  QString keywords;
284  if ( !convertTextNode( child, keywords ) )
285  return false;
286 
287  mTitleInfo->mKeywords = keywords.split( ' ', QString::SkipEmptyParts );
288  } else if ( child.tagName() == QLatin1String( "date" ) ) {
289  if ( !convertDate( child, mTitleInfo->mDate ) )
290  return false;
291  } else if ( child.tagName() == QLatin1String( "coverpage" ) ) {
292  mTitleInfo->mCoverPage = child;
293  } else if ( child.tagName() == QLatin1String( "lang" ) ) {
294  if ( !convertTextNode( child, mTitleInfo->mLanguage ) )
295  return false;
296  }
297  child = child.nextSiblingElement();
298  }
299 
300  return true;
