60 void KIMGIOTest::testExifOrientation()
61 {
62  QFETCH( QString, imgPath );
63 
64  Okular::SettingsCore::instance( "kimgiotest" );
65  Okular::Document *m_document = new Okular::Document( 0 );
66  const KMimeType::Ptr mime = KMimeType::findByPath( imgPath );
67 
68  Okular::DocumentObserver *dummyDocumentObserver = new Okular::DocumentObserver();
69  m_document->addObserver( dummyDocumentObserver );
70 
71  // Load image
72  m_document->openDocument( imgPath, KUrl(), mime );
73  m_document->setRotation( 0 ); // Test the default rotation
74  QCOMPARE( m_document->pages(), 1u );
75 
76  // Check size
77  QCOMPARE( m_document->page(0)->width(), double(3) );
78  QCOMPARE( m_document->page(0)->height(), double(2) );
79 
80  // Generate pixmap
81  Okular::PixmapRequest *req = new Okular::PixmapRequest( dummyDocumentObserver, 0, 3, 2,
82  1, Okular::PixmapRequest::NoFeature );
83  m_document->requestPixmaps( QLinkedList<Okular::PixmapRequest*>() << req );
84  QVERIFY( m_document->page(0)->hasPixmap( dummyDocumentObserver, 3, 2 ) );
85 
86  // Obtain image
87  QImage img( 3, 2, QImage::Format_ARGB32_Premultiplied );
88  QPainter p( &img );
89  PagePainter::paintPageOnPainter( &p, m_document->page(0), dummyDocumentObserver, 0, 3, 2, QRect(0, 0, 3, 2) );
90 
91  // Verify pixel data
92  QCOMPARE( img.pixel(0, 0), qRgb(0, 0, 0) );
93  QCOMPARE( img.pixel(2, 1), qRgb(255, 255, 255) );
94 
95  m_document->removeObserver( dummyDocumentObserver );
96  delete dummyDocumentObserver;
97  delete m_document;
98 }
