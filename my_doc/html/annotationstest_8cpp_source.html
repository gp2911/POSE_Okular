56 void AnnotationTest::testDistance()
57 {
58  QFETCH(Okular::Annotation*, annotation);
59  QFETCH(double, x);
60  QFETCH(double, y);
61  QFETCH(int, distance);
62  Okular::AnnotationObjectRect* rect = new Okular::AnnotationObjectRect( annotation );
63  QCOMPARE( qRound( rect->distanceSqr( x, y, m_document->page(0)->width(), m_document->page(0)->height() ) ), distance );
64  delete rect;
65 }
