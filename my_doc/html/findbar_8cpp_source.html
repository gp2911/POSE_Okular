93 bool FindBar::eventFilter( QObject *target, QEvent *event ) {
94  if ( target == m_search )
95  {
96  if ( event->type() == QEvent::KeyPress )
97  {
98  QKeyEvent *keyEvent = static_cast<QKeyEvent *>( event );
99  if ( keyEvent->key() == Qt::Key_PageUp || keyEvent->key() == Qt::Key_PageDown )
100  {
101  emit forwardKeyPressEvent( keyEvent );
102  return true;
103  }
104  }
105  }

157 void FindBar::caseSensitivityChanged()
158 {
159  m_search->lineEdit()->setSearchCaseSensitivity( m_caseSensitiveAct->isChecked() ? Qt::CaseSensitive : Qt::CaseInsensitive );
160  if ( !m_active )
161  return;
162  Okular::Settings::setSearchCaseSensitive( m_caseSensitiveAct->isChecked() );
163  Okular::Settings::self()->writeConfig();
164  m_search->lineEdit()->restartSearch();
165 }
