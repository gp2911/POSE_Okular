35 static void which_ddjvu_message( const ddjvu_message_t *msg )
36 {
37 #ifdef KDJVU_DEBUG
38  kDebug() << "which_djvu_message(...):" << msg->m_any.tag;
39  switch( msg->m_any.tag )
40  {
41  case DDJVU_ERROR:
42  kDebug().nospace() << "ERROR: file " << msg->m_error.filename << ", line " << msg->m_error.lineno;
43  kDebug().nospace() << "ERROR: function '" << msg->m_error.function << "'";
44  kDebug().nospace() << "ERROR: '" << msg->m_error.message << "'";
45  break;
46  case DDJVU_INFO:
47  kDebug().nospace() << "INFO: '" << msg->m_info.message << "'";
48  break;
49  case DDJVU_CHUNK:
50  kDebug().nospace() << "CHUNK: '" << QByteArray( msg->m_chunk.chunkid ) << "'";
51  break;
52  case DDJVU_PROGRESS:
53  kDebug().nospace() << "PROGRESS: '" << msg->m_progress.percent << "'";
54  break;
55  default: ;
56  }
57 #else
58  Q_UNUSED( msg );
59 #endif

100 static miniexp_t find_second_in_pair( miniexp_t theexp, const char* which )
101 {
102  miniexp_t exp = theexp;
103  while ( exp )
104  {
105  miniexp_t cur = miniexp_car( exp );
106  if ( !miniexp_consp( cur ) || !miniexp_symbolp( miniexp_car( cur ) ) )
107  {
108  exp = miniexp_cdr( exp );
109  continue;
110  }
111 
112  const QString id = QString::fromUtf8( miniexp_to_name( miniexp_car( cur ) ) );
113  if ( id == QLatin1String( which ) )
114  return miniexp_cadr( cur );
115  exp = miniexp_cdr( exp );
116  }
117  return miniexp_nil;
118 }

120 static bool find_replace_or_add_second_in_pair( miniexp_t theexp, const char* which, miniexp_t replacement )
121 {

469 QImage KDjVu::Private::generateImageTile( ddjvu_page_t *djvupage, int& res,
470  int width, int row, int xdelta, int height, int col, int ydelta )
471 {
472  ddjvu_rect_t renderrect;
473  renderrect.x = row * xdelta;
474  renderrect.y = col * ydelta;
475  int realwidth = qMin( width - renderrect.x, xdelta );
476  int realheight = qMin( height - renderrect.y, ydelta );
477  renderrect.w = realwidth;
478  renderrect.h = realheight;
479 #ifdef KDJVU_DEBUG
480  kDebug() << "renderrect:" << renderrect;
481 #endif
482  ddjvu_rect_t pagerect;
483  pagerect.x = 0;
484  pagerect.y = 0;
485  pagerect.w = width;
486  pagerect.h = height;
487 #ifdef KDJVU_DEBUG
488  kDebug() << "pagerect:" << pagerect;
489 #endif
490  handle_ddjvu_messages( m_djvu_cxt, false );
491  QImage res_img( realwidth, realheight, QImage::Format_RGB32 );
492  // the following line workarounds a rare crash in djvulibre;
493  // it should be fixed with >= 3.5.21
494  ddjvu_page_get_width( djvupage );
495  res = ddjvu_page_render( djvupage, DDJVU_RENDER_COLOR,
496  &pagerect, &renderrect, m_format, res_img.bytesPerLine(), (char *)res_img.bits() );
497 #ifdef KDJVU_DEBUG
498  kDebug() << "rendering result:" << res;
499 #endif
500  handle_ddjvu_messages( m_djvu_cxt, false );
501 
502  return res_img;
503 }

505 void KDjVu::Private::readBookmarks()
506 {
507  if ( !m_djvu_document )
508  return;
509 
510  miniexp_t outline;
511  while ( ( outline = ddjvu_document_get_outline( m_djvu_document ) ) == miniexp_dummy )
512  handle_ddjvu_messages( m_djvu_cxt, true );
513 
514  if ( miniexp_listp( outline ) &&
515  ( miniexp_length( outline ) > 0 ) &&
516  miniexp_symbolp( miniexp_nth( 0, outline ) ) &&
517  ( QString::fromUtf8( miniexp_to_name( miniexp_nth( 0, outline ) ) ) == QLatin1String( "bookmarks" ) ) )
518  {
519  m_docBookmarks = new QDomDocument( "KDjVuBookmarks" );

521  ddjvu_miniexp_release( m_djvu_document, outline );

525 void KDjVu::Private::fillBookmarksRecurse( QDomDocument& maindoc, QDomNode& curnode,
526  miniexp_t exp, int offset )
527 {
528  if ( !miniexp_listp( exp ) )
529  return;
530 
531  int l = miniexp_length( exp );
532  for ( int i = qMax( offset, 0 ); i < l; ++i )
533  {
534  miniexp_t cur = miniexp_nth( i, exp );
535 
536  if ( miniexp_consp( cur ) && ( miniexp_length( cur ) > 0 ) &&
537  miniexp_stringp( miniexp_nth( 0, cur ) ) && miniexp_stringp( miniexp_nth( 1, cur ) ) )
538  {
539  QString title = QString::fromUtf8( miniexp_to_str( miniexp_nth( 0, cur ) ) );
540  QString dest = QString::fromUtf8( miniexp_to_str( miniexp_nth( 1, cur ) ) );
541  QDomElement el = maindoc.createElement( "item" );
542  el.setAttribute( "title", title );
543  if ( !dest.isEmpty() )
544  {
545  if ( dest.at( 0 ) == QLatin1Char( '#' ) )
546  {
547  dest.remove( 0, 1 );
548  bool isNumber = false;
549  dest.toInt( &isNumber );
550  if ( isNumber )
551  {
552  // it might be an actual page number, but could also be a page label
553  // so resolve the number, and get the real page number
554  int pageNo = pageWithName( dest );
555  if ( pageNo != -1 )
556  {
557  el.setAttribute( "PageNumber", QString::number( pageNo + 1 ) );
558  }
559  else
560  {
561  el.setAttribute( "PageNumber", dest );
562  }
563  }
564  else
565  {
566  el.setAttribute( "PageName", dest );
567  }
568  }
569  else
570  {
571  el.setAttribute( "URL", dest );
572  }
573  }
574  curnode.appendChild( el );
575  if ( !el.isNull() && ( miniexp_length( cur ) > 2 ) )
576  {
577  fillBookmarksRecurse( maindoc, el, cur, 2 );
578  }
579  }
580  }

583 void KDjVu::Private::readMetaData( int page )
584 {
585  if ( !m_djvu_document )
586  return;
587 
588  miniexp_t annots;
589  while ( ( annots = ddjvu_document_get_pageanno( m_djvu_document, page ) ) == miniexp_dummy )
590  handle_ddjvu_messages( m_djvu_cxt, true );
591 
592  if ( !miniexp_listp( annots ) || miniexp_length( annots ) == 0 )
593  return;
594 
595  miniexp_t exp = miniexp_nth( 0, annots );
596  int size = miniexp_length( exp );
597  if ( size <= 1 ||
598  qstrncmp( miniexp_to_name( miniexp_nth( 0, exp ) ), "metadata", 8 ) )
599  return;
600 
601  for ( int i = 1; i < size; ++i )
602  {
603  miniexp_t cur = miniexp_nth( i, exp );
604  if ( miniexp_length( cur ) != 2 )
605  continue;
606 
607  QString id = QString::fromUtf8( miniexp_to_name( miniexp_nth( 0, cur ) ) );
608  QString value = QString::fromUtf8( miniexp_to_str( miniexp_nth( 1, cur ) ) );
609  m_metaData[ id.toLower() ] = value;
610  }
611 }

613 int KDjVu::Private::pageWithName( const QString & name )
614 {
615  const int pageNo = m_pageNamesCache.value( name, -1 );
616  if ( pageNo != -1 )
617  return pageNo;
618 
619  const QByteArray utfName = name.toUtf8();
620  const int fileNum = ddjvu_document_get_filenum( m_djvu_document );
621  ddjvu_fileinfo_t info;
622  for ( int i = 0; i < fileNum; ++i )
623  {
624  if ( DDJVU_JOB_OK != ddjvu_document_get_fileinfo( m_djvu_document, i, &info ) )
625  continue;
626  if ( info.type != 'P' )
627  continue;
628  if ( ( utfName == info.id ) || ( utfName == info.name ) || ( utfName == info.title ) )
629  {
630  m_pageNamesCache.insert( name, info.pageno );
631  return info.pageno;
632  }
633  }
634  return -1;
635 }

663 bool KDjVu::openFile( const QString & fileName )
664 {
665  // first, close the old file
666  if ( d->m_djvu_document )
667  closeFile();
668 
669  // load the document...
670  d->m_djvu_document = ddjvu_document_create_by_filename( d->m_djvu_cxt, QFile::encodeName( fileName ), true );
671  if ( !d->m_djvu_document ) return false;
672  // ...and wait for its loading
673  wait_for_ddjvu_message( d->m_djvu_cxt, DDJVU_DOCINFO );
674  if ( ddjvu_document_decoding_error( d->m_djvu_document ) )
675  {
676  ddjvu_document_release( d->m_djvu_document );
677  d->m_djvu_document = 0;
678  return false;
679  }
680 
681  kDebug() << "# of pages:" << ddjvu_document_get_pagenum( d->m_djvu_document );
682  int numofpages = ddjvu_document_get_pagenum( d->m_djvu_document );
683  d->m_pages.clear();
684  d->m_pages.resize( numofpages );
685  d->m_pages_cache.clear();

797  return;

844  ulink->m_url = target;
845  link = ulink;
846  }
847  }
848  else if ( annotations &&
849  ( type == QLatin1String( "text" ) ||
850  type == QLatin1String( "line" ) ) )
851  {
852  if ( type == QLatin1String( "text" ) )
853  {
854  KDjVu::TextAnnotation * textann = new KDjVu::TextAnnotation( cur );
855  ann = textann;
856  }

858  {
859  KDjVu::LineAnnotation * lineann = new KDjVu::LineAnnotation( cur );
860  ann = lineann;
861  }
862  }
863  if ( link /* safety check */ && links )
864  {
865  link->m_area = KDjVu::Link::UnknownArea;

871  link->m_size = QSize( miniexp_to_int( miniexp_nth( 3, area ) ), miniexp_to_int( miniexp_nth( 4, area ) ) );
872  if ( type == QLatin1String( "rect" ) )
873  {
874  link->m_area = KDjVu::Link::RectArea;
875  }
876  else
877  {
878  link->m_area = KDjVu::Link::EllipseArea;
879  }
880  }

882  type == QLatin1String( "poly" ) )

885  QPolygon poly;

888  poly << QPoint( miniexp_to_int( miniexp_nth( j, area ) ), miniexp_to_int( miniexp_nth( j + 1, area ) ) );
889  }
890  link->m_poly = poly;
891  }

908 QImage KDjVu::image( int page, int width, int height, int rotation )
909 {
910  if ( d->m_cacheEnabled )
911  {
912  bool found = false;
913  QList<ImageCacheItem*>::Iterator it = d->mImgCache.begin(), itEnd = d->mImgCache.end();
914  for ( ; ( it != itEnd ) && !found; ++it )
915  {
916  ImageCacheItem* cur = *it;
917  if ( ( cur->page == page ) &&
918  ( rotation % 2 == 0
919  ? cur->width == width && cur->height == height
920  : cur->width == height && cur->height == width ) )
921  found = true;
922  }
923  if ( found )
924  {
925  // taking the element and pushing to the top of the list
926  --it;
927  ImageCacheItem* cur2 = *it;
928  d->mImgCache.erase( it );
929  d->mImgCache.push_front( cur2 );
930 
931  return cur2->img;
932  }
933  }
934 
935  if ( !d->m_pages_cache.at( page ) )
936  {
937  ddjvu_page_t *newpage = ddjvu_page_create_by_pageno( d->m_djvu_document, page );
938  // wait for the new page to be loaded
939  ddjvu_status_t sts;
940  while ( ( sts = ddjvu_page_decoding_status( newpage ) ) < DDJVU_JOB_OK )
941  handle_ddjvu_messages( d->m_djvu_cxt, true );
942  d->m_pages_cache[page] = newpage;
943  }
944  ddjvu_page_t *djvupage = d->m_pages_cache[page];
945 
946 /*
947  if ( ddjvu_page_get_rotation( djvupage ) != flipRotation( rotation ) )
948  {
949 // TODO: test documents with initial rotation != 0
950 // ddjvu_page_set_rotation( djvupage, m_pages.at( page )->orientation() );
951  ddjvu_page_set_rotation( djvupage, (ddjvu_page_rotation_t)flipRotation( rotation ) );
952  }
953 */
954 
955  static const int xdelta = 1500;
956  static const int ydelta = 1500;
957 
958  int xparts = width / xdelta + 1;
959  int yparts = height / ydelta + 1;
960 
961  QImage newimg;
962 
963  int res = 10000;
964  if ( ( xparts == 1 ) && ( yparts == 1 ) )
965  {
966  // only one part -- render at once with no need to auxiliary image
967  newimg = d->generateImageTile( djvupage, res,
968  width, 0, xdelta, height, 0, ydelta );
969  }
970  else
971  {
972  // more than one part -- need to render piece-by-piece and to compose
973  // the results
974  newimg = QImage( width, height, QImage::Format_RGB32 );
975  QPainter p;
976  p.begin( &newimg );
977  int parts = xparts * yparts;
978  for ( int i = 0; i < parts; ++i )
979  {
980  int row = i % xparts;
981  int col = i / xparts;
982  int tmpres = 0;
983  QImage tempp = d->generateImageTile( djvupage, tmpres,
984  width, row, xdelta, height, col, ydelta );
985  if ( tmpres )
986  {
987  p.drawImage( row * xdelta, col * ydelta, tempp );
988  }
989  res = qMin( tmpres, res );
990  }
991  p.end();
992  }
993 
994  if ( res && d->m_cacheEnabled )
995  {
996  // delete all the cached pixmaps for the current page with a size that
997  // differs no more than 35% of the new pixmap size
998  int imgsize = newimg.width() * newimg.height();
999  if ( imgsize > 0 )
1000  {
1001  for( int i = 0; i < d->mImgCache.count(); )
1002  {
1003  ImageCacheItem* cur = d->mImgCache.at(i);
1004  if ( ( cur->page == page ) &&
1005  ( abs( cur->img.width() * cur->img.height() - imgsize ) < imgsize * 0.35 ) )
1006  {
1007  d->mImgCache.removeAt( i );
1008  delete cur;
1009  }
1010  else
1011  ++i;
1012  }
1013  }
1014 
1015  // the image cache has too many elements, remove the last
1016  if ( d->mImgCache.size() >= 10 )
1017  {
1018  delete d->mImgCache.last();
1019  d->mImgCache.removeLast();
1020  }
1021  ImageCacheItem* ich = new ImageCacheItem( page, width, height, newimg );
1022  d->mImgCache.push_front( ich );
1023  }
1024 
1025  return newimg;
1026 }
