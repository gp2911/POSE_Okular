99  // Are we drawing text for a hyperlink? And are hyperlinks

266 void dviRenderer::draw_part(double current_dimconv, bool is_vfmacro)
267 {
268 #ifdef DEBUG_RENDER
269  kDebug(kvs::dvi) << "draw_part";
270 #endif
271 
272  qint32 RRtmp=0, WWtmp=0, XXtmp=0, YYtmp=0, ZZtmp=0;
273  quint8 ch;
274 
275  currinf.fontp = NULL;
276  currinf.set_char_p = &dviRenderer::set_no_char;
277 
278  int last_space_index = 0;
279  bool space_encountered = false;
280  bool after_space = false;
281  for (;;) {
282  space_encountered = false;
283  ch = readUINT8();
284  if (ch <= (unsigned char) (SETCHAR0 + 127)) {
285  (this->*currinf.set_char_p)(ch, ch);
286  } else
287  if (FNTNUM0 <= ch && ch <= (unsigned char) (FNTNUM0 + 63)) {
288  currinf.fontp = currinf.fonttable->value(ch - FNTNUM0);
289  if (currinf.fontp == NULL) {
290  errorMsg = i18n("The DVI code referred to font #%1, which was not previously defined.", ch - FNTNUM0);
291  return;
292  }
293  currinf.set_char_p = currinf.fontp->set_char_p;
294  } else {
295  qint32 a, b;
296 
297  switch (ch) {
298  case SET1:
299  case PUT1:
300  (this->*currinf.set_char_p)(ch, readUINT8());
301  break;
302 
303  case SETRULE:
304  if (is_vfmacro == false) {
305  word_boundary_encountered = true;
306  line_boundary_encountered = true;
307  }
308  /* Be careful, dvicopy outputs rules with height =
309  0x80000000. We don't want any SIGFPE here. */
310  a = readUINT32();
311  b = readUINT32();
312  b = ((long) (b * current_dimconv));
313  if (a > 0 && b > 0) {
314  int h = ((int) ROUNDUP(((long) (a * current_dimconv)), shrinkfactor * 65536));
315  int w = ((int) ROUNDUP(b, shrinkfactor * 65536));
316 
317  if (colorStack.isEmpty())
318  foreGroundPainter->fillRect( ((int) ((currinf.data.dvi_h) / (shrinkfactor * 65536))),
319  currinf.data.pxl_v - h + 1, w?w:1, h?h:1, globalColor );
320  else
321  foreGroundPainter->fillRect( ((int) ((currinf.data.dvi_h) / (shrinkfactor * 65536))),
322  currinf.data.pxl_v - h + 1, w?w:1, h?h:1, colorStack.top() );
323  }
324  currinf.data.dvi_h += b;
325  break;
326 
327  case PUTRULE:
328  if (is_vfmacro == false) {
329  word_boundary_encountered = true;
330  line_boundary_encountered = true;
331  }
332  a = readUINT32();
333  b = readUINT32();
334  a = ((long) (a * current_dimconv));
335  b = ((long) (b * current_dimconv));
336  if (a > 0 && b > 0) {
337  int h = ((int) ROUNDUP(a, shrinkfactor * 65536));
338  int w = ((int) ROUNDUP(b, shrinkfactor * 65536));
339  if (colorStack.isEmpty())
340  foreGroundPainter->fillRect( ((int) ((currinf.data.dvi_h) / (shrinkfactor * 65536))),
341  currinf.data.pxl_v - h + 1, w?w:1, h?h:1, globalColor );
342  else
343  foreGroundPainter->fillRect( ((int) ((currinf.data.dvi_h) / (shrinkfactor * 65536))),
344  currinf.data.pxl_v - h + 1, w?w:1, h?h:1, colorStack.top() );
345  }
346  break;
347 
348  case NOP:
349  break;
350 
351  case BOP:
352  if (is_vfmacro == false) {
353  word_boundary_encountered = true;
354  line_boundary_encountered = true;
355  }
356  command_pointer += 11 * 4;
357  currinf.data.dvi_h = 1200 << 16; // Reminder: DVI-coordinates start at (1",1") from top of page
358  currinf.data.dvi_v = 1200;
359  currinf.data.pxl_v = int(currinf.data.dvi_v/shrinkfactor);
360  currinf.data.w = currinf.data.x = currinf.data.y = currinf.data.z = 0;
361  break;
362 
363  case EOP:
364  // Check if we are just at the end of a virtual font macro.
365  if (is_vfmacro == false) {
366  // This is really the end of a page, and not just the end
367  // of a macro. Mark the end of the current word.
368  word_boundary_encountered = true;
369  line_boundary_encountered = true;
370  // Sanity check for the dvi-file: The DVI-standard asserts
371  // that at the end of a page, the stack should always be
372  // empty.
373  if (!stack.isEmpty()) {
374  kDebug(kvs::dvi) << "DRAW: The stack was not empty when the EOP command was encountered.";
375  errorMsg = i18n("The stack was not empty when the EOP command was encountered.");
376  return;
377  }
378  }
379  return;
380 
381  case PUSH:
382  stack.push(currinf.data);
383  break;
384 
385  case POP:
386  if (stack.isEmpty()) {
387  errorMsg = i18n("The stack was empty when a POP command was encountered.");
388  return;
389  } else
390  currinf.data = stack.pop();
391  word_boundary_encountered = true;
392  line_boundary_encountered = true;
393  break;
394 
395  case RIGHT1:
396  case RIGHT2:
397  case RIGHT3:
398  case RIGHT4:
399  RRtmp = readINT(ch - RIGHT1 + 1);
400 
401  // A horizontal motion in the range 4 * font space [f] < h <
402  // font space [f] will be treated as a kern that is not
403  // indicated in the printouts that DVItype produces between
404  // brackets. We allow a larger space in the negative
405  // direction than in the positive one, because TEX makes
406  // comparatively large backspaces when it positions
407  // accents. (comments stolen from the source of dvitype)
408  if ((is_vfmacro == false) &&
409  (currinf.fontp != 0) &&
410  ((RRtmp >= currinf.fontp->scaled_size_in_DVI_units/6) || (RRtmp <= -4*(currinf.fontp->scaled_size_in_DVI_units/6))) &&
411  (currentlyDrawnPage->textBoxList.size() > 0)) {
412  //currentlyDrawnPage->textBoxList[currentlyDrawnPage->textBoxList.size()-1].text += ' ';
413  space_encountered = true;
414  }
415  currinf.data.dvi_h += ((long) (RRtmp * current_dimconv));
416  break;
417 
418  case W1:
419  case W2:
420  case W3:
421  case W4:
422  WWtmp = readINT(ch - W0);
423  currinf.data.w = ((long) (WWtmp * current_dimconv));
424  case W0:
425  if ((is_vfmacro == false) &&
426  (currinf.fontp != 0) &&
427  ((WWtmp >= currinf.fontp->scaled_size_in_DVI_units/6) || (WWtmp <= -4*(currinf.fontp->scaled_size_in_DVI_units/6))) &&
428  (currentlyDrawnPage->textBoxList.size() > 0) ) {
429  //currentlyDrawnPage->textBoxList[currentlyDrawnPage->textBoxList.size()-1].text += ' ';
430  space_encountered = true;
431  }
432  currinf.data.dvi_h += currinf.data.w;
433  break;
434 
435  case X1:
436  case X2:
437  case X3:
438  case X4:
439  XXtmp = readINT(ch - X0);
440  currinf.data.x = ((long) (XXtmp * current_dimconv));
441  case X0:
442  if ((is_vfmacro == false) &&
443  (currinf.fontp != 0) &&
444  ((XXtmp >= currinf.fontp->scaled_size_in_DVI_units/6) || (XXtmp <= -4*(currinf.fontp->scaled_size_in_DVI_units/6))) &&
445  (currentlyDrawnPage->textBoxList.size() > 0)) {
446  //currentlyDrawnPage->textBoxList[currentlyDrawnPage->textBoxList.size()-1].text += ' ';
447  space_encountered = true;
448  }
449  currinf.data.dvi_h += currinf.data.x;
450  break;
451 
452  case DOWN1:
453  case DOWN2:
454  case DOWN3:
455  case DOWN4:
456  {
457  qint32 DDtmp = readINT(ch - DOWN1 + 1);
458  if ((is_vfmacro == false) &&
459  (currinf.fontp != 0) &&
460  (abs(DDtmp) >= 5*(currinf.fontp->scaled_size_in_DVI_units/6)) &&
461  (currentlyDrawnPage->textBoxList.size() > 0)) {
462  word_boundary_encountered = true;
463  line_boundary_encountered = true;
464  space_encountered = true;
465  if (abs(DDtmp) >= 10*(currinf.fontp->scaled_size_in_DVI_units/6))
466  currentlyDrawnPage->textBoxList[currentlyDrawnPage->textBoxList.size()-1].text += '\n';
467  }
468  currinf.data.dvi_v += ((long) (DDtmp * current_dimconv))/65536;
469  currinf.data.pxl_v = int(currinf.data.dvi_v/shrinkfactor);
470  }
471  break;
472 
473  case Y1:
474  case Y2:
475  case Y3:
476  case Y4:
477  YYtmp = readINT(ch - Y0);
478  currinf.data.y = ((long) (YYtmp * current_dimconv));
479  case Y0:
480  if ((is_vfmacro == false) &&
481  (currinf.fontp != 0) &&
482  (abs(YYtmp) >= 5*(currinf.fontp->scaled_size_in_DVI_units/6)) &&
483  (currentlyDrawnPage->textBoxList.size() > 0)) {
484  word_boundary_encountered = true;
485  line_boundary_encountered = true;
486  space_encountered = true;
487  if (abs(YYtmp) >= 10*(currinf.fontp->scaled_size_in_DVI_units/6))
488  currentlyDrawnPage->textBoxList[currentlyDrawnPage->textBoxList.size()-1].text += '\n';
489  }
490  currinf.data.dvi_v += currinf.data.y/65536;
491  currinf.data.pxl_v = int(currinf.data.dvi_v/shrinkfactor);
492  break;
493 
494  case Z1:
495  case Z2:
496  case Z3:
497  case Z4:
498  ZZtmp = readINT(ch - Z0);
499  currinf.data.z = ((long) (ZZtmp * current_dimconv));
500  case Z0:
501  if ((is_vfmacro == false) &&
502  (currinf.fontp != 0) &&
503  (abs(ZZtmp) >= 5*(currinf.fontp->scaled_size_in_DVI_units/6)) &&
504  (currentlyDrawnPage->textBoxList.size() > 0)) {
505  word_boundary_encountered = true;
506  line_boundary_encountered = true;
507  space_encountered = true;
508  if (abs(ZZtmp) >= 10*(currinf.fontp->scaled_size_in_DVI_units/6))
509  currentlyDrawnPage->textBoxList[currentlyDrawnPage->textBoxList.size()-1].text += '\n';
510  }
511  currinf.data.dvi_v += currinf.data.z/65536;
512  currinf.data.pxl_v = int(currinf.data.dvi_v/shrinkfactor);
513  break;
514 
515  case FNT1:
516  case FNT2:
517  case FNT3:
518  currinf.fontp = currinf.fonttable->value(readUINT(ch - FNT1 + 1));
519  if (currinf.fontp == NULL) {
520  errorMsg = i18n("The DVI code referred to a font which was not previously defined.");
521  return;
522  }
523  currinf.set_char_p = currinf.fontp->set_char_p;
524  break;
525 
526  case FNT4:
527  currinf.fontp = currinf.fonttable->value(readINT(ch - FNT1 + 1));
528  if (currinf.fontp == NULL) {
529  errorMsg = i18n("The DVI code referred to a font which was not previously defined.");
530  return;
531  }
532  currinf.set_char_p = currinf.fontp->set_char_p;
533  break;
534 
535  case XXX1:
536  case XXX2:
537  case XXX3:
538  case XXX4:
539  if (is_vfmacro == false) {
540  word_boundary_encountered = true;
541  line_boundary_encountered = true;
542  space_encountered = true;
543  }
544  a = readUINT(ch - XXX1 + 1);
545  if (a > 0) {
546  char *cmd = new char[a+1];
547  strncpy(cmd, (char *)command_pointer, a);
548  command_pointer += a;
549  cmd[a] = '\0';
550  applicationDoSpecial(cmd);
551  delete [] cmd;
552  }
553  break;
554 
555  case FNTDEF1:
556  case FNTDEF2:
557  case FNTDEF3:
558  case FNTDEF4:
