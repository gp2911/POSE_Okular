101 void VideoWidget::Private::load()
102 {
103  if ( loaded )
104  return;
105 
106  loaded = true;
107 
108  QString url = movie->url();
109  KUrl newurl;
110  if ( QDir::isRelativePath( url ) )
111  {
112  newurl = document->currentDocument();
113  newurl.setFileName( url );
114  }
115  else
116  {
117  newurl = url;
118  }
119  if ( newurl.isLocalFile() )
120  player->load( newurl.toLocalFile() );
121  else

125  q, SLOT( stateChanged( Phonon::State, Phonon::State ) ) );
126 
127  seekSlider->setEnabled( true );

175 void VideoWidget::Private::finished()
176 {
177  switch ( movie->playMode() )
178  {
179  case Okular::Movie::PlayOnce:
180  case Okular::Movie::PlayOpen:
181  // playback has ended, nothing to do
182  stopAction->setEnabled( false );
183  setupPlayPauseAction( PlayMode );
184  if ( movie->playMode() == Okular::Movie::PlayOnce )
185  controlBar->setVisible( false );
186  videoStopped();
187  break;
188  case Okular::Movie::PlayRepeat:
189  // repeat the playback
190  player->play();
191  break;
192  case Okular::Movie::PlayPalindrome:
193  // FIXME we should play backward, but we cannot
194  player->play();
195  break;
196  }
197 }

381 bool VideoWidget::eventFilter( QObject * object, QEvent * event )
382 {
383  if ( object == d->player || object == d->posterImagePage )
384  {
385  switch ( event->type() )
386  {
387  case QEvent::MouseButtonPress:
388  {
389  QMouseEvent * me = static_cast< QMouseEvent * >( event );
390  if ( me->button() == Qt::LeftButton )
391  {
392  if ( !d->player->isPlaying() )
393  {
394  play();
395  }
396  event->accept();
397  }
398  }
399  case QEvent::Wheel:
400  {
401  if ( object == d->posterImagePage )
402  {
403  QWheelEvent * we = static_cast< QWheelEvent * >( event );
404 
405  // forward wheel events to parent widget
406  QWheelEvent *copy = new QWheelEvent( we->pos(), we->globalPos(), we->delta(), we->buttons(), we->modifiers(), we->orientation() );
407  QCoreApplication::postEvent( parentWidget(), copy );
408  }
409  break;
410  }
411  default: ;
412  }
413  }
414 
415  return false;
416 }
