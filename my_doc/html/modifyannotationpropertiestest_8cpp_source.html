138 void ModifyAnnotationPropertiesTest::testModifyAnnotationPropertiesWithRotation_Bug318828()
139 {
140  Okular::NormalizedRect boundingRect = Okular::NormalizedRect( 0.1, 0.1, 0.15, 0.15 );
141  Okular::NormalizedRect transformedBoundingRect;
142  m_annot1->setBoundingRectangle( boundingRect );
143  m_document->addPageAnnotation( 0, m_annot1 );
144 
145  transformedBoundingRect = m_annot1->transformedBoundingRectangle();
146 
147  // Before page rotation boundingRect and transformedBoundingRect should be equal
148  QCOMPARE( boundingRect, transformedBoundingRect );
149  m_document->setRotation( 1 );
150 
151  // After rotation boundingRect should remain unchanged but
152  // transformedBoundingRect should no longer equal boundingRect
153  QCOMPARE( boundingRect, m_annot1->boundingRectangle() );
154  transformedBoundingRect = m_annot1->transformedBoundingRectangle();
155  QVERIFY( !( boundingRect == transformedBoundingRect ) );
156 
157  // Modifying the properties of m_annot1 while page is rotated shouldn't
158  // alter either boundingRect or transformedBoundingRect
159  m_document->prepareToModifyAnnotationProperties( m_annot1 );
160  m_annot1->style().setOpacity( 0.5 );
161  m_document->modifyPageAnnotationProperties( 0, m_annot1 );
162 
163  QCOMPARE( boundingRect, m_annot1->boundingRectangle() );
