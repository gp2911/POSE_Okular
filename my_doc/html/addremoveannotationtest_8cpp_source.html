52 void AddRemoveAnnotationTest::testAddAnnotations()
53 {
54  // Undo and Redo should be unavailable when docuemnt is first opened.
55  QVERIFY( !m_document->canUndo() );
56  QVERIFY( !m_document->canRedo() );
57 
58  // Create two distinct text annotations
59  Okular::Annotation *annot1 = new Okular::TextAnnotation();
60  annot1->setBoundingRectangle( Okular::NormalizedRect( 0.1, 0.1, 0.15, 0.15 ) );
61  annot1->setContents( "annot contents" );
62 
63  Okular::Annotation *annot2 = new Okular::TextAnnotation();
64  annot2->setBoundingRectangle( Okular::NormalizedRect( 0.2, 0.2, 0.3, 0.4 ) );
65  annot2->setContents( "annot contents" );
66 
67  // The two annotations shold have different properties XML strings
68  QVERIFY( TestingUtils::getAnnotationXml( annot1 ) != TestingUtils::getAnnotationXml( annot2 ) );
69 
70  // We start with no annotations in the docuemnt
71  QVERIFY( m_document->page( 0 )->annotations().size() == 0 );
72 
73  // After adding annot1 we should have one annotation in the page and it should be annot1.
74  m_document->addPageAnnotation( 0, annot1 );
75  QVERIFY( m_document->page( 0 )->annotations().size() == 1 );
76  QCOMPARE( annot1, m_document->page( 0 )->annotations().first() );
77 
78  // Record the properties and name of annot1 just after insertion for later comparisons
79  QString origLine1Xml = TestingUtils::getAnnotationXml( annot1 );
80  QString annot1Name = annot1->uniqueName();
81  QVERIFY( !annot1Name.isEmpty() );
82 
83  // Now undo the addition of annot1 and verify that annot1's properties haven't changed
84  m_document->undo();
85  QVERIFY( m_document->page( 0 )->annotations().empty() );
86  QVERIFY( !m_document->canUndo() );
87  QVERIFY( m_document->canRedo() );
88  QCOMPARE( TestingUtils::getAnnotationXml( annot1 ), origLine1Xml );
89 
90  // redo addition of annot1
91  m_document->redo();
92  QVERIFY( m_document->page( 0 )->annotations().size() == 1 );
93  QVERIFY( annot1 == m_document->page( 0 )->annotations().first() );
94  QCOMPARE( TestingUtils::getAnnotationXml( annot1 ), origLine1Xml );
95 
96  // undo once more
97  m_document->undo();
98  QVERIFY( m_document->page( 0 )->annotations().empty() );
99  QVERIFY( !m_document->canUndo() );
100  QVERIFY( m_document->canRedo() );
101  QCOMPARE( TestingUtils::getAnnotationXml( annot1 ), origLine1Xml );
102 
103  // Set AnnotationDisposeWatcher dispose function on annot1 so we can detect
104  // when it is deleted
105  annot1->setDisposeDataFunction( TestingUtils::AnnotationDisposeWatcher::disposeAnnotation );
106  TestingUtils::AnnotationDisposeWatcher::resetDisposedAnnotationName();
107  QCOMPARE( TestingUtils::AnnotationDisposeWatcher::disposedAnnotationName(), QString() );
108 
109  // now add annot2
110  m_document->addPageAnnotation( 0, annot2 );
111  QString annot2Name = annot2->uniqueName();
112  QVERIFY( !annot2Name.isEmpty() );
113  QVERIFY( annot1Name != annot2Name );
114  QVERIFY( m_document->page( 0 )->annotations().size() == 1 );
115  QCOMPARE( annot2, m_document->page( 0 )->annotations().first() );
116 
117  // Check that adding annot2 while annot1 was in the unadded state triggered the deletion of annot1
118  QVERIFY( TestingUtils::AnnotationDisposeWatcher::disposedAnnotationName() == annot1Name );
119 }

142 void AddRemoveAnnotationTest::testRemoveAnnotations()
143 {
144  // Undo and Redo should be unavailable when docuemnt is first opened.
145  QVERIFY( !m_document->canUndo() );
146  QVERIFY( !m_document->canRedo() );
147 
148  // Create two distinct text annotations
149  Okular::Annotation *annot1 = new Okular::TextAnnotation();
150  annot1->setBoundingRectangle( Okular::NormalizedRect( 0.1, 0.1, 0.15, 0.15 ) );
151  annot1->setContents( "annot contents" );
152 
153  Okular::Annotation *annot2 = new Okular::TextAnnotation();
154  annot2->setBoundingRectangle( Okular::NormalizedRect( 0.2, 0.2, 0.3, 0.4 ) );
155  annot2->setContents( "annot contents" );
156 
157  // Add annot1 and annot2 to document
158  m_document->addPageAnnotation( 0, annot1 );
159  m_document->addPageAnnotation( 0, annot2 );
160  QVERIFY( m_document->page( 0 )->annotations().size() == 2 );
161  QVERIFY( m_document->page( 0 )->annotations().contains(annot1) );
162  QVERIFY( m_document->page( 0 )->annotations().contains(annot2) );
163 
164  // Now remove annot1
165  m_document->removePageAnnotation( 0, annot1 );
166  QVERIFY( m_document->page( 0 )->annotations().size() == 1 );
167  QVERIFY( m_document->page( 0 )->annotations().contains(annot2) );
168 
169  // Undo removal of annot1
170  m_document->undo();
171  QVERIFY( m_document->page( 0 )->annotations().size() == 2 );
172  QVERIFY( m_document->page( 0 )->annotations().contains(annot1) );
173  QVERIFY( m_document->page( 0 )->annotations().contains(annot2) );
174 
175  // Redo removal
176  m_document->redo();
177  QVERIFY( m_document->page( 0 )->annotations().size() == 1 );
178  QVERIFY( m_document->page( 0 )->annotations().contains(annot2) );
179 
180  // Verify that annot1 is disposed of if document is closed with annot1 in removed state
181  QString annot1Name = annot1->uniqueName();
182  annot1->setDisposeDataFunction( TestingUtils::AnnotationDisposeWatcher::disposeAnnotation );
183  TestingUtils::AnnotationDisposeWatcher::resetDisposedAnnotationName();
184  QVERIFY( TestingUtils::AnnotationDisposeWatcher::disposedAnnotationName() == QString() );
185  m_document->closeDocument();
186  QVERIFY( TestingUtils::AnnotationDisposeWatcher::disposedAnnotationName() == annot1Name );
187 }
