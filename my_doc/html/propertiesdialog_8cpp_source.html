229 void PropertiesDialog::showFontsMenu(const QPoint &pos)
230 {
231  QTreeView *view = static_cast<QTreeView*>(sender());
232  QModelIndex index = view->indexAt(pos);
233  if (index.data(IsExtractableRole).toBool())
234  {
235  QMenu *menu = new QMenu(this);
236  menu->addAction( i18nc("@action:inmenu", "&Extract Font") );
237  QAction *result = menu->exec(view->viewport()->mapToGlobal(pos));
238  if (result)
239  {
240  Okular::FontInfo fi = index.data(FontInfoRole).value<Okular::FontInfo>();
241  const QString caption = i18n( "Where do you want to save %1?", fi.name() );
242  const QString path = KFileDialog::getSaveFileName( fi.name(), QString(), this, caption, KFileDialog::ConfirmOverwrite );
243  if ( path.isEmpty() )
244  return;
245 
246  QFile f( path );
247  if ( f.open( QIODevice::WriteOnly ) )
248  {
249  f.write( m_document->fontData(fi) );
250  f.close();
251  }
252  else
253  {
254  KMessageBox::error( this, i18n( "Could not open \"%1\" for writing. File was not saved.", path ) );
255  }
256  }
257  }

373 QVariant FontsListModel::data( const QModelIndex &index, int role ) const
374 {
375  if ( !index.isValid() || index.row() < 0 || index.row() >= m_fonts.count() )
376  return QVariant();
377 
378  switch ( role )
379  {
380  case Qt::DisplayRole:
381  switch ( index.column() )
382  {
383  case 0:
384  {
385  QString fontname = m_fonts.at( index.row() ).name();
386  return fontname.isEmpty() ? i18nc( "font name not available (empty)", "[n/a]" ) : fontname;
387  break;
388  }
389  case 1:
390  return descriptionForFontType( m_fonts.at( index.row() ).type() );
391  break;
392  case 2:
393  return pathOrDescription( m_fonts.at( index.row() ) );
394  break;
395  }
396  break;
397  case Qt::ToolTipRole:
398  {
399  QString fontname = m_fonts.at( index.row() ).name();
400  if ( fontname.isEmpty() )
401  fontname = i18n( "Unknown font" );
402  QString tooltip = QString::fromLatin1( "<html><b>" ) + fontname + QString::fromLatin1( "</b>" );
403  if ( m_fonts.at( index.row() ).embedType() == Okular::FontInfo::NotEmbedded )
404  tooltip += QString::fromLatin1( " (<span style=\"font-family: '%1'\">%2</span>)" ).arg( fontname ).arg( fontname );
405  tooltip += QString::fromLatin1( "<br />" ) + i18n( "Embedded: %1", descriptionForEmbedType( m_fonts.at( index.row() ).embedType() ) );
406  tooltip += QString::fromLatin1( "</html>" );
407  return tooltip;
408  break;
409  }
410  case IsExtractableRole:
411  {
412  return m_fonts.at( index.row() ).canBeExtracted();
413  }
414  case FontInfoRole:
415  {
416  QVariant v;
417  v.setValue( m_fonts.at( index.row() ) );
418  return v;
419  }
420  }
421 
422  return QVariant();
423 }
