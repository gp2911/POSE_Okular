110 void AnnotsPropertiesDialog::setCaptionTextbyAnnotType()
111 {
112  Okular::Annotation::SubType type=m_annot->subType();
113  QString captiontext;
114  switch(type)
115  {
116  case Okular::Annotation::AText:
117  if(((Okular::TextAnnotation*)m_annot)->textType()==Okular::TextAnnotation::Linked)
118  captiontext = i18n( "Pop-up Note Properties" );
119  else
120  captiontext = i18n( "Inline Note Properties" );
121  break;
122  case Okular::Annotation::ALine:
123  if ( ((Okular::LineAnnotation*)m_annot)->linePoints().count() == 2 )
124  captiontext = i18n( "Straight Line Properties" );
125  else
126  captiontext = i18n( "Polygon Properties" );
127  break;
128  case Okular::Annotation::AGeom:
129  captiontext = i18n( "Geometry Properties" );
130  break;
131  case Okular::Annotation::AHighlight:
132  captiontext = i18n( "Text Markup Properties" );
133  break;
134  case Okular::Annotation::AStamp:
135  captiontext = i18n( "Stamp Properties" );
136  break;
137  case Okular::Annotation::AInk:
138  captiontext = i18n( "Freehand Line Properties" );
139  break;
140  case Okular::Annotation::ACaret:
141  captiontext = i18n( "Caret Properties" );
142  break;
143  case Okular::Annotation::AFileAttachment:
144  captiontext = i18n( "File Attachment Properties" );
145  break;
146  case Okular::Annotation::ASound:
147  captiontext = i18n( "Sound Properties" );
148  break;
149  case Okular::Annotation::AMovie:
150  captiontext = i18n( "Movie Properties" );
151  break;
152  default:
153  captiontext = i18n( "Annotation Properties" );
154  break;
155  }
156  setCaption( captiontext );
157 }
