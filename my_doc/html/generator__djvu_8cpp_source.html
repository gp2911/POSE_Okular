33 static void recurseCreateTOC( QDomDocument &maindoc, const QDomNode &parent, QDomNode &parentDestination, KDjVu *djvu )
34 {
35  QDomNode n = parent.firstChild();
36  while( !n.isNull() )
37  {
38  QDomElement el = n.toElement();
39 
40  QDomElement newel = maindoc.createElement( el.attribute( "title" ) );
41  parentDestination.appendChild( newel );
42 
43  QString dest;
44  if ( !( dest = el.attribute( "PageNumber" ) ).isEmpty() )
45  {
46  Okular::DocumentViewport vp;
47  vp.pageNumber = dest.toInt() - 1;
48  newel.setAttribute( "Viewport", vp.toString() );
49  }
50  else if ( !( dest = el.attribute( "PageName" ) ).isEmpty() )
51  {
52  Okular::DocumentViewport vp;
53  vp.pageNumber = djvu->pageNumber( dest );
54  newel.setAttribute( "Viewport", vp.toString() );
55  }
56  else if ( !( dest = el.attribute( "URL" ) ).isEmpty() )
57  {
58  newel.setAttribute( "URL", dest );
59  }
60 
61  if ( el.hasChildNodes() )
62  {
63  recurseCreateTOC( maindoc, n, newel, djvu );
64  }
65  n = n.nextSibling();
66  }
67 }

266 void DjVuGenerator::loadPages( QVector<Okular::Page*> & pagesVector, int rotation )
267 {
268  const QVector<KDjVu::Page*> &djvu_pages = m_djvu->pages();
269  int numofpages = djvu_pages.count();
270  pagesVector.resize( numofpages );
271 
272  for ( int i = 0; i < numofpages; ++i )
273  {
274  const KDjVu::Page *p = djvu_pages.at( i );
275  if (pagesVector[i])
276  delete pagesVector[i];
277  int w = p->width();
278  int h = p->height();
279  if ( rotation % 2 == 1 )
280  qSwap( w, h );
281  Okular::Page *page = new Okular::Page( i, w, h, (Okular::Rotation)( p->orientation() + rotation ) );
282  pagesVector[i] = page;
283 
284  QList<KDjVu::Annotation*> annots;
285  QList<KDjVu::Link*> links;
286  userMutex()->lock();
287  m_djvu->linksAndAnnotationsForPage( i, &links, &annots );
288  userMutex()->unlock();
289  if ( !links.isEmpty() )
290  {
291  QLinkedList<Okular::ObjectRect *> rects;
292  QList<KDjVu::Link*>::ConstIterator it = links.constBegin();
293  QList<KDjVu::Link*>::ConstIterator itEnd = links.constEnd();
294  for ( ; it != itEnd; ++it )
295  {
296  KDjVu::Link *curlink = (*it);
297  Okular::ObjectRect *newrect = convertKDjVuLink( i, curlink );
298  if ( newrect )
299  rects.append( newrect );
300  // delete the links as soon as we process them
301  delete curlink;
302  }
303  if ( rects.count() > 0 )
304  page->setObjectRects( rects );
305  }
306  if ( !annots.isEmpty() )
307  {
308  QList<KDjVu::Annotation*>::ConstIterator it = annots.constBegin();
309  QList<KDjVu::Annotation*>::ConstIterator itEnd = annots.constEnd();
310  for ( ; it != itEnd; ++it )
311  {
312  KDjVu::Annotation *ann = (*it);
313  Okular::Annotation *newann = convertKDjVuAnnotation( w, h, ann );
314  if ( newann )
315  page->addAnnotation( newann );
316  // delete the annotations as soon as we process them
317  delete ann;
318  }
319  }
320  }

323 Okular::ObjectRect* DjVuGenerator::convertKDjVuLink( int page, KDjVu::Link * link ) const
324 {
325  int newpage = -1;
326  Okular::Action *newlink = 0;
327  Okular::ObjectRect *newrect = 0;
328  switch ( link->type() )
329  {
330  case KDjVu::Link::PageLink:
331  {
332  KDjVu::PageLink* l = static_cast<KDjVu::PageLink*>( link );
333  bool ok = true;
334  QString target = l->page();
335  if ( ( target.length() > 0 ) && target.at(0) == QLatin1Char( '#' ) )
336  target.remove( 0, 1 );
337  int tmppage = target.toInt( &ok );
338  if ( ok || target.isEmpty() )
339  {
340  Okular::DocumentViewport vp;
341  if ( !target.isEmpty() )
342  {
343  vp.pageNumber = ( target.at(0) == QLatin1Char( '+' ) || target.at(0) == QLatin1Char( '-' ) ) ? page + tmppage : tmppage - 1;
344  newpage = vp.pageNumber;
345  }
346  newlink = new Okular::GotoAction( QString(), vp );
347  }
348  break;
349  }
350  case KDjVu::Link::UrlLink:
351  {
352  KDjVu::UrlLink* l = static_cast<KDjVu::UrlLink*>( link );
353  QString url = l->url();
354  newlink = new Okular::BrowseAction( url );
355  break;
356  }
357  }
358  if ( newlink )
359  {
360  const KDjVu::Page* p = m_djvu->pages().value( newpage == -1 ? page : newpage );
361  if ( !p )
362  p = m_djvu->pages().at( page );
363  int width = p->width();
364  int height = p->height();
365  bool scape_orientation = false; // hack by tokoe, should always create default page
366  if ( scape_orientation )
367  qSwap( width, height );
368  switch ( link->areaType() )
369  {
370  case KDjVu::Link::RectArea:
371  case KDjVu::Link::EllipseArea:
372  {
373  QRect r( QPoint( link->point().x(), p->height() - link->point().y() - link->size().height() ), link->size() );
374  bool ellipse = ( link->areaType() == KDjVu::Link::EllipseArea );
375  newrect = new Okular::ObjectRect( Okular::NormalizedRect( Okular::Utils::rotateRect( r, width, height, 0 ), width, height ), ellipse, Okular::ObjectRect::Action, newlink );
376  break;
377  }
378  case KDjVu::Link::PolygonArea:
379  {
380  QPolygon poly = link->polygon();
381  QPolygonF newpoly;
382  for ( int i = 0; i < poly.count(); ++i )
383  {
384  int x = poly.at(i).x();
385  int y = poly.at(i).y();
386  if ( scape_orientation )
387  qSwap( x, y );
388  else
389  {
390  y = height - y;
391  }
392  newpoly << QPointF( (double)(x)/width, (double)(y)/height );
393  }
394  if ( !newpoly.isEmpty() )
395  {
396  newpoly << newpoly.first();
397  newrect = new Okular::ObjectRect( newpoly, Okular::ObjectRect::Action, newlink );
398  }
399  break;
400  }
401  default: ;
402  }
403  if ( !newrect )
404  {
405  delete newlink;
406  }
407  }
408  return newrect;
409 }

411 Okular::Annotation* DjVuGenerator::convertKDjVuAnnotation( int w, int h, KDjVu::Annotation * ann ) const
412 {
413  Okular::Annotation *newann = 0;
414  switch ( ann->type() )
415  {
416  case KDjVu::Annotation::TextAnnotation:
417  {
418  KDjVu::TextAnnotation* txtann = static_cast<KDjVu::TextAnnotation*>( ann );
419  Okular::TextAnnotation * newtxtann = new Okular::TextAnnotation();
420  // boundary
421  QRect rect( QPoint( txtann->point().x(), h - txtann->point().y() - txtann->size().height() ), txtann->size() );
422  newtxtann->setBoundingRectangle( Okular::NormalizedRect( Okular::Utils::rotateRect( rect, w, h, 0 ), w, h ) );
423  // type
424  newtxtann->setTextType( txtann->inlineText() ? Okular::TextAnnotation::InPlace : Okular::TextAnnotation::Linked );
425  newtxtann->style().setOpacity( txtann->color().alphaF() );
426  // FIXME remove once the annotation text handling is fixed
427  newtxtann->setContents( ann->comment() );
428  newann = newtxtann;
429  break;
430  }
431  case KDjVu::Annotation::LineAnnotation:
432  {
433  KDjVu::LineAnnotation* lineann = static_cast<KDjVu::LineAnnotation*>( ann );
434  Okular::LineAnnotation * newlineann = new Okular::LineAnnotation();
435  // boundary
436  QPoint a( lineann->point().x(), h - lineann->point().y() );
437  QPoint b( lineann->point2().x(), h - lineann->point2().y() );
438  QRect rect = QRect( a, b ).normalized();
439  newlineann->setBoundingRectangle( Okular::NormalizedRect( Okular::Utils::rotateRect( rect, w, h, 0 ), w, h ) );
440  // line points
441  QLinkedList<Okular::NormalizedPoint> points;
442  points.append( Okular::NormalizedPoint( a.x(), a.y(), w, h ) );
443  points.append( Okular::NormalizedPoint( b.x(), b.y(), w, h ) );
444  newlineann->setLinePoints( points );
445  // arrow?
446  if ( lineann->isArrow() )
447  newlineann->setLineEndStyle( Okular::LineAnnotation::OpenArrow );
448  // width
449  newlineann->style().setWidth( lineann->width() );
450  newann = newlineann;
451  break;
452  }
453  }
454  if ( newann )
455  {
456  // setting the common parameters
457  newann->style().setColor( ann->color() );
458  newann->setContents( ann->comment() );
459  // creating an id as name for the annotation
460  QString uid = QUuid::createUuid().toString();
461  uid.remove( 0, 1 );
462  uid.chop( 1 );
463  uid.remove( QLatin1Char( '-' ) );
464  newann->setUniqueName( uid );
465  // is external
466  newann->setFlags( newann->flags() | Okular::Annotation::External );
467  }
468  return newann;
469 }
