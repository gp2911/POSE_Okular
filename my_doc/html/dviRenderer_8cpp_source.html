396 {
397  QFile f(filename);
398  if (!f.open(QIODevice::ReadOnly))
399  return false;
400 
401  unsigned char test[4];
402  if ( f.read( (char *)test,2)<2 || test[0] != 247 || test[1] != 2 )
403  return false;
404 
405  int n = f.size();
406  if ( n < 134 ) // Too short for a dvi file
407  return false;

413  return false;

567 Anchor dviRenderer::parseReference(const QString &reference)
568 {
569  QMutexLocker locker(&mutex);
570 
571 #ifdef DEBUG_DVIRENDERER
572  kError(kvs::dvi) << "dviRenderer::parseReference( " << reference << " ) called" << endl;
573 #endif
574 
575  if (dviFile == 0)
576  return Anchor();
577 
578  // case 1: The reference is a number, which we'll interpret as a
579  // page number.
580  bool ok;
581  int page = reference.toInt ( &ok );
582  if (ok == true) {
583  if (page < 0)
584  page = 0;
585  if (page > dviFile->total_pages)
586  page = dviFile->total_pages;
587 
588  return Anchor(page, Length() );
589  }
590 
591  // case 2: The reference is of form "src:1111Filename", where "1111"
592  // points to line number 1111 in the file "Filename". KDVI then
593  // looks for source specials of the form "src:xxxxFilename", and
594  // tries to find the special with the biggest xxxx
595  if (reference.indexOf("src:", 0, Qt::CaseInsensitive) == 0) {
596 
597  // Extract the file name and the numeral part from the reference string
598  DVI_SourceFileSplitter splitter(reference, dviFile->filename);
599  quint32 refLineNumber = splitter.line();
600  QString refFileName = splitter.filePath();
601 
602  if (sourceHyperLinkAnchors.isEmpty()) {
603  emit warning(i18n("You have asked Okular to locate the place in the DVI file which corresponds to "
604  "line %1 in the TeX-file %2. It seems, however, that the DVI file "
605  "does not contain the necessary source file information. ",
606  refLineNumber, refFileName), -1);
607  return Anchor();
608  }
609 
610  // Go through the list of source file anchors, and find the anchor
611  // whose line number is the biggest among those that are smaller
612  // than the refLineNumber. That way, the position in the DVI file
613  // which is highlighted is always a little further up than the
614  // position in the editor, e.g. if the DVI file contains
615  // positional information at the beginning of every paragraph,
616  // KDVI jumps to the beginning of the paragraph that the cursor is
617  // in, and never to the next paragraph. If source file anchors for
618  // the refFileName can be found, but none of their line numbers is
619  // smaller than the refLineNumber, the reason is most likely, that
620  // the cursor in the editor stands somewhere in the preamble of
621  // the LaTeX file. In that case, we jump to the beginning of the
622  // document.
623  bool anchorForRefFileFound = false; // Flag that is set if source file anchors for the refFileName could be found at all
624 
625  QVector<DVI_SourceFileAnchor>::iterator bestMatch = sourceHyperLinkAnchors.end();
626  QVector<DVI_SourceFileAnchor>::iterator it;
627  for( it = sourceHyperLinkAnchors.begin(); it != sourceHyperLinkAnchors.end(); ++it )
628  if (refFileName.trimmed() == it->fileName.trimmed()
629  || refFileName.trimmed() == it->fileName.trimmed() + ".tex"
630  ) {
631  anchorForRefFileFound = true;
632 
633  if ( (it->line <= refLineNumber) &&
634  ( (bestMatch == sourceHyperLinkAnchors.end()) || (it->line > bestMatch->line) ) )
635  bestMatch = it;
636  }
637 
638  if (bestMatch != sourceHyperLinkAnchors.end())
639  return Anchor(bestMatch->page, bestMatch->distance_from_top);
640  else
641  if (anchorForRefFileFound == false)
642  {
643  emit warning(i18n("Okular was not able to locate the place in the DVI file which corresponds to "
644  "line %1 in the TeX-file %2.", refLineNumber, refFileName), -1);
645  }
646  else
647  return Anchor();
648  return Anchor();

684 QString dviRenderer::PDFencodingToQString(const QString& _pdfstring)
685 {
686  // This method locates special PDF characters in a string and
687  // replaces them by UTF8. See Section 3.2.3 of the PDF reference
688  // guide for information.
689  QString pdfstring = _pdfstring;
690  pdfstring = pdfstring.replace("\\n", "\n");
691  pdfstring = pdfstring.replace("\\r", "\n");
692  pdfstring = pdfstring.replace("\\t", "\t");
693  pdfstring = pdfstring.replace("\\f", "\f");
694  pdfstring = pdfstring.replace("\\(", "(");
695  pdfstring = pdfstring.replace("\\)", ")");
696  pdfstring = pdfstring.replace("\\\\", "\\");
697 
698  // Now replace octal character codes with the characters they encode
699  int pos;
700  QRegExp rx( "(\\\\)(\\d\\d\\d)" ); // matches "\xyz" where x,y,z are numbers
701  while((pos = rx.indexIn( pdfstring )) != -1) {
702  pdfstring = pdfstring.replace(pos, 4, QChar(rx.cap(2).toInt(0,8)));
703  }
704  rx.setPattern( "(\\\\)(\\d\\d)" ); // matches "\xy" where x,y are numbers
705  while((pos = rx.indexIn( pdfstring )) != -1) {
706  pdfstring = pdfstring.replace(pos, 3, QChar(rx.cap(2).toInt(0,8)));
707  }
708  rx.setPattern( "(\\\\)(\\d)" ); // matches "\x" where x is a number
709  while((pos = rx.indexIn( pdfstring )) != -1) {
710  pdfstring = pdfstring.replace(pos, 4, QChar(rx.cap(2).toInt(0,8)));
711  }
712  return pdfstring;
713 }
