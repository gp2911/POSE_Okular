78 QString MobiDocument::fixMobiMarkup(const QString& data)
79 {
80  QString ret=data;
81  QMap<int,QString> anchorPositions;
82  static QRegExp anchors("<a(?: href=\"[^\"]*\"){0,1}[\\s]+filepos=['\"]{0,1}([\\d]+)[\"']{0,1}", Qt::CaseInsensitive);
83  int pos=0;
84 
85  // find all link destinations
86  while ( (pos=anchors.indexIn( data, pos ))!=-1) {
87  int filepos=anchors.cap( 1 ).toUInt( );
88  if (filepos) anchorPositions[filepos]=anchors.cap(1);
89  pos+=anchors.matchedLength();
90  }
91 
92  // put HTML anchors in all link destinations
93  int offset=0;
94  QMapIterator<int,QString> it(anchorPositions);
95  while (it.hasNext()) {
96  it.next();
97  // link pointing outside the document, ignore
98  if ( (it.key()+offset) >= ret.size()) continue;
99  int fixedpos=outsideTag(ret, it.key()+offset);
100  ret.insert(fixedpos,QString("<a name=\"")+it.value()+QString("\">&nbsp;</a>"));
101  // inserting anchor shifts all offsets after the anchor
102  offset+=21+it.value().size();
103  }
104 
105  // replace links referencing filepos with normal internal links
106  ret.replace(anchors,"<a href=\"#\\1\"");
107  // Mobipocket uses strange variang of IMG tags: <img recindex="3232"> where recindex is number of
108  // record containing image

110

113  ret.replace("<mbp:pagebreak/>","<p style=\"page-break-after:always\"></p>");
