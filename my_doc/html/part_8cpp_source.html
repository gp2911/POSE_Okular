185 static QString compressedMimeFor( const QString& mime_to_check )
186 {
187  // The compressedMimeMap is here in case you have a very old shared mime database
188  // that doesn't have inheritance info for things like gzeps, etc
189  // Otherwise the "is()" calls below are just good enough
190  static QHash< QString, QString > compressedMimeMap;
191  static bool supportBzip = false;
192  static bool supportXz = false;
193  const QString app_gzip( QString::fromLatin1( "application/x-gzip" ) );
194  const QString app_bzip( QString::fromLatin1( "application/x-bzip" ) );
195  const QString app_xz( QString::fromLatin1( "application/x-xz" ) );
196  if ( compressedMimeMap.isEmpty() )
197  {
198  std::auto_ptr< KFilterBase > f;
199  compressedMimeMap[ QString::fromLatin1( "image/x-gzeps" ) ] = app_gzip;
200  // check we can read bzip2-compressed files
201  f.reset( KFilterBase::findFilterByMimeType( app_bzip ) );
202  if ( f.get() )
203  {
204  supportBzip = true;
205  compressedMimeMap[ QString::fromLatin1( "application/x-bzpdf" ) ] = app_bzip;
206  compressedMimeMap[ QString::fromLatin1( "application/x-bzpostscript" ) ] = app_bzip;
207  compressedMimeMap[ QString::fromLatin1( "application/x-bzdvi" ) ] = app_bzip;
208  compressedMimeMap[ QString::fromLatin1( "image/x-bzeps" ) ] = app_bzip;
209  }
210  // check we can read XZ-compressed files
211  f.reset( KFilterBase::findFilterByMimeType( app_xz ) );
212  if ( f.get() )
213  {
214  supportXz = true;
215  }
216  }
217  QHash< QString, QString >::const_iterator it = compressedMimeMap.constFind( mime_to_check );
218  if ( it != compressedMimeMap.constEnd() )
219  return it.value();
220 
221  KMimeType::Ptr mime = KMimeType::mimeType( mime_to_check );
222  if ( mime )
223  {
224  if ( mime->is( app_gzip ) )
225  return app_gzip;
226  else if ( supportBzip && mime->is( app_bzip ) )
227  return app_bzip;
228  else if ( supportXz && mime->is( app_xz ) )
229  return app_xz;
230  }
231 
232  return QString();
233 }

239  if ( parent
240  && ( parent->objectName() == QLatin1String( "okular::Shell" )
241  || parent->objectName() == QLatin1String( "okular/okular__Shell" ) ) )
242  return Okular::NativeShellMode;

246  return Okular::KHTMLPartMode;

251  {
252  if ( arg.toString() == QLatin1String( "Print/Preview" ) )

257  {
258  return Okular::ViewerWidgetMode;
259  }
260  }
261  }
262 
263  return Okular::UnknownEmbedMode;

1140 QString Part::documentMetaData( const QString &metaData ) const
1141 {
1142  const Okular::DocumentInfo * info = m_document->documentInfo();
1143  if ( info )
1144  {
1145  QDomElement docElement = info->documentElement();
1146  for ( QDomNode node = docElement.firstChild(); !node.isNull(); node = node.nextSibling() )
1147  {
1148  const QDomElement element = node.toElement();

1569 void Part::slotFileDirty( const QString& path )
1570 {

1572  // This means that timeout() is only fired when there have
1573  // no changes to the file for the last 750 milisecs.
1574  // This ensures that we don't update on every other byte that gets
1575  // written to the file.
1576  if ( path == localFilePath() )
1577  {
1578  // Only start watching the file in case if it wasn't removed
1579  if (QFile::exists(localFilePath()))

1581  else
1582  m_fileWasRemoved = true;

1584  else
1585  {

1588  {
1589  // Our parent has been dirtified
1590  if (!QFile::exists(localFilePath()))
1591  {
1592  m_fileWasRemoved = true;

1597  m_watcher->removeFile(localFilePath());
1598  m_watcher->addFile(localFilePath());
1599  m_dirtyHandler->start( 750 );

1603  {
1604  if ( QFile::exists( fi.readLink() ))
