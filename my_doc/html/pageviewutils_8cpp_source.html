134 void PageViewItem::setWHZC( int w, int h, double z, const Okular:: NormalizedRect & c )
135 {
136  m_croppedGeometry.setWidth( w );
137  m_croppedGeometry.setHeight( h );
138  m_zoomFactor = z;
139  m_crop = c;
140  m_uncroppedGeometry.setWidth( qRound( w / ( c.right - c.left ) ) );
141  m_uncroppedGeometry.setHeight( qRound( h / ( c.bottom - c.top ) ) );
142  foreach(FormWidgetIface *fwi, m_formWidgets)
143  {
144  Okular::NormalizedRect r = fwi->rect();
145  fwi->setWidthHeight(
146  qRound( fabs( r.right - r.left ) * m_uncroppedGeometry.width() ),
147  qRound( fabs( r.bottom - r.top ) * m_uncroppedGeometry.height() ) );
148  }
149  Q_FOREACH ( VideoWidget *vw, m_videoWidgets )
150  {
151  const Okular::NormalizedRect r = vw->normGeometry();
152  vw->resize(
153  qRound( fabs( r.right - r.left ) * m_uncroppedGeometry.width() ),
154  qRound( fabs( r.bottom - r.top ) * m_uncroppedGeometry.height() ) );
155  }
156 }

295 QRect PageViewMessage::computeTextRect( const QString & message, int extra_width ) const
296 // Return the QRect which embeds the text
297 {
298  int charSize = fontMetrics().averageCharWidth();
299  /* width of the viewport, minus 20 (~ size removed by further resizing),
300  minus the extra size (usually the icon width), minus (a bit empirical)
301  twice the mean width of a character to ensure that the bounding box is
302  really smaller than the container.
303  */
304  const int boundingWidth = qobject_cast<QAbstractScrollArea*>(parentWidget())->viewport()->width() - 20 - ( extra_width > 0 ? 2 + extra_width : 0 ) - 2*charSize;
305  QRect textRect = fontMetrics().boundingRect( 0, 0, boundingWidth, 0,
306  Qt::AlignLeft | Qt::AlignTop | Qt::TextWordWrap, message );
307  textRect.translate( -textRect.left(), -textRect.top() );
308  textRect.adjust( 0, 0, 2, 2 );
309 
310  return textRect;
311 }

313 void PageViewMessage::computeSizeAndResize()
314 {
315  // determine text rectangle
316  const QRect textRect = computeTextRect( m_message, m_symbol.width() );
317  int width = textRect.width(),
318  height = textRect.height();
319 
320  if ( !m_details.isEmpty() )
321  {
322  // determine details text rectangle
323  const QRect detailsRect = computeTextRect( m_details, m_symbol.width() );
324  width = qMax( width, detailsRect.width() );
325  height += detailsRect.height();
326 
327  // plus add a ~60% line spacing
328  m_lineSpacing = static_cast< int >( fontMetrics().height() * 0.6 );
329  height += m_lineSpacing;
330  }
331 
332  // update geometry with icon information
333  if ( ! m_symbol.isNull() )
334  {
335  width += 2 + m_symbol.width();
336  height = qMax( height, m_symbol.height() );
337  }
338 
339  // resize widget
340  resize( QRect( 0, 0, width + 10, height + 8 ).size() );
341 
342  // if the layout is RtL, we can move it to the right place only after we
343  // know how much size it will take
344  if ( layoutDirection() == Qt::RightToLeft )
345  move( parentWidget()->width() - geometry().width() - 10 - 1, 10 );
346 }

349 bool PageViewMessage::eventFilter(QObject * obj, QEvent * event )
350 {
351  /* if the parent object (scroll area) resizes, the message should
352  resize as well */
353  if (event->type() == QEvent::Resize)
354  {
355  QResizeEvent *resizeEvent = static_cast<QResizeEvent *>(event);

365 void PageViewMessage::paintEvent( QPaintEvent * /* e */ )
366 {
367  const QRect textRect = computeTextRect( m_message, m_symbol.width() );
368 
369  QRect detailsRect;
370  if ( !m_details.isEmpty() )
371  {
372  detailsRect = computeTextRect( m_details, m_symbol.width() );
373  }
374 
375  int textXOffset = 0,
376  // add 2 to account for the reduced drawRoundRect later
377  textYOffset = ( geometry().height() - textRect.height() - detailsRect.height() - m_lineSpacing + 2 ) / 2,
378  iconXOffset = 0,
379  iconYOffset = !m_symbol.isNull() ? ( geometry().height() - m_symbol.height() ) / 2 : 0,
380  shadowOffset = 1;
381 
382  if ( layoutDirection() == Qt::RightToLeft )
383  iconXOffset = 2 + textRect.width();
384  else
385  textXOffset = 2 + m_symbol.width();
386 
387  // draw background
388  QPainter painter( this );
389  painter.setRenderHint( QPainter::Antialiasing, true );
390  painter.setPen( Qt::black );
391  painter.setBrush( palette().color( QPalette::Window ) );
392  painter.translate( 0.5, 0.5 );
393  painter.drawRoundRect( 1, 1, width()-2, height()-2, 1600 / width(), 1600 / height() );
394 
395  // draw icon if present
396  if ( !m_symbol.isNull() )
397  painter.drawPixmap( 5 + iconXOffset, iconYOffset, m_symbol, 0, 0, m_symbol.width(), m_symbol.height() );
398 
399  const int xStartPoint = 5 + textXOffset;
400  const int yStartPoint = textYOffset;
401  const int textDrawingFlags = Qt::AlignLeft | Qt::AlignTop | Qt::TextWordWrap;
402 
403  // draw shadow and text
404  painter.setPen( palette().color( QPalette::Window ).dark( 115 ) );
405  painter.drawText( xStartPoint + shadowOffset, yStartPoint + shadowOffset, textRect.width(), textRect.height(), textDrawingFlags, m_message );
406  if ( !m_details.isEmpty() )
407  painter.drawText( xStartPoint + shadowOffset, yStartPoint + textRect.height() + m_lineSpacing + shadowOffset, textRect.width(), detailsRect.height(), textDrawingFlags, m_details );
408  painter.setPen( palette().color( QPalette::WindowText ) );
409  painter.drawText( xStartPoint, yStartPoint, textRect.width(), textRect.height(), textDrawingFlags, m_message );
410  if ( !m_details.isEmpty() )
411  painter.drawText( xStartPoint + shadowOffset, yStartPoint + textRect.height() + m_lineSpacing, textRect.width(), detailsRect.height(), textDrawingFlags, m_details );
412 }

713  return;
714

727 void ToolBarPrivate::buildToolBar()
728 {
729  int buttonsNumber = buttons.count(),
730  parentWidth = anchorWidget->width(),
731  parentHeight = anchorWidget->height(),
732  myCols = 1,
733  myRows = 1;
734 
735  // 1. find out columns and rows we're going to use
736  bool topLeft = anchorSide == PageViewToolBar::Left || anchorSide == PageViewToolBar::Top;
737  bool vertical = anchorSide == PageViewToolBar::Left || anchorSide == PageViewToolBar::Right;
738  if ( vertical )
739  {
740  myCols = 1 + (buttonsNumber * toolBarGridSize) /
741  (parentHeight - toolBarGridSize);
742  myRows = (int)ceil( (float)buttonsNumber / (float)myCols );
743  }
744  else
745  {
746  myRows = 1 + (buttonsNumber * toolBarGridSize) /
747  (parentWidth - toolBarGridSize);
748  myCols = (int)ceil( (float)buttonsNumber / (float)myRows );
749  }
750 
751  // 2. compute widget size (from rows/cols)
752  int myWidth = myCols * toolBarGridSize,
753  myHeight = myRows * toolBarGridSize,
754  xOffset = (toolBarGridSize - ToolBarButton::buttonSize) / 2,
755  yOffset = (toolBarGridSize - ToolBarButton::buttonSize) / 2;
756 
757  if ( vertical )
758  {
759  myHeight += 16;
760  myWidth += 4;
761  yOffset += 12;
762  if ( anchorSide == PageViewToolBar::Right )
763  xOffset += 4;
764  }
765  else
766  {
767  myWidth += 16;
768  myHeight += 4;
769  xOffset += 12;
770  if ( anchorSide == PageViewToolBar::Bottom )
771  yOffset += 4;
772  }
773 
774  bool prevUpdates = q->updatesEnabled();
775  q->setUpdatesEnabled( false );
776 
777  // 3. resize pixmap, mask and widget
778  QBitmap mask( myWidth + 1, myHeight + 1 );
779  backgroundPixmap = QPixmap( myWidth + 1, myHeight + 1 );
780  backgroundPixmap.fill(Qt::transparent);
781  q->resize( myWidth + 1, myHeight + 1 );
782 
783  // 4. create and set transparency mask // 4. draw background
784  QPainter maskPainter( &mask);
785  mask.fill( Qt::white );
786  maskPainter.setBrush( Qt::black );
787  if ( vertical )
788  maskPainter.drawRoundRect( topLeft ? -10 : 0, 0, myWidth + 11, myHeight, 2000 / (myWidth + 10), 2000 / myHeight );
789  else
790  maskPainter.drawRoundRect( 0, topLeft ? -10 : 0, myWidth, myHeight + 11, 2000 / myWidth, 2000 / (myHeight + 10) );
791  maskPainter.end();
792  q->setMask( mask );
793 
794  // 5. draw background
795  QPainter bufferPainter( &backgroundPixmap );
796  bufferPainter.translate( 0.5, 0.5 );
797  QPalette pal = q->palette();
798  // 5.1. draw horizontal/vertical gradient
799  QLinearGradient grad;
800  switch ( anchorSide )
801  {
802  case PageViewToolBar::Left:
803  grad = QLinearGradient( 0, 1, myWidth + 1, 1 );
804  break;
805  case PageViewToolBar::Right:
806  grad = QLinearGradient( myWidth + 1, 1, 0, 1 );
807  break;
808  case PageViewToolBar::Top:
809  grad = QLinearGradient( 1, 0, 1, myHeight + 1 );
810  break;
811  case PageViewToolBar::Bottom:
812  grad = QLinearGradient( 1, myHeight + 1, 0, 1 );
813  break;
814  }
815  grad.setColorAt( 0, pal.color( QPalette::Active, QPalette::Button ) );
816  grad.setColorAt( 1, pal.color( QPalette::Active, QPalette::Light ) );
817  bufferPainter.setBrush( QBrush( grad ) );
818  // 5.2. draw rounded border
819  bufferPainter.setPen( pal.color( QPalette::Active, QPalette::Dark ).lighter( 140 ) );
820  bufferPainter.setRenderHints( QPainter::Antialiasing );
821  if ( vertical )
822  bufferPainter.drawRoundRect( topLeft ? -10 : 0, 0, myWidth + 10, myHeight, 2000 / (myWidth + 10), 2000 / myHeight );
823  else
824  bufferPainter.drawRoundRect( 0, topLeft ? -10 : 0, myWidth, myHeight + 10, 2000 / myWidth, 2000 / (myHeight + 10) );
825  // 5.3. draw handle
826  bufferPainter.translate( -0.5, -0.5 );
827  bufferPainter.setPen( pal.color( QPalette::Active, QPalette::Mid ) );
828  if ( vertical )
829  {
830  int dx = anchorSide == PageViewToolBar::Left ? 2 : 4;
831  bufferPainter.drawLine( dx, 6, dx + myWidth - 8, 6 );
832  bufferPainter.drawLine( dx, 9, dx + myWidth - 8, 9 );
833  bufferPainter.setPen( pal.color( QPalette::Active, QPalette::Light ) );
834  bufferPainter.drawLine( dx + 1, 7, dx + myWidth - 7, 7 );
835  bufferPainter.drawLine( dx + 1, 10, dx + myWidth - 7, 10 );
836  }
837  else
838  {
839  int dy = anchorSide == PageViewToolBar::Top ? 2 : 4;
840  bufferPainter.drawLine( 6, dy, 6, dy + myHeight - 8 );
841  bufferPainter.drawLine( 9, dy, 9, dy + myHeight - 8 );
842  bufferPainter.setPen( pal.color( QPalette::Active, QPalette::Light ) );
843  bufferPainter.drawLine( 7, dy + 1, 7, dy + myHeight - 7 );
844  bufferPainter.drawLine( 10, dy + 1, 10, dy + myHeight - 7 );
845  }
846  bufferPainter.end();
847 
848  // 6. reposition buttons (in rows/col grid)
849  int gridX = 0,
850  gridY = 0;
851  QLinkedList< ToolBarButton * >::const_iterator it = buttons.begin(), end = buttons.end();
852  for ( ; it != end; ++it )
853  {
854  ToolBarButton * button = *it;
855  button->move( gridX * toolBarGridSize + xOffset,
856  gridY * toolBarGridSize + yOffset );
857  button->show();
858  if ( ++gridX == myCols )
859  {
860  gridX = 0;
861  gridY++;
862  }
863  }
864 
865  q->setUpdatesEnabled( prevUpdates );
866 }
