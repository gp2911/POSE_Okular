51 inline static void mergeResults ( LCHMSearchProgressResults & results, const LCHMSearchProgressResults & src, bool add )
52 {
53  if ( results.empty() && add )
54  {

57  }
58

60  {
61  bool found = false;
62 
63  for ( int s2 = 0; s2 < src.size(); s2++ )
64  {
65  if ( results[s1].urloff == src[s2].urloff )
66  {
67  found = true;
68  break;
69  }
70  }
71 
72  // If we're adding, we only add the items found (i.e. any item, which is not found, is removed.

74  if ( (found && !add) || (!found && add) )
75  {
76  results.erase ( results.begin() + s1 );
77  s1--;
78  }

83 static inline void findNextWords ( QVector<uint64_t> & src, const QVector<uint64_t> & needle )
84 {
85  for ( int s1 = 0; s1 < src.size(); s1++ )
86  {
87  bool found = false;
88  uint64_t target_offset = src[s1] + 1;
89 
90  DEBUG_SEARCH (("Offset loop: offset at %u is %u, target %u", (unsigned int) s1,
91  (unsigned int) src[s1], (unsigned int) target_offset));
92 
93  // Search in the offsets list in attempt to find next word
94  for ( int s2 = 0; s2 < needle.size(); s2++ )
95  {
96  if ( needle[s2] == target_offset )
97  {
98  found = true;
99  break;
100  }
101  }
102 
103  if ( !found )
104  {
105  // Remove this offset, we don't need it anymore
106  DEBUG_SEARCH (("Offset loop failed: offset %u not found", (unsigned int) target_offset));
107  src.erase ( src.begin() + s1 );
108  s1--;
109  }
110  else
111  {
112  DEBUG_SEARCH (("Offset loop succeed: offset %u found", (unsigned int) target_offset));
113  src[s1]++;
114  }
115  }
116 }

119 inline bool searchPhrase( LCHMFileImpl * impl, const QStringList & phrase, LCHMSearchProgressResults & results )
120 {
121  // Accumulate the phrase data here.
122  LCHMSearchProgressResults phrasekeeper;
123 
124  // On the first word, just fill the phrasekeeper with every occupence of the first word
125  DEBUG_SEARCH (("Search word(0): '%s'", phrase[0].ascii()));
126  if ( !impl->searchWord ( phrase[0], true, false, phrasekeeper, true ) )
127  return false; // the word not found, so the whole phrase is not found either.
128 
129  for ( int i = 1; i < phrase.size(); i++ )
130  {
131  LCHMSearchProgressResults srchtmp;
132 
133  DEBUG_SEARCH (("Search word(%d): '%s'", i, phrase[i].ascii()));
134  if ( !impl->searchWord ( phrase[i], true, false, srchtmp, true ) )
135  return false; // the ith word not found, so the whole phrase is not found either.
136 
137  // Iterate the both arrays, and remove every word in phrasekeeper, which is not found
138  // in the srchtmp, or is found on a different position.
139  for ( int p1 = 0; p1 < phrasekeeper.size(); p1++ )
140  {
141  bool found = false;
142 
143  DEBUG_SEARCH (("Ext loop (it %d): urloff %d", p1, phrasekeeper[p1].urloff));
144 
145  for ( int p2 = 0; p2 < srchtmp.size(); p2++ )
146  {
147  // look up for words on the same page
148  if ( srchtmp[p2].urloff != phrasekeeper[p1].urloff )
149  continue;
150 
151  // Now check every offset to find the one which is 1 bigger than the
152  findNextWords ( phrasekeeper[p1].offsets, srchtmp[p2].offsets );
153 
154  // If at least one next word is found, we leave the block intact, otherwise remove it.
155  if ( !phrasekeeper[p1].offsets.empty() )
156  found = true;
157  }
158 
159  if ( !found )
160  {
161  DEBUG_SEARCH (("Ext loop: this word not found on %d, remove it", phrasekeeper[p1].urloff));
162  phrasekeeper.erase ( phrasekeeper.begin() + p1 );
163  p1--;
164  }
165  }
166  }
167 
168  for ( int o = 0; o < phrasekeeper.size(); o++ )
169  results.push_back ( LCHMSearchProgressResult (phrasekeeper[o].titleoff, phrasekeeper[o].urloff) );
170 
171  return !results.empty();
172 }

176 bool LCHMFile::searchQuery( const QString& inquery, QStringList * searchresults, unsigned int limit )
177 {
178  QStringList words_must_exist, words_must_not_exist, words_highlight;
179  QVector<QStringList> phrases_must_exist;
180  QString query = inquery;
181  bool query_valid = true;
182  LCHMSearchProgressResults results;
183  int pos;
184  int i;
185 
186  /*
187  * Parse the search query with a simple state machine.
188  * Query should consist of one of more words separated by a space with a possible prefix.
189  * A prefix may be:
190  * + indicates that the word is required; any page without this word is excluded from the result.
191  * - indicates that the word is required to be absent; any page with this word is excluded from
192  * the result.
193  * "." indicates a phrase. Anything between quotes indicates a phrase, which is set of space-separated
194  * words. Will be in result only if the words in phrase are in page in the same sequence, and
195  * follow each other.
196  * If there is no prefix, the word considered as required.
197  */
198 
199  QRegExp rxphrase( "\"(.*)\"" );
200  QRegExp rxword( "([^\\s]+)" );
201  rxphrase.setMinimal( TRUE );
202 
203  // First, get the phrase queries
204  while ( (pos = rxphrase.indexIn (query, 0)) != -1 )
205  {
206  // A phrase query found. Locate its boundaries, and parse it.
207  QStringList plist = rxphrase.cap ( 1 ).split ( QRegExp ("\\s+") );
208 
209  validateWords ( plist, query_valid );
210 
211  if ( plist.size() > 0 )
212  phrases_must_exist.push_back( plist );
213 
214  query.remove (pos, rxphrase.matchedLength());
215  }
216 
217  // Then, parse the rest query
218  while ( (pos = rxword.indexIn( query, 0 )) != -1 )
219  {
220  // A phrase query found. Locate its boundaries, and parse it.
221  QString word = rxword.cap ( 1 );
222  QChar type = '+';
223 
224  if ( word[0] == '-' || word[0] == '+' )
225  {
226  type = word[0];
227  word.remove (0, 1);
228  }
229 
230  validateWord ( word, query_valid );
231 
232  if ( type == '-' )
233  words_must_not_exist.push_back ( word );
234  else
235  words_must_exist.push_back ( word );
236 
237  query.remove (pos, rxword.matchedLength());
238  }
239 
240 #if defined (DUMP_SEARCH_QUERY)
241  // Dump the search query
242  QString qdump;
243  for ( i = 0; i < phrases_must_exist.size(); i++ )
244  qdump += QString(" \"") + phrases_must_exist[i].join (" ") + QString ("\"");
245 
246  for ( i = 0; i < words_must_not_exist.size(); i++ )
247  qdump += QString (" -") + words_must_not_exist[i];
248 
249  for ( i = 0; i < words_must_exist.size(); i++ )
250  qdump += QString (" +") + words_must_exist[i];
251 
252  qDebug ("Search query dump: %s", qdump.ascii());
253 #endif
254 
255  // First search for phrases
256  if ( phrases_must_exist.size() > 0 )
257  {
258  LCHMSearchProgressResults tempres;
259 
260  for ( i = 0; i < phrases_must_exist.size(); i++ )
261  {
262  if ( !searchPhrase ( impl(), phrases_must_exist[i], tempres ) )
263  return false;
264 
265  mergeResults ( results, tempres, true );
266  }
267  }
268 
269  for ( i = 0; i < words_must_exist.size(); i++ )
270  {
271  LCHMSearchProgressResults tempres;
272 
273  if ( !m_impl->searchWord ( words_must_exist[i], true, false, tempres, false ) )
274  return false;
275 
276  mergeResults ( results, tempres, true );
277  }
278 
279  for ( i = 0; i < words_must_not_exist.size(); i++ )
280  {
281  LCHMSearchProgressResults tempres;
282 
283  m_impl->searchWord ( words_must_not_exist[i], true, false, tempres, false );
284  mergeResults ( results, tempres, false );
285  }
286 
287  m_impl->getSearchResults( results, searchresults, limit );
288  return true;
289 }
