60 void TOC::notifySetup( const QVector< Okular::Page * > & /*pages*/, int setupFlags )
61 {
62  if ( !( setupFlags & Okular::DocumentObserver::DocumentChanged ) )
63  return;
64 
65  // clear contents
66  m_model->clear();
67 
68  // request synopsis description (is a dom tree)
69  const Okular::DocumentSynopsis * syn = m_document->documentSynopsis();
70  if ( !syn )
71  {
72  if ( m_document->isOpened() )
73  {
74  // Make sure we clear the reload old model data
75  m_model->setOldModelData( 0, QVector<QModelIndex>() );
76  }
77  emit hasTOC( false );
78  return;
79  }
80 
81  m_model->fill( syn );
82  emit hasTOC( !m_model->isEmpty() );
83 }

119 QVector<QModelIndex> TOC::expandedNodes( const QModelIndex &parent ) const

145 void TOC::slotExecuted( const QModelIndex &index )
146 {
147  if ( !index.isValid() )
148  return;
149 
150  QString url = m_model->urlForIndex( index );
151  if ( !url.isEmpty() )
152  {
153  Okular::BrowseAction action( url );
154  m_document->processAction( &action );
155  return;
156  }
157 
158  QString externalFileName = m_model->externalFileNameForIndex( index );
159  Okular::DocumentViewport viewport = m_model->viewportForIndex( index );
160  if ( !externalFileName.isEmpty() )
161  {
162  Okular::GotoAction action( externalFileName, viewport );
163  m_document->processAction( &action );
164  }
165  else if ( viewport.isValid() )
166  {
167  m_document->setViewport( viewport );
168  }
169 }
