88 //The test testNextAndPrevious checks that
89 //a) if one starts a new search, then the first or last match is found, depending on the search direction
90 // (2 cases: FromTop/FromBottom)
91 //b) if the last search has found a match,
92 // then clicking the "Next" button moves to the next occurrence an "Previous" to the previous one
93 // (if there is any). Altogether there are four combinations of the last search and new search
94 // direction: Next-Next, Previous-Previous, Next-Previous, Previous-Next; the first two combination
95 // have two subcases (the new search may give a match or not, so altogether 6 cases to test).
96 //This gives 8 cases altogether. By taking into account the cases where the last search has given no match,
97 //we would have 4 more cases (Next (no match)-Next, Next (no match)-Previous, Previous (no match)-Previous,
98 //Previous (no match)-Next), but those are more the business of Okular::Document::searchText rather than
99 //Okular::TextPage (at least in the multi-page case).
100 
101 // We have four test situations: four documents and four corresponding search strings.
102 // The first situation (document="ababa", search string="b") is a generic one where the
103 //two matches are not side-by-side and neither the first character nor the last character of
104 //the document match. The only special thing is that the search string has only length 1.
105 // The second situation (document="abab", search string="ab") is notable for that the two occurrences
106 //of the search string are side-by-side with no characters in between, so some off-by-one errors
107 //would be detected by this test. As the first match starts at the beginning at the document the
108 //last match ends at the end of the document, it also detects off-by-one errors for finding the first/last match.
109 // The third situation (document="abababa", search string="aba") is notable for it shows whether
110 //the next match is allowed to contain letters from the previous one: currently it is not
111 //(as in the majority of browsers, viewers and editors), and therefore "abababa" is considered to
112 //contain not three but two occurrences of "aba" (if one starts search from the beginning of the document).
113 // The fourth situation (document="a ba b", search string="a b") demonstrates the case when one TinyTextEntity
114 //contains multiple characters that are contained in different matches (namely, the middle "ba" is one TinyTextEntity);
115 //in particular, since these matches are side-by-side, this test would detect some off-by-one
116 //offset errors.

123  QVector<QString> texts[TEST_NEXT_PREV_SITUATION_COUNT] = {

134  "a b"
135  };
136 
137  for (int i = 0; i < TEST_NEXT_PREV_SITUATION_COUNT; i++) {
138  const QVector<QString>& text = texts[i];
139  const QString& searchString = searchStrings[i];
140 
141  QVector<Okular::NormalizedRect> rect; \
142 
143  for (int i = 0; i < text.size(); i++) {
144  rect << Okular::NormalizedRect(0.1*i, 0.0, 0.1*(i+1), 0.1); \
145  }
146 
147  CREATE_PAGE;
148 
149  //Test 3 of the 8 cases listed above:
150  //FromTop, Next-Next (match) and Next-Next (no match)
151  TEST_NEXT_PREV(Okular::FromTop, true);
152  TEST_NEXT_PREV(Okular::NextResult, true);
153  TEST_NEXT_PREV(Okular::NextResult, false);
154 
155  //Test 5 cases: FromBottom, Previous-Previous (match), Previous-Next,
156  //Next-Previous, Previous-Previous (no match)
157  TEST_NEXT_PREV(Okular::FromBottom, true);
158  TEST_NEXT_PREV(Okular::PreviousResult, true);
159  TEST_NEXT_PREV(Okular::NextResult, true);
160  TEST_NEXT_PREV(Okular::PreviousResult, true);
161  TEST_NEXT_PREV(Okular::PreviousResult, false);

178

240 void SearchTest::testDottedI()
241 {
242  //Earlier versions of okular had the bug that the letter "İ" (capital dotter i) did not match itself
243  //in case-insensitive mode (this was caused by an unnecessary call of toLower() and the fact that
244  //QString::fromUtf8("İ").compare(QString::fromUtf8("İ").toLower(), Qt::CaseInsensitive) == FALSE,
245  //at least in Qt 4.8).
246 
247  //In the future it would be nice to add support for matching "İ"<->"i" and "I"<->"ı" in case-insensitive
248  //mode as well (QString::compare does not match them, at least in non-Turkish locales, since it follows
249  //the Unicode case-folding rules http://www.unicode.org/Public/6.2.0/ucd/CaseFolding.txt).
250 
251  QVector<QString> text;
252  text << QString::fromUtf8("İ");
253 
254  QVector<Okular::NormalizedRect> rect;
255  rect << Okular::NormalizedRect(1, 2, 3, 4);
256 
257  CREATE_PAGE;
258 
259  Okular::RegularAreaRect* result = tp->findText(0, QString::fromUtf8("İ"), Okular::FromTop, Qt::CaseInsensitive, NULL);
260  QVERIFY(result);
261  delete result;
262 
263  delete page;

373 void SearchTest::testOneColumn()
374 {
375  //Tests that the layout analysis algorithm does not create too many columns.
376  //Bug 326207 was caused by the fact that if all the horizontal breaks in a line
377  //had the same length and were smaller than vertical breaks between lines then
378  //the horizontal breaks were treated as column separators.
379  //(Note that "same length" means "same length after rounding rectangles to integer pixels".
380  //The resolution used by the XY Cut algorithm with a square page is 1000 x 1000,
381  //and the horizontal spaces in the example are 0.1, so they are indeed both exactly 100 pixels.)
382 
383  QVector<QString> text;
384  text << "Only" << "one" << "column"
385  << "here";
386 
387  //characters and line breaks have length 0.05, word breaks 0.1
388  QVector<Okular::NormalizedRect> rect;
389  rect << Okular::NormalizedRect(0.0, 0.0, 0.2, 0.1)
390  << Okular::NormalizedRect(0.3, 0.0, 0.5, 0.1)
391  << Okular::NormalizedRect(0.6, 0.0, 0.9, 0.1)
392  << Okular::NormalizedRect(0.0, 0.15, 0.2, 0.25);
393 
394  CREATE_PAGE;
395 
396  Okular::RegularAreaRect* result = tp->findText(0, "Only one column",
397  Okular::FromTop, Qt::CaseSensitive, NULL);
398  QVERIFY(result);
399  delete result;
