74 void Converter::_handle_anchors(const QTextBlock &start, const QString &name) {
75  const QString curDir = QFileInfo(name).path();
76 
77  for (QTextBlock bit = start; bit != mTextDocument->end(); bit = bit.next()) {
78  for (QTextBlock::iterator fit = bit.begin(); !(fit.atEnd()); ++fit) {
79 
80  QTextFragment frag = fit.fragment();
81 
82  if (frag.isValid() && frag.charFormat().isAnchor()) {
83  QString hrefString = frag.charFormat().anchorHref();
84 
85  // remove ./ or ../
86  // making it easier to compare, with links
87  while(!hrefString.isNull() && ( hrefString.at(0) == '.' || hrefString.at(0) == '/') ){
88  hrefString.remove(0,1);
89  }
90 
91  QUrl href(hrefString);
92  if (href.isValid() && !href.isEmpty()) {
93  if (href.isRelative()) { // Inside document link
94  if(!hrefString.indexOf('#'))
95  hrefString = name + hrefString;
96  else if(QFileInfo(hrefString).path() == "." && curDir != ".")
97  hrefString = curDir + '/' + hrefString;
98 
99  // QTextCharFormat sometimes splits a link in two
100  // if there's no white space between words & the first one is an anchor
101  // consider whole word to be an anchor
102  ++fit;
103  int fragLen = frag.length();
104  if(!fit.atEnd() && ((fit.fragment().position() - frag.position()) == 1))
105  fragLen += fit.fragment().length();
106  --fit;
107 
108  _insert_local_links(hrefString,
109  QPair<int, int>(frag.position(),
110  frag.position()+fragLen));
111  } else { // Outside document link
112  Okular::BrowseAction *action =
113  new Okular::BrowseAction(href.toString());
114 
115  emit addAction(action, frag.position(),
116  frag.position() + frag.length());
117  }
118  }
119 
120  const QStringList &names = frag.charFormat().anchorNames();
121  if (!names.empty()) {
122  for (QStringList::const_iterator lit = names.constBegin();
123  lit != names.constEnd(); ++lit) {
124  mSectionMap.insert(name + '#' + *lit, bit);
125  }
126  }
127 
128  } // end anchor case
129  }
130  }

144 static QPoint calculateXYPosition( QTextDocument *document, int startPosition )
145 {
146  const QTextBlock startBlock = document->findBlock( startPosition );
147  const QRectF startBoundingRect = document->documentLayout()->blockBoundingRect( startBlock );
148 
149  QTextLayout *startLayout = startBlock.layout();
150  if (!startLayout) {
151  kWarning() << "Start layout not found" << startLayout;
152  return QPoint();
153  }
154 
155  int startPos = startPosition - startBlock.position();
156  const QTextLine startLine = startLayout->lineForTextPosition( startPos );
157 
158  double x = startBoundingRect.x() ;
159  double y = startBoundingRect.y() + startLine.y();
160 
161  y = (int)y % 800;
162 
163  return QPoint(x,y);
