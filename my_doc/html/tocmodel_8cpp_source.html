111 void TOCModelPrivate::addChildren( const QDomNode & parentNode, TOCItem * parentItem )
112 {
113  TOCItem * currentItem = 0;
114  QDomNode n = parentNode.firstChild();
115  while( !n.isNull() )
116  {
117  // convert the node to an element (sure it is)
118  QDomElement e = n.toElement();
119 
120  // insert the entry as top level (listview parented) or 2nd+ level
121  currentItem = new TOCItem( parentItem, e );
122 
123  // descend recursively and advance to the next node
124  if ( e.hasChildNodes() )
125  addChildren( n, currentItem );
126 
127  // open/keep close the item
128  bool isOpen = false;
129  if ( e.hasAttribute( "Open" ) )
130  isOpen = QVariant( e.attribute( "Open" ) ).toBool();
131  if ( isOpen )
132  itemsToOpen.append( currentItem );
133 
134  n = n.nextSibling();
135  }
136 }

178 QVariant TOCModel::data( const QModelIndex &index, int role ) const
179 {
180  if ( !index.isValid() )
181  return QVariant();
182 
183  TOCItem *item = static_cast< TOCItem* >( index.internalPointer() );
184  switch ( role )
185  {
186  case Qt::DisplayRole:
187  case Qt::ToolTipRole:
188  return item->text;
189  break;
190  case Qt::DecorationRole:
191  if ( item->highlight )
192  return KIcon( QApplication::layoutDirection() == Qt::RightToLeft ? "arrow-left" : "arrow-right" );
193  break;
194  case PageItemDelegate::PageRole:
195  if ( item->viewport.isValid() )
196  return item->viewport.pageNumber + 1;
197  break;
198  case PageItemDelegate::PageLabelRole:
199  if ( item->viewport.isValid() && item->viewport.pageNumber < int(d->document->pages()) )
200  return d->document->page( item->viewport.pageNumber )->label();
201  break;
202  }
203  return QVariant();
204 }

267 void TOCModel::fill( const Okular::DocumentSynopsis *toc )
268 {
269  if ( !toc )
270  return;
271 
272  clear();
273  emit layoutAboutToBeChanged();
274  d->addChildren( *toc, d->root );
275  d->dirty = true;
276  emit layoutChanged();
277  if ( equals( d->m_oldModel ) )
278  {
279  foreach( const QModelIndex &oldIndex, d->m_oldTocExpandedIndexes )
280  {
281  const QModelIndex index = indexForIndex( oldIndex, this );
282  if ( !index.isValid() )
283  continue;
284 
285  QMetaObject::invokeMethod( QObject::parent(), "expand", Qt::QueuedConnection, Q_ARG( QModelIndex, index ) );
286  }
287  }
288  else
289  {
290  foreach ( TOCItem *item, d->itemsToOpen )
291  {
292  const QModelIndex index = d->indexForItem( item );
293  if ( !index.isValid() )
294  continue;
295 
296  QMetaObject::invokeMethod( QObject::parent(), "expand", Qt::QueuedConnection, Q_ARG( QModelIndex, index ) );
297  }
298  }
299  d->itemsToOpen.clear();
300  delete d->m_oldModel;
301  d->m_oldModel = 0;
302  d->m_oldTocExpandedIndexes.clear();
303 }

318 {

320  {

413 bool TOCModel::checkequality( const TOCModel *model, const QModelIndex & parentA, const QModelIndex & parentB ) const
414 {
415  if ( rowCount( parentA ) != model->rowCount( parentB ) )
416  return false;
417  for ( int i = 0; i < rowCount( parentA ); i++ )
418  {
419  QModelIndex indxA = index( i, 0, parentA );
420  QModelIndex indxB = model->index( i, 0, parentB );
421  if ( indxA.data() != indxB.data() )
422  {
423  return false;
424  }
425  if ( hasChildren( indxA ) != model->hasChildren( indxB ) )
426  {
427  return false;
428  }
429  if ( !checkequality( model, indxA, indxB ) )
430  {
431  return false;
432  }
433  }
434  return true;
435 }
