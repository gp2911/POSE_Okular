84 bool CHMGenerator::loadDocument( const QString & fileName, QVector< Okular::Page * > & pagesVector )
85 {
86  m_file = new LCHMFile();
87  if (!m_file->loadFile(fileName))
88  {
89  delete m_file;
90  m_file = 0;
91  return false;
92  }
93  m_fileName=fileName;
94  QVector< LCHMParsedEntry > topics;
95  m_file->parseTableOfContents(&topics);
96 
97  // fill m_docSyn
98  QMap<int, QDomElement> lastIndentElement;
99  foreach(const LCHMParsedEntry &e, topics)
100  {
101  QDomElement item = m_docSyn.createElement(e.name);
102  if (!e.urls.isEmpty())
103  {
104  QString url = e.urls.first();
105  if (url.contains(QChar::fromLatin1('%')))
106  url = QString::fromUtf8(QByteArray::fromPercentEncoding(url.toUtf8()));
107  item.setAttribute("ViewportName", url);
108  }
109  item.setAttribute("Icon", e.imageid);
110  if (e.indent == 0) m_docSyn.appendChild(item);
111  else lastIndentElement[e.indent - 1].appendChild(item);
112  lastIndentElement[e.indent] = item;
113  }
114 
115  // fill m_urlPage and m_pageUrl
116  int pageNum = 0;
117  QStringList pageList;
118  m_file->enumerateFiles(&pageList);
119  const QString home = m_file->homeUrl();
120  if (home != QLatin1String("/"))
121  pageList.prepend(home);
122  foreach (const QString &url, pageList)
123  {
124  const QString urlLower = url.toLower();
125  if (!urlLower.endsWith(QLatin1String(".html")) && !urlLower.endsWith(QLatin1String(".htm")))
126  continue;
127 
128  int pos = url.indexOf ('#');
129  QString tmpUrl = pos == -1 ? url : url.left(pos);
130 
131  // url already there, abort insertion
132  if (m_urlPage.contains(tmpUrl)) continue;
133 
134  // insert the url into the maps, but insert always the variant without the #ref part
135  m_urlPage.insert(tmpUrl, pageNum);
136  m_pageUrl.append(tmpUrl);
137  pageNum++;
138  }
139 
140  pagesVector.resize(m_pageUrl.count());
141  m_textpageAddedList.fill(false, pagesVector.count());
142  m_rectsGenerated.fill(false, pagesVector.count());
143 
144  if (!m_syncGen)
145  {
146  m_syncGen = new KHTMLPart();
147  }
148  disconnect( m_syncGen, 0, this, 0 );
149 
150  for (int i = 0; i < m_pageUrl.count(); ++i)
151  {
152  preparePageForSyncOperation(100, m_pageUrl.at(i));
153  pagesVector[ i ] = new Okular::Page (i, m_syncGen->view()->contentsWidth(),
154  m_syncGen->view()->contentsHeight(), Okular::Rotation0 );
155  }
156 
157  connect( m_syncGen, SIGNAL(completed()), this, SLOT(slotCompleted()) );
158  connect( m_syncGen, SIGNAL(canceled(QString)), this, SLOT(slotCompleted()) );
159 
160  return true;
161 }

269  if (requestWidth<300)

288  m_syncGen->openUrl ( pAddress );

292 void CHMGenerator::recursiveExploreNodes(DOM::Node node,Okular::TextPage *tp)
293 {
294  if (node.nodeType() == DOM::Node::TEXT_NODE && !node.getRect().isNull())
295  {
296  QString nodeText=node.nodeValue().string();
297  QRect r=node.getRect();
298  int vWidth=m_syncGen->view()->width();
299  int vHeight=m_syncGen->view()->height();
300  Okular::NormalizedRect *nodeNormRect;
301 #define NOEXP
302 #ifndef NOEXP
303  int x,y,height;
304  int x_next,y_next,height_next;
305  int nodeTextLength = nodeText.length();
306  if (nodeTextLength==1)
307  {
308  nodeNormRect=new Okular::NormalizedRect (r,vWidth,vHeight);
309  tp->append(nodeText,nodeNormRect,nodeNormRect->bottom,0,(nodeText=="\n"));
310  }
311  else
312  {
313  for (int i=0;i<nodeTextLength;i++)
314  {
315  node.getCursor(i,x,y,height);
316  if (i==0)
317  // i is 0, use left rect boundary
318  {
319 // if (nodeType[i+1]
320  node.getCursor(i+1,x_next,y_next,height_next);
321  nodeNormRect=new Okular::NormalizedRect (QRect(x,y,x_next-x-1,height),vWidth,vHeight);
322  }
323  else if ( i <nodeTextLength -1 )
324  // i is between zero and the last element
325  {
326  node.getCursor(i+1,x_next,y_next,height_next);
327  nodeNormRect=new Okular::NormalizedRect (QRect(x,y,x_next-x-1,height),vWidth,vHeight);
328  }
329  else
330  // the last element use right rect boundary
331  {
332  node.getCursor(i-1,x_next,y_next,height_next);
333  }
334  }
335 #else
336  nodeNormRect=new Okular::NormalizedRect (r,vWidth,vHeight);
337  tp->append(nodeText,nodeNormRect/*,0*/);
338 #endif
339  }
340  DOM::Node child = node.firstChild();
341  while ( !child.isNull() )
342  {
343  recursiveExploreNodes(child,tp);
344  child = child.nextSibling();
345  }

348 void CHMGenerator::additionalRequestData()
349 {
350  Okular::Page * page=m_request->page();
351  const bool genObjectRects = !m_rectsGenerated.at( m_request->page()->number() );
352  const bool genTextPage = !m_request->page()->hasTextPage() && genObjectRects;
353 
354  if (genObjectRects || genTextPage )
355  {
356  DOM::HTMLDocument domDoc=m_syncGen->htmlDocument();
357  // only generate object info when generating a full page not a thumbnail
358  if ( genObjectRects )
359  {
360  QLinkedList< Okular::ObjectRect * > objRects;
361  int xScale=m_syncGen->view()->width();
362  int yScale=m_syncGen->view()->height();
363  // getting links
364  DOM::HTMLCollection coll=domDoc.links();
365  DOM::Node n;
366  QRect r;
367  if (! coll.isNull() )
368  {
369  int size=coll.length();
370  for(int i=0;i<size;i++)
371  {
372  n=coll.item(i);
373  if ( !n.isNull() )
374  {
375  QString url = n.attributes().getNamedItem("href").nodeValue().string();
376  r=n.getRect();
377  // there is no way for us to support javascript properly
378  if (url.startsWith("JavaScript:"), Qt::CaseInsensitive)
379  continue;
380  else if (url.contains (":"))
381  {
382  objRects.push_back(
383  new Okular::ObjectRect ( Okular::NormalizedRect(r,xScale,yScale),
384  false,
385  Okular::ObjectRect::Action,
386  new Okular::BrowseAction ( url )));
387  }
388  else
389  {
390  Okular::DocumentViewport viewport( metaData( "NamedViewport", absolutePath( m_chmUrl, url ) ).toString() );
391  objRects.push_back(
392  new Okular::ObjectRect ( Okular::NormalizedRect(r,xScale,yScale),
393  false,
394  Okular::ObjectRect::Action,
395  new Okular::GotoAction ( QString::null, viewport))); //krazy:exclude=nullstrassign for old broken gcc
396  }
397  }
398  }
399  }
400 
401  // getting images
402  coll=domDoc.images();
403  if (! coll.isNull() )
404  {
405  int size=coll.length();
406  for(int i=0;i<size;i++)
407  {
408  n=coll.item(i);
409  if ( !n.isNull() )
410  {
411  objRects.push_back(
412  new Okular::ObjectRect ( Okular::NormalizedRect(n.getRect(),xScale,yScale),
413  false,
414  Okular::ObjectRect::Image,
415  0));
416  }
417  }
418  }
419  m_request->page()->setObjectRects( objRects );
420  m_rectsGenerated[ m_request->page()->number() ] = true;
421  }
422 
423  if ( genTextPage )
424  {
425  Okular::TextPage *tp=new Okular::TextPage();
426  recursiveExploreNodes(domDoc,tp);
427  page->setTextPage (tp);
428  }
429  }
430 }
