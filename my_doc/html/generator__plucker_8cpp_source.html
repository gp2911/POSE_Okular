41 static void calculateBoundingRect( QTextDocument *document, int startPosition, int endPosition,
42  QRectF &rect )
43 {
44  const QTextBlock startBlock = document->findBlock( startPosition );
45  const QRectF startBoundingRect = document->documentLayout()->blockBoundingRect( startBlock );
46 
47  const QTextBlock endBlock = document->findBlock( endPosition );
48  const QRectF endBoundingRect = document->documentLayout()->blockBoundingRect( endBlock );
49 
50  QTextLayout *startLayout = startBlock.layout();
51  QTextLayout *endLayout = endBlock.layout();
52 
53  int startPos = startPosition - startBlock.position();
54  int endPos = endPosition - endBlock.position();
55  const QTextLine startLine = startLayout->lineForTextPosition( startPos );
56  const QTextLine endLine = endLayout->lineForTextPosition( endPos );
57 
58  double x = startBoundingRect.x() + startLine.cursorToX( startPos );
59  double y = startBoundingRect.y() + startLine.y();
60  double r = endBoundingRect.x() + endLine.cursorToX( endPos );
61  double b = endBoundingRect.y() + endLine.y() + endLine.height();
62 
63  const QSizeF size = document->size();
64  rect = QRectF( x / size.width(), y / size.height(),
65  (r - x) / size.width(), (b - y) / size.height() );
66 }

78 bool PluckerGenerator::loadDocument( const QString & fileName, QVector<Okular::Page*> & pagesVector )
79 {
80  QUnpluck unpluck;
81 
82  if ( !unpluck.open( fileName ) )
83  return false;
84 
85  mPages = unpluck.pages();
86  mLinks = unpluck.links();
87 
88  const QMap<QString, QString> infos = unpluck.infos();
89  QMapIterator<QString, QString> it( infos );
90  while ( it.hasNext() ) {
91  it.next();
92  if ( !it.value().isEmpty() ) {
93  if ( it.key() == QLatin1String( "name" ) )
94  mDocumentInfo.set( "name", it.value(), i18n( "Name" ) );
95  else if ( it.key() == QLatin1String( "title" ) )
96  mDocumentInfo.set( Okular::DocumentInfo::Title, it.value() );
97  else if ( it.key() == QLatin1String( "author" ) )
98  mDocumentInfo.set( Okular::DocumentInfo::Author, it.value() );
99  else if ( it.key() == QLatin1String( "time" ) )
100  mDocumentInfo.set( Okular::DocumentInfo::CreationDate, it.value() );
101  }
102  }
103 
104  pagesVector.resize( mPages.count() );
105 
106  for ( int i = 0; i < mPages.count(); ++i ) {
107  QSizeF size = mPages[ i ]->size();
108  Okular::Page * page = new Okular::Page( i, size.width(), size.height(), Okular::Rotation0 );
109  pagesVector[i] = page;
110  }
111 
112  return true;
113 }

134 QImage PluckerGenerator::image( Okular::PixmapRequest *request )
135 {
136  const QSizeF size = mPages[ request->pageNumber() ]->size();
137 
138  QImage image( request->width(), request->height(), QImage::Format_ARGB32_Premultiplied );
139  image.fill( Qt::white );
140 
141  QPainter p;
142  p.begin( &image );
143 
144  qreal width = request->width();
145  qreal height = request->height();
146 
147  p.scale( width / (qreal)size.width(), height / (qreal)size.height() );
148  mPages[ request->pageNumber() ]->drawContents( &p );
149  p.end();
150 
151  if ( !mLinkAdded.contains( request->pageNumber() ) ) {
152  QLinkedList<Okular::ObjectRect*> objects;
153  for ( int i = 0; i < mLinks.count(); ++i ) {
154  if ( mLinks[ i ].page == request->pageNumber() ) {
155  QTextDocument *document = mPages[ request->pageNumber() ];
156 
157  QRectF rect;
158  calculateBoundingRect( document, mLinks[ i ].start,
159  mLinks[ i ].end, rect );
160 
161  objects.append( new Okular::ObjectRect( rect.left(), rect.top(), rect.right(), rect.bottom(), false, Okular::ObjectRect::Action, mLinks[ i ].link ) );
162  }
163  }
164 
165  if ( !objects.isEmpty() )
166  request->page()->setObjectRects( objects );
167 
168  mLinkAdded.insert( request->pageNumber() );
169  }
170 
171  return image;
172 }
